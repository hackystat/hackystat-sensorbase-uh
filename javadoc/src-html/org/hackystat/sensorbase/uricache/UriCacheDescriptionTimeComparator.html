001    package org.hackystat.sensorbase.uricache;
002    
003    import java.io.Serializable;
004    import java.util.Comparator;
005    
006    /**
007     * Compares two caches creation time by using time saved in the description.
008     * 
009     * @author Pavel Senin.
010     * 
011     */
012    public class UriCacheDescriptionTimeComparator implements Comparator<UriCacheDescription>,
013        Serializable {
014    
015      /**
016       * fancy Eclipse serial number.
017       */
018      private static final long serialVersionUID = -2848820550774349045L;
019    
020      /**
021       * Compares two caches creation time by using time saved in the description.
022       * 
023       * @param desc1 UriCacheDescription instance #1.
024       * @param desc2 UriCacheDescription instance #2.
025       * 
026       * @return 0 if both caches created in the same tome, value greater than 0 if cache2 creation time
027       *         is less than cache #1 creation time and returns value less than 0 if cache #1 creation
028       *         time is less than cache #2 creation time.
029       */
030      public int compare(UriCacheDescription desc1, UriCacheDescription desc2) {
031        Long cache1CreationTime = desc1.getCreationTime();
032        Long cache2CreationTime = desc2.getCreationTime();
033        return cache1CreationTime.compareTo(cache2CreationTime);
034      }
035    
036    }
