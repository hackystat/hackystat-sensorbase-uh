001    package org.hackystat.sensorbase.uricache;
002    
003    import java.io.File;
004    import java.io.FilenameFilter;
005    import java.util.regex.Pattern;
006    
007    /**
008     * Filters UriCache description file names.
009     * 
010     * @author Pavel Senin.
011     * 
012     */
013    public class UriCacheDescriptionFilter implements FilenameFilter {
014    
015      /** Pattern for the .desc files. */
016      private Pattern pattern = Pattern.compile(".*\\.desc");
017    
018      /**
019       * Filters out only cache description file names.
020       * 
021       * @param dir the folder name.
022       * @param fileName the file name.
023       * 
024       * @return true if the file name is cache description.
025       */
026      public boolean accept(File dir, String fileName) {
027        return pattern.matcher(fileName).matches();
028      }
029    
030    }
