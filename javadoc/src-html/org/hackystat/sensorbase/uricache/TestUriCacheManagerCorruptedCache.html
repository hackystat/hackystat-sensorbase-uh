001    package org.hackystat.sensorbase.uricache;
002    
003    import static org.junit.Assert.assertEquals;
004    import static org.junit.Assert.assertNotNull;
005    import static org.junit.Assert.assertTrue;
006    
007    import java.io.BufferedWriter;
008    import java.io.File;
009    import java.io.FileOutputStream;
010    import java.io.FileWriter;
011    import java.io.IOException;
012    import java.util.List;
013    import java.util.Properties;
014    import java.util.TreeMap;
015    
016    import org.hackystat.utilities.uricache.UriCache;
017    import org.hackystat.utilities.uricache.UriCacheException;
018    import org.junit.After;
019    import org.junit.Before;
020    import org.junit.Test;
021    
022    /**
023     * Tests CacheManager functionality if the cache data is corrupted. This situation could happen when
024     * system is restarted after crash.
025     * 
026     * @author Pavel Senin.
027     * 
028     */
029    public class TestUriCacheManagerCorruptedCache {
030    
031      private static final String fileSeparator = System.getProperty("file.separator");
032    
033      /** Used for temporarily caches home */
034      private static final String tmpFolderName = String.valueOf(System.currentTimeMillis());
035      /** The general storage place. */
036      private static final String cacheHome = System.getProperties().getProperty("user.dir")
037          + fileSeparator + "build" + fileSeparator + "uricache-tests" + fileSeparator + tmpFolderName;
038    
039      /** Test cache and files names. */
040      private static final String testCacheName = "TestCorruptedCache";
041      private static final String testCacheDescName = testCacheName + ".desc";
042      private static final String testCacheDataName = testCacheName + ".data";
043      private static final String testCacheKeyName = testCacheName + ".key";
044    
045      /** User e-mail key. */
046      private static final String USER_EMAIL_KEY = "uricache.user.email";
047      /** User e-mail */
048      private static final String userEmail = "javadude@javatesthost.org";
049    
050      /** The host key. */
051      private static final String HOST_KEY = "uricache.host";
052      /** User host key. */
053      private static final String sensorBaseHost = "http://sensorbase143.javatesthost.org:20910";
054    
055      private static final String key = "key:";
056      private static final String data = "data:";
057    
058      /**
059       * Sets up test with the temporarily test caches.
060       * 
061       * @throws Exception if unable to proceed.
062       */
063      @Before
064      public void setUp() throws Exception {
065        // setting up the test area
066        File f = new File(cacheHome);
067        if (!f.exists()) {
068          f.mkdirs();
069        }
070    
071        // creating description file
072        String fName = cacheHome + fileSeparator + testCacheDescName;
073        Properties prop = new Properties();
074        prop.setProperty(USER_EMAIL_KEY, userEmail);
075        prop.setProperty(HOST_KEY, sensorBaseHost);
076        FileOutputStream stream = null;
077        stream = new FileOutputStream(fName);
078        prop.store(stream, "the UriCache properties test file");
079        stream.close();
080    
081        // populating junk into .data file
082        fName = cacheHome + fileSeparator + testCacheDataName;
083        BufferedWriter bw = new BufferedWriter(new FileWriter(new File(fName)));
084        for (int i = 0; i < 5000; i++) {
085          bw.write("number:" + i);
086        }
087        bw.close();
088    
089        // populating junk into .key file
090        fName = cacheHome + fileSeparator + testCacheKeyName;
091        bw = new BufferedWriter(new FileWriter(new File(fName)));
092        for (int i = 5000; i > 0; i--) {
093          bw.write("number:" + i);
094        }
095        bw.close();
096    
097      }
098    
099      /**
100       * Tests the corrupted cache processing.
101       * 
102       * @throws UriCacheException in the case of error.
103       * @throws IOException if IO error occurs.
104       * @throws InterruptedException if thread unable to sleep.
105       */
106      @Test
107      public void testCorruptedCache() throws UriCacheException, IOException, InterruptedException {
108        // get list of caches
109        List<UriCacheDescription> caches = UriCacheManager.getCaches(cacheHome);
110        assertEquals("Should find one cache only", 1, caches.size());
111        // put caches in map to ease the test
112        TreeMap<String, UriCacheDescription> cachesMap = new TreeMap<String, UriCacheDescription>();
113        for (UriCacheDescription d : caches) {
114          cachesMap.put(d.getName(), d);
115        }
116    
117        // now run test of description - should pass
118        assertTrue("Should report right e-mail", userEmail.equalsIgnoreCase(cachesMap
119            .get(testCacheName).getUserEmail()));
120        assertTrue("Should report right host", sensorBaseHost.equalsIgnoreCase(cachesMap.get(
121            testCacheName).getsensorBaseHost()));
122    
123        // get cache
124        UriCache<String, Object> cache;
125        cache = UriCacheManager.getCache(cacheHome, sensorBaseHost, userEmail);
126        //
127        // Should pass this point without any exception and reset the cache
128        //
129        // now load darta into cache
130        for (int i = 0; i < 500; i++) {
131          cache.cache(key + i, data + i);
132        }
133        //
134        // and finally should be ABLE to read data back
135        for (int i = 0; i < 500; i++) {
136          String element = (String) cache.lookup(key + i);
137          assertNotNull("Should have recevied an element. " + i, element);
138          assertEquals("Element is wrong.", data + i, element);
139        }
140        cache.shutdown();
141    
142        //
143        // now just for fun try to test persistence
144        //
145        Thread.yield();
146        Thread.sleep(1000);
147        Thread.yield();
148        cache = UriCacheManager.getCache(cacheHome, sensorBaseHost, userEmail);
149        // should be ABLE to read data back
150        for (int i = 0; i < 500; i++) {
151          String element = (String) cache.lookup(key + i);
152          assertNotNull("Should have recevied an element. " + i, element);
153          assertEquals("Element is wrong.", data + i, element);
154        }
155        cache.shutdown();
156      }
157    
158      /**
159       * Tears down test environment and clears all leftovers.
160       * 
161       * @throws Exception if error encountered.
162       */
163      @After
164      public void tearDown() throws Exception {
165        // remove files
166        String fName = cacheHome + fileSeparator + testCacheDescName;
167        File f = new File(fName);
168        f.delete();
169    
170        fName = cacheHome + fileSeparator + testCacheDataName;
171        f = new File(fName);
172        f.delete();
173    
174        fName = cacheHome + fileSeparator + testCacheKeyName;
175        f = new File(fName);
176        f.delete();
177    
178        // remove folder
179        f = new File(cacheHome);
180        if (f.exists()) {
181          f.delete();
182        }
183      }
184
