001    package org.hackystat.sensorbase.test;
002    
003    import org.hackystat.sensorbase.resource.projects.ProjectManager;
004    import org.hackystat.sensorbase.resource.sensordata.SensorDataManager;
005    import org.hackystat.sensorbase.resource.sensordatatypes.SdtManager;
006    import org.hackystat.sensorbase.resource.users.UserManager;
007    import org.hackystat.sensorbase.server.Server;
008    import org.hackystat.sensorbase.server.ServerProperties;
009    
010    import static org.hackystat.sensorbase.server.ServerProperties.ADMIN_EMAIL_KEY;
011    import static org.hackystat.sensorbase.server.ServerProperties.ADMIN_PASSWORD_KEY;
012    import org.junit.BeforeClass;
013    
014    
015    /**
016     * Provides helpful utility methods to SensorBase test classes, which will
017     * normally want to extend this class. 
018     * @author Philip Johnson
019     *
020     */
021    public class SensorBaseRestApiHelper {
022      /** The SensorBase server used in these tests. */
023      protected static Server server;
024    
025      /** The Client instance used in these tests. */
026      //private static Client client;
027      
028      /** Make a Manager available to this test class. */
029      protected static SensorDataManager sensorDataManager; 
030    
031      /** Make a Manager available to this test class. */
032      protected static UserManager userManager; 
033    
034      /** Make a Manager available to this test class. */
035      protected static SdtManager sdtManager; 
036    
037      /** Make a Manager available to this test class. */
038      protected static ProjectManager projectManager; 
039      
040      /** The admin email. */
041      protected static String adminEmail;
042      /** The admin password. */
043      protected static String adminPassword;
044    
045      /**
046       * Starts the server going for these tests. 
047       * @throws Exception If problems occur setting up the server. 
048       */
049      @BeforeClass public static void setupServer() throws Exception {
050        // Create a 'testing' version of ServerProperties.
051        ServerProperties properties = new ServerProperties();
052        properties.setTestProperties();
053        // Now instantiate the server parameterized for testing purposes. 
054        SensorBaseRestApiHelper.server = Server.newInstance(properties);
055    
056        SensorBaseRestApiHelper.sensorDataManager = 
057          (SensorDataManager)server.getContext().getAttributes().get("SensorDataManager");
058        SensorBaseRestApiHelper.userManager = 
059          (UserManager)server.getContext().getAttributes().get("UserManager");
060        SensorBaseRestApiHelper.sdtManager = 
061          (SdtManager)server.getContext().getAttributes().get("SdtManager");
062        SensorBaseRestApiHelper.projectManager = 
063          (ProjectManager)server.getContext().getAttributes().get("ProjectManager");
064        SensorBaseRestApiHelper.adminEmail = server.getServerProperties().get(ADMIN_EMAIL_KEY);
065        SensorBaseRestApiHelper.adminPassword = server.getServerProperties().get(ADMIN_PASSWORD_KEY);
066      }
067    
068      /**
069       * Returns the hostname associated with this test server. 
070       * @return The host name, including the context root. 
071       */
072      protected String getHostName() {
073        return SensorBaseRestApiHelper.server.getHostName();
074      }
075    }
