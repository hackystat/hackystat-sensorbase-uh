073      /**
074       * Initializes a new SensorBaseClient, given the host, userEmail, and password. 
075       * @param sensorBaseHost The host, such as 'http://localhost:9876/sensorbase'.
076       * @param userEmail The user's email that we will use for authentication. 
077       * @param password The password we will use for authentication.
078       */
079      public SensorBaseClient(String sensorBaseHost, String userEmail, String password) {
080        validateArg(sensorBaseHost);
081        validateArg(userEmail);
082        validateArg(password);
083        this.userEmail = userEmail;
084        this.password = password;
085        this.sensorBaseHost = sensorBaseHost;
086        if (!this.sensorBaseHost.endsWith("/")) {
087          this.sensorBaseHost = this.sensorBaseHost + "/";
088        }
089        this.client = new Client(Protocol.HTTP);
090        try {
091          this.sdtJAXB = 
092            JAXBContext.newInstance("org.hackystat.sensorbase.resource.sensordatatypes.jaxb");
093          this.userJAXB = 
094            JAXBContext.newInstance("org.hackystat.sensorbase.resource.users.jaxb");
095          this.sensordataJAXB = 
096            JAXBContext.newInstance("org.hackystat.sensorbase.resource.sensordata.jaxb");
097          this.projectJAXB = 
098            JAXBContext.newInstance("org.hackystat.sensorbase.resource.projects.jaxb");
099        }
100        catch (Exception e) {
101          throw new RuntimeException("Couldn't create JAXB context instances.");
102        }
103      }

310      /**
311       * GETs the URI string and returns the Restlet Response if the server indicates success.
312       * @param uriString The URI String, such as "http://localhost:9876/sensorbase/sensordatatypes".
313       * @return The response instance if the GET request succeeded.
314       * @throws SensorBaseClientException If the server indicates that a problem occurred. 
315       */
316      public synchronized Response getUri(String uriString) throws SensorBaseClientException {
317        Reference reference = new Reference(uriString);
318        Request request = new Request(Method.GET, reference);
319        request.getClientInfo().getAcceptedMediaTypes().add(xmlMedia); 
320        ChallengeResponse authentication = new ChallengeResponse(scheme, this.userEmail, this.password);
321        request.setChallengeResponse(authentication);
322        Response response = this.client.handle(request);
323        if (!response.getStatus().isSuccess()) {
324          throw new SensorBaseClientException(response.getStatus());
325        }
326        return response; 
327      }

640        // Allow this reassignment of the parameter, even though PMD doesn't like it. 
641        if (!sensorBaseHost.endsWith("/")) {    
642          sensorBaseHost = sensorBaseHost + "/";
643        }
644        Reference reference = new Reference(sensorBaseHost + "users?email=" + email);
645        Request request = new Request(Method.POST, reference);
646        Client client = new Client(Protocol.HTTP);
647        Response response = client.handle(request);
648        if (!response.getStatus().isSuccess()) {
649          throw new SensorBaseClientException(response.getStatus());
650        }
651      }

663      /**
664       * Does the housekeeping for making HTTP requests to the SensorBase by a test or admin user. 
665       * @param method The type of Method.
666       * @param requestString A string, such as "users". No preceding slash. 
667       * @param entity The representation to be sent with the request, or null if not needed.  
668       * @return The Response instance returned from the server.
669       */
670      private Response makeRequest(Method method, String requestString, Representation entity) {
671        Reference reference = new Reference(this.sensorBaseHost + requestString);
672        Request request = (entity == null) ? 
673            new Request(method, reference) :
674              new Request(method, reference, entity);
675        request.getClientInfo().getAcceptedMediaTypes().add(xmlMedia); 
676        ChallengeResponse authentication = new ChallengeResponse(scheme, this.userEmail, this.password);
677        request.setChallengeResponse(authentication);
678        return this.client.handle(request);
679      }

704       * Returns the passed SensorDataType instance as a String encoding of its XML representation.
705       * @param sdt The SensorDataType instance. 
706       * @return The XML String representation.
707       * @throws Exception If problems occur during translation. 
708       */
709      private String makeSensorDataType (SensorDataType sdt) throws Exception {
710        Marshaller marshaller = this.sdtJAXB.createMarshaller(); 
711        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
712        dbf.setNamespaceAware(true);
713        DocumentBuilder documentBuilder = dbf.newDocumentBuilder();
714        Document doc = documentBuilder.newDocument();
715        marshaller.marshal(sdt, doc);
716        DOMSource domSource = new DOMSource(doc);
717        StringWriter writer = new StringWriter();
718        StreamResult result = new StreamResult(writer);
719        TransformerFactory tf = TransformerFactory.newInstance();
720        Transformer transformer = tf.newTransformer();
721        transformer.transform(domSource, result);
722        String xmlString = writer.toString();
723        // Now remove the processing instruction.  This approach seems like a total hack.
724        xmlString = xmlString.substring(xmlString.indexOf('>') + 1);
725        return xmlString;
726      }
