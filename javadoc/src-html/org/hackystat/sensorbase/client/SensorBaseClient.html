114       * Initializes a new SensorBaseClient, given the host, userEmail, and password.
115       * 
116       * @param host The host, such as 'http://localhost:9876/sensorbase'.
117       * @param email The user's email that we will use for authentication.
118       * @param password The password we will use for authentication.
119       */
120      public SensorBaseClient(String host, String email, String password) {
121        validateArg(host);
122        validateArg(email);
123        validateArg(password);
124        this.userEmail = email;
125        this.password = password;
126        this.sensorBaseHost = host;
127        if (!this.sensorBaseHost.endsWith("/")) {
128          this.sensorBaseHost = this.sensorBaseHost + "/";
129        }
130        if (this.isTraceEnabled) {
131          System.out.println("SensorBaseClient Tracing: INITIALIZE " + "host='" + host + "', email='"
132              + email + "', password='" + password + "'");
133        }
134        this.client = new Client(Protocol.HTTP);
135      }

162      /**
163       * Authenticates this user and password with the server.
164       * 
165       * @return This SensorBaseClient instance.
166       * @throws SensorBaseClientException If authentication is not successful.
167       */
168      public synchronized SensorBaseClient authenticate() throws SensorBaseClientException {
169        String uri = "ping?user=" + this.userEmail + "&password=" + this.password;
170        Response response = makeRequest(Method.GET, uri, null);
171        if (!response.getStatus().isSuccess()) {
172          throw new SensorBaseClientException(response.getStatus());
173        }
174        String responseString;
175        try {
176          responseString = response.getEntity().getText();
177        }
178        catch (Exception e) {
179          throw new SensorBaseClientException("Bad response", e);
180        }
181        if (!"SensorBase authenticated".equals(responseString)) {
182          throw new SensorBaseClientException("Authentication failed");
183        }
184        return this;
185      }

187      /**
188       * Provides an easy way to construct SensorData instances. The keyValMap is processed and the
189       * key-value pairs are processed in the following way.
190       * <ul>
191       * <li> ["Owner", email] (if not supplied, defaults to this client's email)
192       * <li> ["Timestamp", timestamp string] (if not supplied, defaults to the current time.)
193       * <li> ["Runtime", timestamp string] (if not supplied, defaults to Timestamp.)
194       * <li> ["Resource", resource string] (if not supplied, defaults to "")
195       * <li> ["SensorDataType", sdt string] (if not supplied, defaults to "")
196       * <li> ["Tool", tool string] (if not supplied, defaults to "")
197       * <li> Any other key-val pairs are extracted and put into the SensorData properties.
198       * </ul>
199       * Throws an exception if Timestamp or Runtime are supplied but can't be parsed into an
200       * XMLGregorianCalendar instance.
201       * 
202       * @param keyValMap The map of key-value pairs corresponding to SensorData fields and properties.
203       * @return A SensorData instance.
204       * @throws SensorBaseClientException If errors occur parsing the contents of the keyValMap.
205       */
206      public synchronized SensorData makeSensorData(Map<String, String> keyValMap)
207          throws SensorBaseClientException {
208        // Begin by creating the sensor data instance.
209        SensorData data = new SensorData();
210        XMLGregorianCalendar defaultTstamp = Tstamp.makeTimestamp();
211        XMLGregorianCalendar tstamp = null;
212        try {
213          tstamp = Tstamp.makeTimestamp(getMap(keyValMap, "Timestamp", defaultTstamp.toString()));
214        }
215        catch (Exception e) {
216          throw new SensorBaseClientException("Error parsing tstamp", e);
217        }
218        XMLGregorianCalendar runtime = null;
219        try {
220          runtime = Tstamp.makeTimestamp(getMap(keyValMap, "Runtime", defaultTstamp.toString()));
221        }
222        catch (Exception e) {
223          throw new SensorBaseClientException("Error parsing runtime", e);
224        }
225        data.setOwner(getMap(keyValMap, "Owner", userEmail));
226        data.setResource(getMap(keyValMap, "Resource", ""));
227        data.setRuntime(runtime);
228        data.setSensorDataType(getMap(keyValMap, "SensorDataType", ""));
229        data.setTimestamp(tstamp);
230        data.setTool(getMap(keyValMap, "Tool", "unknown"));
231        // Add all remaining key-val pairs to the property list.
232        data.setProperties(new org.hackystat.sensorbase.resource.sensordata.jaxb.Properties());
233        for (Map.Entry<String, String> entry : keyValMap.entrySet()) {
234          Property property = new Property();
235          property.setKey(entry.getKey());
236          property.setValue(entry.getValue());
237          data.getProperties().getProperty().add(property);
238        }
239        return data;
240      }

242      /**
243       * Returns the value associated with key in keyValMap, or the default, and also removes the
244       * mapping associated with key from the keyValMap.
245       * 
246       * @param keyValMap The map
247       * @param key The key
248       * @param defaultValue The value to return if the key has no mapping.
249       * @return The value to be used.
250       */
251      private String getMap(Map<String, String> keyValMap, String key, String defaultValue) {
252        String value = (keyValMap.get(key) == null) ? defaultValue : keyValMap.get(key);
253        keyValMap.remove(key);
254        return value;
255      }

353        catch (Exception e) {
354          throw new SensorBaseClientException("Error marshalling SDT", e);
355        }

478      /**
479       * GETs the URI string and returns the Restlet Response if the server indicates success.
480       * 
481       * @param uriString The URI String, such as "http://localhost:9876/sensorbase/sensordatatypes".
482       * @return The response instance if the GET request succeeded.
483       * @throws SensorBaseClientException If the server indicates that a problem occurred.
484       */
485      public synchronized Response getUri(String uriString) throws SensorBaseClientException {
486        Reference reference = new Reference(uriString);
487        Request request = new Request(Method.GET, reference);
488        request.getClientInfo().getAcceptedMediaTypes().add(xmlMedia);
489        ChallengeResponse authentication = new ChallengeResponse(scheme, this.userEmail, this.password);
490        request.setChallengeResponse(authentication);
491        if (this.isTraceEnabled) {
492          System.out.println("SensorBaseClient Tracing: GET " + reference);
493        }
494        Response response = this.client.handle(request);
495        if (this.isTraceEnabled) {
496          Status status = response.getStatus();
497          System.out.println("  => " + status.getCode() + " " + status.getDescription());
498        }
499        if (!response.getStatus().isSuccess()) {
500          throw new SensorBaseClientException(response.getStatus());
501        }
502        return response;
503      }

529      /**
530       * Computes the lastMod value for this index. Iterates through the individual refs to find their
531       * lastMod values, and stores the most recent lastMod value as the result. If the index is empty,
532       * then a default lastMod of 1000-01-01 is returned. We hope that indexes are cached so that this
533       * is not done a lot.
534       * 
535       * @param index The index.
536       * @throws SensorBaseClientException If problems occur parsing the lastMod fields.
537       */
538      private void setSensorDataIndexLastMod(SensorDataIndex index) throws SensorBaseClientException {
539        try {
540          index.setLastMod(Tstamp.makeTimestamp("1000-01-01"));
541          for (SensorDataRef ref : index.getSensorDataRef()) {
542            XMLGregorianCalendar lastMod = ref.getLastMod();
543            if ((!(lastMod == null)) && Tstamp.greaterThan(lastMod, index.getLastMod())) {
544              index.setLastMod(lastMod);
545            }
546          }
547        }
548        catch (Exception e) {
549          throw new SensorBaseClientException("Error setting LastMod for index: " + index, e);
550        }
551      }

605      /**
606       * Returns an index to SensorData for the specified user of all sensor data that have arrived at
607       * the server since the specified start and end times. Uses the LastMod field to determine what
608       * data will be retrieved.
609       * <p>
610       * Note that data could be sent recently with a Timestamp (as opposed to LastMod field) from far
611       * back in the past, and the index will include references to such data. This method thus differs
612       * from all other SensorDataIndex-returning methods, because the others compare passed timestamp
613       * values to the Timestamp associated with the moment at which a sensor data instance is created,

1060       * Returns true if the passed host is a SensorBase host.
1061       * 
1062       * @param host The URL of a sensorbase host, such as "http://localhost:9876/sensorbase".
1063       * @return True if this URL responds as a SensorBase host.
1064       */
1065      public static boolean isHost(String host) {
1066        // All sensorbase hosts use the HTTP protocol.
1067        if (!host.startsWith("http://")) {
1068          return false;
1069        }
1070        // Create the host/register URL.
1071        try {
1072          String registerUri = host.endsWith("/") ? host + "ping" : host + "/ping";
1073          Request request = new Request();
1074          request.setResourceRef(registerUri);
1075          request.setMethod(Method.GET);
1076          Client client = new Client(Protocol.HTTP);
1077          Response response = client.handle(request);
1078          String pingText = response.getEntity().getText();
1079          return (response.getStatus().isSuccess() && "SensorBase".equals(pingText));
1080        }
1081        catch (Exception e) {
1082          return false;
1083        }
1084      }

1129      private Response makeRequest(Method method, String requestString, Representation entity) {
1130        Reference reference = new Reference(this.sensorBaseHost + requestString);

1132            reference, entity);

1134        ChallengeResponse authentication = new ChallengeResponse(scheme, this.userEmail, this.password);
1135        request.setChallengeResponse(authentication);
1136        if (this.isTraceEnabled) {
1137          System.out.println("SensorBaseClient Tracing: " + method + " " + reference);
1138          if (entity != null) {
1139            try {
1140              System.out.println(entity.getText());
1141            }
1142            catch (Exception e) {
1143              System.out.println("  Problems with getText() on entity.");
1144            }
1145          }
1146        }
1147        Response response = this.client.handle(request);
1148        if (this.isTraceEnabled) {
1149          Status status = response.getStatus();
1150          System.out.println("  => " + status.getCode() + " " + status.getDescription());
1151        }

1187        Marshaller marshaller = sdtJAXB.createMarshaller();
1188        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
1189        dbf.setNamespaceAware(true);
1190        DocumentBuilder documentBuilder = dbf.newDocumentBuilder();
1191        Document doc = documentBuilder.newDocument();
1192        marshaller.marshal(sdt, doc);
1193        DOMSource domSource = new DOMSource(doc);
