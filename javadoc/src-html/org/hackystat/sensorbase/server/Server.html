046      private Component component; 
047      
048      /** Holds the host name associated with this Server. */
049      private String hostName;
050      
051      /**
052       * Creates a new instance of a SensorBase HTTP server, listening on the supplied port.  
053       * @return The Server instance created. 
054       * @throws Exception If problems occur starting up this server. 
055       */
056      public static Server newInstance() throws Exception {
057        Server server = new Server();
058        ServerProperties.initializeProperties();
059        server.hostName = "http://" +
060                          ServerProperties.get(HOSTNAME_KEY) + 
061                          ":" + 
062                          ServerProperties.get(PORT_KEY) + 
063                          "/" +
064                          ServerProperties.get(CONTEXT_ROOT_KEY) +
065                          "/";
066        int port = Integer.valueOf(ServerProperties.get(PORT_KEY));
067        server.component = new Component();
068        server.component.getServers().add(Protocol.HTTP, port);
069        server.component.getDefaultHost()
070          .attach("/" + ServerProperties.get(CONTEXT_ROOT_KEY), server);
071     
072        
073        try {
074          Mailer.getInstance();
075        }
076        catch (Throwable e) {
077          String msg = "ERROR: JavaMail not installed correctly! Mail services will fail!";
078          SensorBaseLogger.getLogger().warning(msg);
079        }
080    
081        // Now create all of the Resource Managers and store them in the Context.
082        // Ordering constraints: 
083        // - DbManager must precede all resource managers so it can initialize the tables
084        //   before the resource managers add data to them.
085        // - UserManager must be initialized before ProjectManager, since ProjectManager needs
086        //   to know about the Users. 
087        Map<String, Object> attributes = server.getContext().getAttributes();
088        attributes.put("DbManager", new DbManager(server));
089        attributes.put("SdtManager", new SdtManager(server));
090        attributes.put("UserManager", new UserManager(server));
091        attributes.put("ProjectManager", new ProjectManager(server));
092        attributes.put("SensorDataManager", new SensorDataManager(server));
093        
094        // Now let's open for business. 
095        SensorBaseLogger.getLogger().warning("Host: " + server.hostName);
096        SensorBaseLogger.setLoggingLevel(ServerProperties.get(LOGGING_LEVEL_KEY));
097        ServerProperties.echoProperties();
098        SensorBaseLogger.getLogger().warning("SensorBase (Version " + getVersion() + ") now running.");
099        server.component.start();
100        disableRestletLogging();
101        
102        return server;
103      }
104    
105      /**
106       * Disable all loggers from com.noelios and org.restlet. 
107       */
108      private static void disableRestletLogging() {
109        LogManager logManager = LogManager.getLogManager();
110        for (Enumeration e = logManager.getLoggerNames(); e.hasMoreElements() ;) {
111          String logName = e.nextElement().toString();
112          if (logName.startsWith("com.noelios") ||
113              logName.startsWith("org.restlet")) {
114            logManager.getLogger(logName).setLevel(Level.OFF);
115          }
116        }
117      }
118      
119      /**
120       * Starts up the SensorBase web service on port 9876.  Control-c to exit. 
121       * @param args Ignored. 
122       * @throws Exception if problems occur.
123       */
124      public static void main(final String[] args) throws Exception {
125        Server.newInstance();
126      }

128      /**
129       * Dispatch to the Projects, SensorData, SensorDataTypes, or Users Resource depending on the URL.
130       * We will authenticate all requests except for registration (users?email={email}).
131       * @return The router Restlet.
132       */
133      @Override
134      public Restlet createRoot() {
135        // First, create a Router that will have a Guard placed in front of it so that this Router's
136        // requests will require authentication.
137        Router authRouter = new Router(getContext());
138        authRouter.attach("/sensordatatypes", SensorDataTypesResource.class);
139        authRouter.attach("/sensordatatypes/{sensordatatypename}", SensorDataTypeResource.class);
140        authRouter.attach("/users", UsersResource.class);
141        authRouter.attach("/users/{user}", UserResource.class);
142        authRouter.attach("/sensordata", SensorDataResource.class);
143        authRouter.attach("/sensordata/{user}", UserSensorDataResource.class);
144        authRouter.attach("/sensordata/{user}?sdt={sensordatatype}", UserSensorDataResource.class);
145        authRouter.attach("/sensordata/{user}/{timestamp}", 
146            UserSensorDataResource.class);
147        authRouter.attach("/projects", ProjectsResource.class);
148        authRouter.attach("/projects/{user}", UserProjectsResource.class);
149        authRouter.attach("/projects/{user}/{projectname}", UserProjectResource.class);
150        authRouter.attach("/projects/{user}/{projectname}/sensordata", 
151            UserProjectSensorDataResource.class);
152        authRouter.attach(
153            "/projects/{user}/{projectname}/sensordata?startTime={startTime}&endTime={endTime}", 
154            UserProjectSensorDataResource.class);
155        // Here's the Guard that we will place in front of authRouter.
156        Guard guard = new Authenticator(getContext());
157        guard.setNext(authRouter);
158        
159        // Now create our "top-level" router which will allow the registration URI to proceed without
160        // authentication, but all other URI patterns will go to the guarded Router. 
161        Router router = new Router(getContext());
162        router.attach("/register", RegistrationResource.class);
163        router.attachDefault(guard);
164        
165        return router;
166      }
