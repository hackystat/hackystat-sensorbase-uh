001    package org.hackystat.sensorbase.server;
002    
003    import org.restlet.Component;
004    import org.restlet.Restlet;
005    import org.restlet.data.Form;
006    import org.restlet.data.MediaType;
007    import org.restlet.data.Method;
008    import org.restlet.data.Parameter;
009    import org.restlet.data.Protocol;
010    import org.restlet.data.Request;
011    import org.restlet.data.Response;
012    import org.restlet.data.Status;
013    
014    /**
015     * The Server class. 
016     * @author Philip Johnson
017     */
018    public class Server {
019      
020      /**
021       * The main program that is invoked when running the SensorBase from the command line. 
022       * @param args ignored. for now. 
023       */
024      public static void main(String[] args) {
025        try {
026            // Create a new Restlet component
027            Component component = new Component();
028    
029            // Create the HTTP server connector, then add it as a server
030            // connector to the Restlet component. Note that the component
031            // is the call restlet.
032            component.getServers().add(Protocol.HTTP, 9876);
033    
034            // Prepare and attach a test Handler
035            Restlet handler = new Restlet(component.getContext()) {
036                @Override
037                public void handle(Request request, Response response) {
038                    if (request.getMethod().equals(Method.PUT)) {
039                        System.out.println("Handling the call...");
040                        System.out
041                                .println("Trying to get the entity as a form...");
042                        Form form = request.getEntityAsForm();
043    
044                        System.out.println("Trying to getParameters...");
045                        StringBuffer sb = new StringBuffer("foo");
046                        for (Parameter p : form) {
047                            System.out.println(p);
048    
049                            sb.append("field name = ");
050                            sb.append(p.getName());
051                            sb.append("value = ");
052                            sb.append(p.getValue());
053                            sb.append("\n");
054                            System.out.println(sb.toString());
055                        }
056    
057                        response.setEntity(sb.toString(), MediaType.TEXT_PLAIN);
058                        System.out.println("Done!");
059                    } else {
060                        response.setStatus(Status.SERVER_ERROR_NOT_IMPLEMENTED);
061                    }
062                }
063            };
064    
065            component.getDefaultHost().attach("/test", handler);
066    
067            // Now, start the component
068            component.start();
069        } catch (Exception e) {
070            e.printStackTrace();
071        }
072    }
073    }
