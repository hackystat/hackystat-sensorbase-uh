081      /**
082       * Creates a new instance of a SensorBase HTTP server, listening on the supplied port.
083       * @param  serverProperties The ServerProperties used to initialize this server.
084       * @return The Server instance created. 
085       * @throws Exception If problems occur starting up this server. 
086       */
087      public static Server newInstance(ServerProperties serverProperties) throws Exception {
088        Server server = new Server();
089        server.logger = HackystatLogger.getLogger("org.hackystat.sensorbase");
090        server.serverProperties = serverProperties;
091        server.hostName = "http://" +
092                          server.serverProperties.get(HOSTNAME_KEY) + 
093                          ":" + 
094                          server.serverProperties.get(PORT_KEY) + 
095                          "/" +
096                          server.serverProperties.get(CONTEXT_ROOT_KEY) +
097                          "/";
098        int port = Integer.valueOf(server.serverProperties.get(PORT_KEY));
099        server.component = new Component();
100        server.component.getServers().add(Protocol.HTTP, port);
101        server.component.getDefaultHost()
102          .attach("/" + server.serverProperties.get(CONTEXT_ROOT_KEY), server);
103         
104        try {
105          Mailer.getInstance();
106        }
107        catch (Throwable e) {
108          String msg = "ERROR: JavaMail not installed correctly! Mail services will fail!";
109          server.logger.warning(msg);
110        }
111    
112        // Now create all of the Resource Managers and store them in the Context.
113        // Ordering constraints: 
114        // - DbManager must precede all resource managers so it can initialize the tables
115        //   before the resource managers add data to them.
116        // - UserManager must be initialized before ProjectManager, since ProjectManager needs
117        //   to know about the Users. 
118        Map<String, Object> attributes = server.getContext().getAttributes();
119        attributes.put("DbManager", new DbManager(server));
120        attributes.put("SdtManager", new SdtManager(server));
121        attributes.put("UserManager", new UserManager(server));
122        attributes.put("ProjectManager", new ProjectManager(server));
123        attributes.put("SensorDataManager", new SensorDataManager(server));
124        attributes.put("SensorBaseServer", server);
125        attributes.put("ServerProperties", server.serverProperties);
126        
127        // Move Restlet Logging into a file. 
128        RestletLoggerUtil.useFileHandler("sensorbase");
129        
130        // Now let's open for business. 
131        server.logger.warning("Host: " + server.hostName);
132        HackystatLogger.setLoggingLevel(server.logger, server.serverProperties.get(LOGGING_LEVEL_KEY));
133        server.logger.info(server.serverProperties.echoProperties());
134        server.logger.warning("SensorBase (Version " + getVersion() + ") now running.");
135        server.component.start();
136        return server;
137      }

141       * Starts up the SensorBase web service using the properties specified in sensor.properties.  
142       * Control-c to exit. 
143       * @param args Ignored. 
144       * @throws Exception if problems occur.
145       */

147        Server.newInstance();

150      /**
151       * Dispatch to the Projects, SensorData, SensorDataTypes, or Users Resource depending on the URL.
152       * We will authenticate all requests except for registration (users?email={email}).
153       * @return The router Restlet.
154       */
155      @Override
156      public Restlet createRoot() {
157        // First, create a Router that will have a Guard placed in front of it so that this Router's
158        // requests will require authentication.
159        Router authRouter = new Router(getContext());
160        authRouter.attach("/sensordatatypes", SensorDataTypesResource.class);
161        authRouter.attach("/sensordatatypes/{sensordatatypename}", SensorDataTypeResource.class);
162        authRouter.attach("/users", UsersResource.class);
163        authRouter.attach("/users/{user}", UserResource.class);
164        authRouter.attach("/sensordata", SensorDataResource.class);
165        authRouter.attach("/sensordata/{user}", UserSensorDataResource.class);
166        authRouter.attach("/sensordata/{user}?sdt={sensordatatype}", UserSensorDataResource.class);
167        authRouter.attach(
168            "/sensordata/{user}?lastModStartTime={lastModStartTime}&lastModEndTime={lastModEndTime}", 
169            UserSensorDataResource.class);
170        authRouter.attach("/sensordata/{user}/{timestamp}", 
171            UserSensorDataResource.class);
172        authRouter.attach("/projects", ProjectsResource.class);
173        authRouter.attach("/projects/{user}", UserProjectsResource.class);
174        authRouter.attach("/projects/{user}/{projectname}", UserProjectResource.class);
175        authRouter.attach("/projects/{user}/{projectname}/sensordata", 
176            UserProjectSensorDataResource.class);
177        authRouter.attach("/projects/{user}/{projectname}/sensordata" + 
178            "?startTime={startTime}&endTime={endTime}", UserProjectSensorDataResource.class);
179        authRouter.attach("/projects/{user}/{projectname}/sensordata" + 
180            "?sdt={sdt}&startTime={startTime}&endTime={endTime}", UserProjectSensorDataResource.class);
181        // Here's the Guard that we will place in front of authRouter.
182        Guard guard = new Authenticator(getContext());
183        guard.setNext(authRouter);
184        
185        // Now create our "top-level" router which will allow the registration URI to proceed without
186        // authentication, but all other URI patterns will go to the guarded Router. 
187        Router router = new Router(getContext());
188        router.attach("/register", RegistrationResource.class);
189        router.attach("/ping", PingResource.class);
190        router.attach("/ping?user={user}&password={password}", PingResource.class);
191        router.attachDefault(guard);
192        
193        return router;
194      }
