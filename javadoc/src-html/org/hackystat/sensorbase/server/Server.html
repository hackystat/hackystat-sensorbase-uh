059      /**
060       * Creates a new instance of a SensorBase HTTP server, listening on the supplied port.  
061       * @return The Server instance created. 
062       * @throws Exception If problems occur starting up this server. 
063       */
064      public static Server newInstance() throws Exception {
065        Server server = new Server();
066        server.logger = HackystatLogger.getLogger("org.hackystat.sensorbase");
067        server.serverProperties = new ServerProperties();
068        server.hostName = "http://" +
069                          server.serverProperties.get(HOSTNAME_KEY) + 
070                          ":" + 
071                          server.serverProperties.get(PORT_KEY) + 
072                          "/" +
073                          server.serverProperties.get(CONTEXT_ROOT_KEY) +
074                          "/";
075        int port = Integer.valueOf(server.serverProperties.get(PORT_KEY));
076        server.component = new Component();
077        server.component.getServers().add(Protocol.HTTP, port);
078        server.component.getDefaultHost()
079          .attach("/" + server.serverProperties.get(CONTEXT_ROOT_KEY), server);
080     
081        
082        try {
083          Mailer.getInstance();
084        }
085        catch (Throwable e) {
086          String msg = "ERROR: JavaMail not installed correctly! Mail services will fail!";
087          server.logger.warning(msg);
088        }
089    
090        // Now create all of the Resource Managers and store them in the Context.
091        // Ordering constraints: 
092        // - DbManager must precede all resource managers so it can initialize the tables
093        //   before the resource managers add data to them.
094        // - UserManager must be initialized before ProjectManager, since ProjectManager needs
095        //   to know about the Users. 
096        Map<String, Object> attributes = server.getContext().getAttributes();
097        attributes.put("DbManager", new DbManager(server));
098        attributes.put("SdtManager", new SdtManager(server));
099        attributes.put("UserManager", new UserManager(server));
100        attributes.put("ProjectManager", new ProjectManager(server));
101        attributes.put("SensorDataManager", new SensorDataManager(server));
102        attributes.put("SensorBaseServer", server);
103        attributes.put("ServerProperties", server.serverProperties);
104        
105        // Now let's open for business. 
106        server.logger.warning("Host: " + server.hostName);
107        HackystatLogger.setLoggingLevel(server.logger, server.serverProperties.get(LOGGING_LEVEL_KEY));
108        server.serverProperties.echoProperties(server);
109        server.logger.warning("SensorBase (Version " + getVersion() + ") now running.");
110        server.component.start();
111        disableRestletLogging();
112        return server;
113      }
114    
115      /**
116       * Disable all loggers from com.noelios and org.restlet. 
117       */
118      private static void disableRestletLogging() {
119        LogManager logManager = LogManager.getLogManager();
120        for (Enumeration<String> e = logManager.getLoggerNames(); e.hasMoreElements() ;) {
121          String logName = e.nextElement();
122          if (logName.startsWith("com.noelios") ||
123              logName.startsWith("org.restlet")) {
124            logManager.getLogger(logName).setLevel(Level.OFF);
125          }
126        }
127      }

138      /**
139       * Dispatch to the Projects, SensorData, SensorDataTypes, or Users Resource depending on the URL.
140       * We will authenticate all requests except for registration (users?email={email}).
141       * @return The router Restlet.
142       */
143      @Override
144      public Restlet createRoot() {
145        // First, create a Router that will have a Guard placed in front of it so that this Router's
146        // requests will require authentication.
147        Router authRouter = new Router(getContext());
148        authRouter.attach("/sensordatatypes", SensorDataTypesResource.class);
149        authRouter.attach("/sensordatatypes/{sensordatatypename}", SensorDataTypeResource.class);
150        authRouter.attach("/users", UsersResource.class);
151        authRouter.attach("/users/{user}", UserResource.class);
152        authRouter.attach("/sensordata", SensorDataResource.class);
153        authRouter.attach("/sensordata/{user}", UserSensorDataResource.class);
154        authRouter.attach("/sensordata/{user}?sdt={sensordatatype}", UserSensorDataResource.class);
155        authRouter.attach("/sensordata/{user}?since={lastModTimestamp}", UserSensorDataResource.class);
156        authRouter.attach("/sensordata/{user}/{timestamp}", 
157            UserSensorDataResource.class);
158        authRouter.attach("/projects", ProjectsResource.class);
159        authRouter.attach("/projects/{user}", UserProjectsResource.class);
160        authRouter.attach("/projects/{user}/{projectname}", UserProjectResource.class);
161        authRouter.attach("/projects/{user}/{projectname}/sensordata", 
162            UserProjectSensorDataResource.class);
163        authRouter.attach(
164            "/projects/{user}/{projectname}/sensordata?startTime={startTime}&endTime={endTime}", 
165            UserProjectSensorDataResource.class);
166        // Here's the Guard that we will place in front of authRouter.
167        Guard guard = new Authenticator(getContext());
168        guard.setNext(authRouter);
169        
170        // Now create our "top-level" router which will allow the registration URI to proceed without
171        // authentication, but all other URI patterns will go to the guarded Router. 
172        Router router = new Router(getContext());
173        router.attach("/register", RegistrationResource.class);
174        router.attach("/ping", PingResource.class);
175        router.attach("/ping?user={user}&password={password}", PingResource.class);
176        router.attachDefault(guard);
177        
178        return router;
179      }
