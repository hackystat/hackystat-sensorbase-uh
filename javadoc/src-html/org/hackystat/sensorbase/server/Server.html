001    package org.hackystat.sensorbase.server;
002    
003    import java.util.Map;
004    
005    import org.hackystat.sensorbase.logger.SensorBaseLogger;
006    import org.hackystat.sensorbase.resource.sensordatatypes.SdtManager;
007    import org.hackystat.sensorbase.resource.sensordatatypes.SensorDataTypeResource;
008    import org.hackystat.sensorbase.resource.sensordatatypes.SensorDataTypesResource;
009    import org.restlet.Application;
010    import org.restlet.Component;
011    import org.restlet.Restlet;
012    import org.restlet.Router;
013    import org.restlet.data.Protocol;
014    
015    
016    /**
017     * Sets up the HTTP Server process and dispatching to the associated resources. 
018     * @author Philip Johnson
019     */
020    public class Server extends Application {
021    
022      /** Holds the Restlet Component associated with this Server. */
023      private Component component; 
024      
025      /** Holds the host name associated with this Server. */
026      private String hostName;
027      
028      /**
029       * Creates a new instance of a SensorBase HTTP server, listening on the supplied port.  
030       * @param port The port number for this Server. 
031       * @return The Server instance created. 
032       * @throws Exception If problems occur starting up this server. 
033       */
034      public static Server newInstance(int port) throws Exception {
035        Server server = new Server();
036        server.hostName = "http://localhost:" + port + "/";
037        server.component = new Component();
038        server.component.getServers().add(Protocol.HTTP, port);
039        server.component.getDefaultHost().attach("/sensorbase", server);
040        server.component.start();
041        SensorBaseLogger.getLogger().warning("Started SensorBase (Version " + getVersion() + ")");
042        SensorBaseLogger.getLogger().warning("Host: " + server.hostName);
043        // Save a pointer to this Server instance in this Application's context. 
044        Map<String, Object> attributes = server.getContext().getAttributes();
045        attributes.put("SdtManager", new SdtManager(server));
046        return server;
047      }
048      
049      /**
050       * Starts up the SensorBase web service on port 9876.  Control-c to exit. 
051       * @param args Ignored. 
052       * @throws Exception if problems occur.
053       */
054      public static void main(final String[] args) throws Exception {
055        Server.newInstance(9876);
056      }
057    
058      /**
059       * Dispatch to the Projects, SensorData, SensorDataTypes, or Users Resource depending on the URL.
060       * @return The router Restlet.
061       */
062      @Override
063      public Restlet createRoot() {
064        Router router = new Router(getContext());
065        router.attach("/sensordatatypes", SensorDataTypesResource.class);
066        router.attach("/sensordatatypes/{sensordatatypename}", SensorDataTypeResource.class);
067        return router;
068      }
069    
070    
071      /**
072       * Returns the version associated with this Package, if available from the jar file manifest.
073       * If not being run from a jar file, then returns "Development". 
074       * @return The version.
075       */
076      public static String getVersion() {
077        String version = 
078          Package.getPackage("org.hackystat.sensorbase.server").getImplementationVersion();
079        return (version == null) ? "Development" : version; 
080      }
081      
082      /**
083       * Returns the host name associated with this server. 
084       * Example: "http://localhost:9876/"
085       * @return The host name. 
086       */
087      public String getHostName() {
088        return this.hostName;
089      }
090    }
