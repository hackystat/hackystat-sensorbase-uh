001    package org.hackystat.sensorbase.server;
002    
003    import org.hackystat.sensorbase.resource.users.UserManager;
004    import org.restlet.Context;
005    import org.restlet.Guard;
006    import org.restlet.data.ChallengeScheme;
007    
008    /**
009     * Performs authentication of each HTTP request using HTTP Basic authentication. 
010     * 
011     * @author Philip Johnson
012     */
013    public class Authenticator extends Guard {
014      
015    
016      /**
017       * Initializes this Guard to do HTTP Basic authentication.
018       * @param context The server context.
019       */
020      public Authenticator (Context context) {
021        super(context, ChallengeScheme.HTTP_BASIC,  "SensorBase");
022      }
023      
024      /**
025       * Returns true if the passed credentials are OK.
026       * @param identifier The account name.
027       * @param secret The password. 
028       * @return If the credentials are valid.
029       */
030      @Override protected boolean checkSecret(String identifier, char[] secret) {
031        UserManager manager = (UserManager)getContext().getAttributes().get("UserManager");
032        //SensorBaseLogger.getLogger().info("Authenticating: " + identifier + " " + new String(secret));
033        return manager.isUser(identifier, new String(secret));
034      }
035    }
