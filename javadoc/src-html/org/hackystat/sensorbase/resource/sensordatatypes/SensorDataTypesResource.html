001    package org.hackystat.sensorbase.resource.sensordatatypes;
002    
003    import org.restlet.Context;
004    import org.restlet.data.MediaType;
005    import org.restlet.data.Request;
006    import org.restlet.data.Response;
007    import org.restlet.resource.DomRepresentation;
008    import org.restlet.resource.Representation;
009    import org.restlet.resource.Resource;
010    import org.restlet.resource.Variant;
011    
012    /**
013     * Implements a Restlet Resource representing an index of Hackystat Sensor Data Types. 
014     * @author Philip Johnson
015     */
016    public class SensorDataTypesResource extends Resource {
017      
018      /**
019       * Provides the following representational variants: TEXT_XML.
020       * @param context The context.
021       * @param request The request object.
022       * @param response The response object.
023       */
024      public SensorDataTypesResource(Context context, Request request, Response response) {
025        super(context, request, response);
026        getVariants().clear(); // copyied from BookmarksResource.java, not sure why needed.
027        getVariants().add(new Variant(MediaType.TEXT_XML));
028      }
029      
030      /**
031       * Returns the representation of the SensorDataTypes resource. 
032       * @param variant The representational variant requested.
033       * @return The representation. 
034       */
035      @Override
036      public Representation getRepresentation(Variant variant) {
037        Representation result = null;
038        SdtManager manager = (SdtManager)getContext().getAttributes().get("SdtManager");
039        if (manager == null) {
040          throw new RuntimeException("Failed to find SdtManager");
041        }
042        if (variant.getMediaType().equals(MediaType.TEXT_XML)) {
043          result = new DomRepresentation(MediaType.TEXT_XML, manager.getSensorDataTypeIndexDocument());
044        }
045        return result;
046      }
047    
048      
049    }
