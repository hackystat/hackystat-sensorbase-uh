001    package org.hackystat.sensorbase.resource.sensordatatypes;
002    
003    import static org.junit.Assert.assertEquals;
004    import static org.junit.Assert.assertTrue;
005    
006    import org.junit.Test;
007    import org.restlet.Client;
008    import org.restlet.data.Method;
009    import org.restlet.data.Protocol;
010    import org.restlet.data.Reference;
011    import org.restlet.data.Request;
012    import org.restlet.data.Response;
013    import org.restlet.resource.XmlRepresentation;
014    
015    /**
016     * Tests the SensorBase REST API for the SensorDataType resource.
017     *
018     * @author Philip M. Johnson
019     */
020    public class TestSdtRestApi {
021    
022      /**
023       * Test that GET host/hackystat/sensordatatypes returns an appropriate value.
024       */
025      @Test public void getSdtIndex() {
026        // Set up the call.
027        Method method = Method.GET;
028        Reference reference = new Reference("http://localhost:9090/hackystat/sensordatatypes");
029        Request request = new Request(method, reference);
030    
031        // Make the call.
032        Client client = new Client(Protocol.HTTP);
033        Response response = client.handle(request);
034    
035        // Test that the request was received and processed by the server OK. 
036        assertTrue("Testing for successful status", response.getStatus().isSuccess());
037    
038        // Now test that the response is OK.
039        XmlRepresentation data = response.getEntityAsSax();
040        assertEquals("Checking SDT", "SampleSDT", data.getText("SensorDataTypes/SensorDataType/@Name"));
041        }
042    }
