001    package org.hackystat.sensorbase.resource.sensordatatypes;
002    
003    import java.io.File;
004    import java.io.StringWriter;
005    
006    import javax.xml.bind.JAXBContext;
007    import javax.xml.bind.Marshaller;
008    import javax.xml.bind.Unmarshaller;
009    
010    import org.hackystat.sensorbase.resource.sensordatatypes.jaxb.SensorDataTypes;
011    import org.hackystat.sensorbase.logger.SensorBaseLogger;
012    import org.hackystat.sensorbase.logger.StackTrace;
013    
014    /**
015     * A singleton, thread-safe class providing access to the Sensor Data Type resources.
016     * Uses the System property "hackystat.sensorbase.defaults.sensordatatypes" to find the defaults. 
017     * @author Philip Johnson
018     */
019    public class SdtManager {
020    
021      /** Set to true once the init() method has been called.   */ 
022      private boolean initialized = false;
023    
024      /** The set of defined SensorDataType instances. */
025      private SensorDataTypes sensorDataTypes = new SensorDataTypes();
026    
027      /** The private singleton instance. */
028      private static SdtManager manager = new SdtManager();
029    
030      /** The property name that should resolve to the file containing default SDT definitions. */
031      private static String sdtDefaultsProperty = "hackystat.sensorbase.defaults.sensordatatypes";
032    
033      /** The private constructor of the singleton instance. */
034      private SdtManager() {
035      }
036    
037    
038      /**
039       * Initialize the set of SensorDataTypes by reading in default definitions, if any.
040       */
041      private void init() {
042        // First, see if we can find a file containing default definitions for SDTs.
043        String sdtDefaultsFileName =  System.getProperty(sdtDefaultsProperty, "");
044        File defaultsFile = new File(sdtDefaultsFileName);
045    
046        // Initialize the SDTs if we've found a default file. 
047        if (defaultsFile.exists()) {
048          try {
049            JAXBContext jc = 
050              JAXBContext.newInstance("org.hackystat.sensorbase.resource.sensordatatypes.jaxb");
051            Unmarshaller unmarshaller = jc.createUnmarshaller();
052            this.sensorDataTypes = (SensorDataTypes) unmarshaller.unmarshal(defaultsFile);
053          }
054          catch (Exception e) {
055            SensorBaseLogger.getLogger().warning("Failed to unmarshall SDTs." + StackTrace.toString(e));
056            throw new RuntimeException("Failed to unmarshall SDTs", e);
057          }
058        }
059      }
060    
061      /**
062       * Returns the singleton instance of the SDT Manager. 
063       * @return The singleton manager. 
064       */
065      public static synchronized SdtManager getInstance() {
066        // invoke the init() method the first time getInstance is called. 
067        if (!SdtManager.manager.initialized) {
068          SdtManager.manager.initialized = true;
069          SdtManager.manager.init();
070        }
071        return SdtManager.manager;
072      } 
073    
074      /**
075       * Returns the set of SDTs as an XML string.
076       * @return The set of SDTs as an XML string.
077       */
078      public synchronized String getSensorDataTypes() {
079        String sdtString = "";
080        try {
081          JAXBContext jaxbContext = 
082            JAXBContext.newInstance("org.hackystat.sensorbase.resource.sensordatatypes.jaxb");
083          Marshaller marshaller = jaxbContext.createMarshaller();
084          marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT , new Boolean(true));
085          StringWriter writer = new StringWriter();
086          marshaller.marshal(this.sensorDataTypes, writer);
087          sdtString = writer.toString();
088        } 
089        catch (Exception e ) {
090          SensorBaseLogger.getLogger().warning("Failed to marshall SDTs." + StackTrace.toString(e));
091          throw new RuntimeException("Failed to marshall SDTs", e);
092        }
093        return sdtString;
094      }
095    }
