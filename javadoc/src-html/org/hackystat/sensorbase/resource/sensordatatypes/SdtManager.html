001    package org.hackystat.sensorbase.resource.sensordatatypes;
002    
003    import java.io.File;
004    import java.io.StringWriter;
005    
006    import javax.xml.bind.JAXBContext;
007    import javax.xml.bind.Marshaller;
008    import javax.xml.bind.Unmarshaller;
009    import javax.xml.parsers.DocumentBuilder;
010    import javax.xml.parsers.DocumentBuilderFactory;
011    
012    import org.hackystat.sensorbase.resource.sensordatatypes.jaxb.SensorDataType;
013    import org.hackystat.sensorbase.resource.sensordatatypes.jaxb.SensorDataTypeIndex;
014    import org.hackystat.sensorbase.resource.sensordatatypes.jaxb.SensorDataTypeRef;
015    import org.hackystat.sensorbase.resource.sensordatatypes.jaxb.SensorDataTypes;
016    import org.hackystat.sensorbase.server.Server;
017    import org.hackystat.sensorbase.logger.SensorBaseLogger;
018    import org.hackystat.sensorbase.logger.StackTrace;
019    import org.w3c.dom.Document;
020    
021    /**
022     * Manages access to both the SensorDataType and SensorDataTypes resources. 
023     * Loads default definitions if available. 
024     * @author Philip Johnson
025     */
026    public class SdtManager {
027    
028      /** The set of defined SensorDataType instances. */
029      private SensorDataTypes sensorDataTypes = new SensorDataTypes();
030    
031      /** The property name that should resolve to the file containing default SDT definitions. */
032      private static String sdtDefaultsProperty = "hackystat.sensorbase.defaults.sensordatatypes";
033      
034      /** The JAXB marshaller for SensorDataTypes. */
035      private Marshaller marshaller; 
036      
037      /** The JAXB ummarshaller for SensorDataTypes. */
038      private Unmarshaller unmarshaller;
039      
040      /** The DocumentBuilder for documents. */
041      private DocumentBuilder documentBuilder; 
042      
043      /** The Server associated with this SdtManager. */
044      Server server; 
045      
046      /** 
047       * The constructor for SdtManagers. 
048       * There is one SdtManager per Server. 
049       * @param server The Server instance associated with this SdtManager. 
050       */
051      public SdtManager(Server server) {
052        this.server = server;
053        File defaultsFile = findDefaultsFile();
054        // Initialize the SDTs if we've found a default file. 
055        if (defaultsFile.exists()) {
056          SensorBaseLogger.getLogger().info("Loading SDT defaults from " + defaultsFile.getPath());
057          try {
058            JAXBContext jc = 
059              JAXBContext.newInstance("org.hackystat.sensorbase.resource.sensordatatypes.jaxb");
060            this.unmarshaller = jc.createUnmarshaller();
061            this.marshaller = jc.createMarshaller(); 
062            this.sensorDataTypes = (SensorDataTypes) unmarshaller.unmarshal(defaultsFile);
063            
064            // Initialize documentBuilder
065            DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
066            dbf.setNamespaceAware(true);
067            this.documentBuilder = dbf.newDocumentBuilder();
068          }
069          catch (Exception e) {
070            String msg = "Exception during SDT JAXB initialization processing";
071            SensorBaseLogger.getLogger().warning(msg + StackTrace.toString(e));
072            throw new RuntimeException(msg, e);
073          }
074        }
075      }
076      
077      /**
078       * Checks the System property in SdtManager.sdtDefaultsProperty for the file to load.
079       * If this property is null, returns the File for ./xml/defaults/sensordatatypes.defaults.xml.
080       * @return The File instance (which might not point to an existing file.)
081       */
082      private File findDefaultsFile() {
083        return (System.getProperties().containsKey(sdtDefaultsProperty)) ?
084            new File (System.getProperty(sdtDefaultsProperty)) :
085              new File (System.getProperty("user.dir") + "/xml/defaults/sensordatatypes.defaults.xml");
086      }
087    
088      /**
089       * Returns the set of SDTs as an XML string.
090       * @return The set of SDTs as an XML string.
091       */
092      public synchronized String getSensorDataTypesString() {
093        String sdtString = "";
094        try {
095          StringWriter writer = new StringWriter();
096          this.marshaller.marshal(this.sensorDataTypes, writer);
097          sdtString = writer.toString();
098        } 
099        catch (Exception e ) {
100          String msg = "Failed to marshall SDTs into a String";
101          SensorBaseLogger.getLogger().warning(msg + StackTrace.toString(e));
102          throw new RuntimeException(msg, e);
103        }
104        return sdtString;
105      }
106      
107      /**
108       * Returns the XML Index for all current defined SDTs.
109       * @return The XML Document instance providing an index to all current SDTs.
110       */
111      public synchronized Document getSensorDataTypeIndexDocument() {
112        // First, create the freakin index.
113        SensorDataTypeIndex index = new SensorDataTypeIndex();
114        for (SensorDataType sdt : sensorDataTypes.getSensorDataType()) {
115          SensorDataTypeRef ref = new SensorDataTypeRef();
116          ref.setName(sdt.getName());
117          ref.setHref(this.server.getHostName() + "sensorbase/sensordatatypes/" + sdt.getName());
118          index.getSensorDataTypeRef().add(ref);
119        }
120        // Now convert it to XML.
121        Document doc;
122        try {
123          doc = this.documentBuilder.newDocument();
124          this.marshaller.marshal(index, doc);
125        } 
126        catch (Exception e ) {
127          String msg = "Failed to marshall SDTs into an Index";
128          SensorBaseLogger.getLogger().warning(msg + StackTrace.toString(e));
129          throw new RuntimeException(msg, e);
130        }
131        return doc;
132      }
133      
134      /**
135       * Returns the XML representation of the named SDT.
136       * @param name The name of the SDT.
137       * @return The XML representation of that SDT, or null if not found.
138       */
139      public synchronized Document getSensorDataTypeDocument(String name) {
140        // Iterate through all SDTs, break when we find the one we need. 
141        SensorDataType sdt = null; 
142        Document doc = null;
143        for (SensorDataType tempSdt : sensorDataTypes.getSensorDataType()) {
144          if (tempSdt.getName().equals(name)) {
145            sdt = tempSdt;
146            // Now convert it to XML.
147            try {
148              doc = this.documentBuilder.newDocument();
149              this.marshaller.marshal(sdt, doc);
150            } 
151            catch (Exception e ) {
152              String msg = "Failed to marshall the SDT named: " + name;
153              SensorBaseLogger.getLogger().warning(msg + StackTrace.toString(e));
154              throw new RuntimeException(msg, e);
155            }
156            return doc;
157          }
158        }
159        // If we got here, it's not good, since that means we didn't find the SDT. 
160        // Return a null doc.
161        return doc;
162      }
163    }
