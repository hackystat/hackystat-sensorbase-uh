031    /**
032     * Provides a manager for the SensorDataType resource.
033     * As with all of the Resource managers the methods in this class can be grouped into 
034     * three general categories, of which this Manager uses the following:
035     * <ul>
036     * <li> Database access methods (get*, has*, put*, delete*):  these methods communicate with the 
037     * underlying storage system (or cache). 
038     * <li> XML/Java translation methods (make*): these methods translate between the XML String
039     * representation of a resource and its Java class instance representation. 
040     * </ul>
041     * <p>  
042     * See https://jaxb.dev.java.net/guide/Performance_and_thread_safety.html for info 
043     * on JAXB performance and thread safety.
044     * <p>
045     * All public methods of this class are synchronized so that we can maintain the cache along with
046     * the underlying persistent store in a thread-safe fashion.
047     * 
048     * @author Philip Johnson
049     *
050     */
051    public class SdtManager {
052      
053      /** Holds the class-wide JAXBContext, which is thread-safe. */
054      private JAXBContext jaxbContext;
055      
056      /** The Server associated with this SdtManager. */
057      Server server; 
058      
059      /** The DbManager associated with this server. */
060      DbManager dbManager;
061      
062      /** The SensorDataTypeIndex open tag. */
063      public static final String sensorDataTypeIndexOpenTag = "<SensorDataTypeIndex>";
064      
065      /** The SensorDataTypeIndex close tag. */
066      public static final String sensorDataTypeIndexCloseTag = "</SensorDataTypeIndex>";
067      
068      /** An in-memory cache mapping SDT names to their corresponding instance. */
069      private Map<String, SensorDataType> name2sdt = new HashMap<String, SensorDataType>();
070      
071      /** An in-memory cache mapping a SensorDataType instance to its SensorDataType XML string. */
072      private Map<SensorDataType, String> sdt2xml = new HashMap<SensorDataType, String>();
073      
074      /** An in-memory cache mapping a SensorDataType instance to its SensorDataTypeRef XML string. */
075      private Map<SensorDataType, String> sdt2ref = new HashMap<SensorDataType, String>();
076      
077      
078      /** 
079       * The constructor for SdtManagers. Loads in default data and sets up the in-memory caches. 
080       * @param server The Server instance associated with this SdtManager. 
081       */
082      public SdtManager(Server server) {
083        this.server = server;
084        this.dbManager = (DbManager)this.server.getContext().getAttributes().get("DbManager");
085        try {
086          this.jaxbContext = 
087            JAXBContext.newInstance("org.hackystat.sensorbase.resource.sensordatatypes.jaxb");
088          loadDefaultSensorDataTypes(); //NOPMD it's throwing a false warning. 
089          initializeCache();            //NOPMD 
090        }
091        catch (Exception e) {
092          String msg = "Exception during SdtManager initialization processing";
093          server.getLogger().warning(msg + "\n" + StackTrace.toString(e));
094          throw new RuntimeException(msg, e);
095        }
096      }
097    
098      /**
099       * Loads the default SensorDataTypes from the defaults file. 
100       * @throws Exception If problems occur. 
101       */
102      private void loadDefaultSensorDataTypes() throws Exception {
103        // Get the default SDT definitions from the XML defaults file. 
104        File defaultsFile = findDefaultsFile();
105        // Initialize the SDTs if we've found a default file. 
106        if (defaultsFile.exists()) {
107          server.getLogger().info("Loading SDT defaults from " + defaultsFile.getPath()); 
108          Unmarshaller unmarshaller = jaxbContext.createUnmarshaller();
109          SensorDataTypes sensorDataTypes = (SensorDataTypes) unmarshaller.unmarshal(defaultsFile);
110          for (SensorDataType sdt : sensorDataTypes.getSensorDataType()) {
111            sdt.setLastMod(Tstamp.makeTimestamp());
112            this.dbManager.storeSensorDataType(sdt, this.makeSensorDataType(sdt), 
113                this.makeSensorDataTypeRefString(sdt));
114          }
115        }
116      }
117      
118      /**
119       * Read the SDTs from the underlying database and initialize the in-memory cache.
120       */
121      private void initializeCache() {
122        try {
123          SensorDataTypeIndex index = makeSensorDataTypeIndex(this.dbManager.getSensorDataTypeIndex());
124          for (SensorDataTypeRef ref : index.getSensorDataTypeRef()) {
125            String sdtName = ref.getName();
126            String sdtString = this.dbManager.getSensorDataType(sdtName);
127            SensorDataType sdt = makeSensorDataType(sdtString);
128            String sdtRef = this.makeSensorDataTypeRefString(sdt);
129            this.name2sdt.put(sdtName, sdt);
130            this.sdt2ref.put(sdt, sdtRef);
131            this.sdt2xml.put(sdt, sdtString);
132          }
133        }
134        catch (Exception e) {
135          server.getLogger().warning("Failed to initialize SDTs " + StackTrace.toString(e));
136        }
137      }
138      
139      /**
140       * Checks the ServerProperties for the XML_DIR property.
141       * If this property is null, returns the File for ./xml/defaults/sensordatatypes.defaults.xml.
142       * @return The File instance (which might not point to an existing file.)
143       */
144      private File findDefaultsFile() {
145        String defaultsPath = "/defaults/sensordatatypes.defaults.xml";
146        String xmlDir = server.getServerProperties().get(XML_DIR_KEY);
147        return (xmlDir == null) ?
148            new File (System.getProperty("user.dir") + "/xml" + defaultsPath) :
149              new File (xmlDir + defaultsPath);
150      }
151    
152    
153      /**
154       * Returns the XML string containing the SensorDataTypeIndex with all defined SDTs.
155       * Uses the in-memory cache of SensorDataTypeRef strings.  
156       * @return The XML string providing an index to all current SDTs.
157       */
158      public synchronized String getSensorDataTypeIndex() {
159        StringBuilder builder = new StringBuilder(512);
160        builder.append(sensorDataTypeIndexOpenTag);
161        for (String ref : this.sdt2ref.values()) {
162          builder.append(ref);
163        }
164        builder.append(sensorDataTypeIndexCloseTag);
165        return builder.toString();
166      }
167      
168      /**
169       * Returns the XML String representation of a SensorDataType, given its name.  
170       * @param sdtName The name of the SDT. 
171       * @return Its string representation, or null if not found. 
172       */
173      public synchronized String getSensorDataTypeString(String sdtName) {
174        SensorDataType sdt = this.name2sdt.get(sdtName);
175        return (sdt == null) ? null : this.sdt2xml.get(sdt);
176      }
177      
178      /**
179       * Updates the Manager with this SDT. Any old definition with this name is overwritten.
180       * Updates in-memory caches and persistent store. 
181       * @param sdt The SensorDataType.
182       */
183      public synchronized void putSdt(SensorDataType sdt) {
184        try {
185          sdt.setLastMod(Tstamp.makeTimestamp());
186          String sdtString = this.makeSensorDataType(sdt);
187          String sdtRefString = this.makeSensorDataTypeRefString(sdt);
188          this.dbManager.storeSensorDataType(sdt, sdtString, sdtRefString);
189          this.name2sdt.put(sdt.getName(), sdt);
190          this.sdt2ref.put(sdt, sdtRefString);
191          this.sdt2xml.put(sdt, sdtString);
192        }
193        catch (Exception e) {
194          server.getLogger().warning("Failed to put SDT" + StackTrace.toString(e));
195        }
196      }
197      
198      /**
199       * Returns true if the passed SDT name is defined. 
200       * @param sdtName A SensorDataType name
201       * @return True if a SensorDataType with that name is already known to this SdtManager.
202       */
203      public synchronized boolean hasSdt(String sdtName) {
204        return this.name2sdt.containsKey(sdtName);
205      }
206      
207      /**
208       * Ensures that the passed sdtName is no longer present in this Manager. 
209       * @param sdtName The name of the SDT to remove if currently present.
210       */
211      public synchronized void deleteSdt(String sdtName) {
212        SensorDataType sdt = this.name2sdt.get(sdtName);
213        if (sdt != null) {
214          this.name2sdt.remove(sdtName);
215          this.sdt2ref.remove(sdt);
216          this.sdt2xml.remove(sdt);
217        }
218        this.dbManager.deleteSensorDataType(sdtName);
219      }

221      /**
222       * Takes a String encoding of a SensorDataType in XML format and converts it to an instance. 
223       * 
224       * @param xmlString The XML string representing a SensorDataType
225       * @return The corresponding SensorDataType instance. 
226       * @throws Exception If problems occur during unmarshalling.
227       */
228      public final synchronized SensorDataType makeSensorDataType(String xmlString) throws Exception {
229        Unmarshaller unmarshaller = this.jaxbContext.createUnmarshaller();
230        return (SensorDataType)unmarshaller.unmarshal(new StringReader(xmlString));
231      }
232      
233      /**
234       * Takes a String encoding of a SensorDataTypeIndex in XML format and converts it to an instance. 
235       * Note that this does not affect the state of any SdtManager instance. 
236       * 
237       * @param xmlString The XML string representing a SensorDataTypeIndex.
238       * @return The corresponding SensorDataTypeIndex instance. 
239       * @throws Exception If problems occur during unmarshalling.
240       */
241      public final synchronized SensorDataTypeIndex makeSensorDataTypeIndex(String xmlString) 
242      throws Exception {
243        Unmarshaller unmarshaller = this.jaxbContext.createUnmarshaller();
244        return (SensorDataTypeIndex)unmarshaller.unmarshal(new StringReader(xmlString));
245      }
246      
247      /**
248       * Returns the passed SensorDataType instance as a String encoding of its XML representation.
249       * Final because it's called in constructor.
250       * @param sdt The SensorDataType instance. 
251       * @return The XML String representation.
252       * @throws Exception If problems occur during translation. 
253       */
254      public final synchronized String makeSensorDataType (SensorDataType sdt) throws Exception {
255        Marshaller marshaller = jaxbContext.createMarshaller(); 
256        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
257        dbf.setNamespaceAware(true);
258        DocumentBuilder documentBuilder = dbf.newDocumentBuilder();
259        Document doc = documentBuilder.newDocument();
260        marshaller.marshal(sdt, doc);
261        DOMSource domSource = new DOMSource(doc);
262        StringWriter writer = new StringWriter();
263        StreamResult result = new StreamResult(writer);
264        TransformerFactory tf = TransformerFactory.newInstance();
265        Transformer transformer = tf.newTransformer();
266        transformer.transform(domSource, result);
267        String xmlString = writer.toString();
268        // Now remove the processing instruction.  This approach seems like a total hack.
269        xmlString = xmlString.substring(xmlString.indexOf('>') + 1);
270        return xmlString;
271      }

273      /**
274       * Returns the passed SensorDataType instance as a String encoding of its XML representation 
275       * as a SensorDataTypeRef object.
276       * Final because it's called in constructor.
277       * @param sdt The SensorDataType instance. 
278       * @return The XML String representation of it as a SensorDataTypeRef
279       * @throws Exception If problems occur during translation. 
280       */
281      public final synchronized String makeSensorDataTypeRefString (SensorDataType sdt) 
282      throws Exception {
283        SensorDataTypeRef ref = makeSensorDataTypeRef(sdt);
284        Marshaller marshaller = jaxbContext.createMarshaller(); 
285        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
286        dbf.setNamespaceAware(true);
287        DocumentBuilder documentBuilder = dbf.newDocumentBuilder();
288        Document doc = documentBuilder.newDocument();
289        marshaller.marshal(ref, doc);
290        DOMSource domSource = new DOMSource(doc);
291        StringWriter writer = new StringWriter();
292        StreamResult result = new StreamResult(writer);
293        TransformerFactory tf = TransformerFactory.newInstance();
294        Transformer transformer = tf.newTransformer();
295        transformer.transform(domSource, result);
296        String xmlString = writer.toString();
297        // Now remove the processing instruction.  This approach seems like a total hack.
298        xmlString = xmlString.substring(xmlString.indexOf('>') + 1);
299        return xmlString;
300      }
301      
302      /**
303       * Returns a SensorDataTypeRef instance constructed from a SensorDataType instance.
304       * @param sdt The SensorDataType instance. 
305       * @return A SensorDataTypeRef instance. 
306       */
307      public synchronized SensorDataTypeRef makeSensorDataTypeRef(SensorDataType sdt) {
308        SensorDataTypeRef ref = new SensorDataTypeRef();
309        ref.setName(sdt.getName());
310        ref.setHref(this.server.getHostName() + "sensordatatypes/" + sdt.getName()); 
311        return ref;
312      }
