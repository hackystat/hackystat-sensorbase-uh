001    package org.hackystat.sensorbase.resource.sensordatatypes;
002    
003    import java.io.File;
004    import javax.xml.bind.JAXBContext;
005    import javax.xml.bind.Unmarshaller;
006    
007    import org.hackystat.sensorbase.resource.sensordatatypes.jaxb.SensorDataType;
008    import org.hackystat.sensorbase.resource.sensordatatypes.jaxb.SensorDataTypes;
009    
010    /**
011     * Just some simple code to illustrate JAXB processing with the SensorBase resource jar file.
012     * @author Philip Johnson
013     *
014     */
015    public class SimpleSDTExample {
016    
017      /**
018       * Illustrate how to do JAXB processing.
019       * @param args ignored. 
020       * @throws Exception Ignored.
021       */
022      public static void main(String[] args) throws Exception {
023        // Example file containing sensor data type definitions.
024        File dataFile = 
025          new File("/Users/johnson/svn-google/sensorbase-uh/xml/examples/sensordatatypes.example.xml");
026        // Next two lines are special JAXB magic.
027        JAXBContext context = 
028          JAXBContext.newInstance("org.hackystat.sensorbase.resource.sensordatatypes.jaxb");
029        Unmarshaller unmarshaller = context.createUnmarshaller();
030        // Get an instance of our XML file as a Java class.
031        SensorDataTypes sensorDataTypes = (SensorDataTypes) unmarshaller.unmarshal(dataFile);
032        // Now we can manipulate our XML data in Java easily.
033        for (SensorDataType sensorDataType : sensorDataTypes.getSensorDataType()) {
034          System.out.println(sensorDataType.getName());
035        }
036      }
037    }
