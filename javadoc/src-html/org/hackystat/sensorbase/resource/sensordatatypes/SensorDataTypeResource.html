001    package org.hackystat.sensorbase.resource.sensordatatypes;
002    
003    import org.hackystat.utilities.stacktrace.StackTrace;
004    import org.hackystat.sensorbase.resource.sensorbase.SensorBaseResource;
005    import org.hackystat.sensorbase.resource.sensordatatypes.jaxb.SensorDataType;
006    import org.restlet.Context;
007    import org.restlet.data.MediaType;
008    import org.restlet.data.Request;
009    import org.restlet.data.Response;
010    import org.restlet.data.Status;
011    import org.restlet.resource.Representation;
012    import org.restlet.resource.Variant;
013    
014    /**
015     * Implements a resource for PUT, GET, DELETE of host/sensordatatype/{sensordatatypename}.
016     * @author Philip Johnson
017     */
018    public class SensorDataTypeResource extends SensorBaseResource {
019      
020      /** To be retrieved from the URL. */
021      private String sdtName; 
022      
023      /**
024       * Provides the following representational variants: TEXT_XML.
025       * @param context The context.
026       * @param request The request object.
027       * @param response The response object.
028       */
029      public SensorDataTypeResource(Context context, Request request, Response response) {
030        super(context, request, response);
031        this.sdtName = (String) request.getAttributes().get("sensordatatypename");
032      }
033      
034      /**
035       * Returns the representation of the specified SensorDataType resource.
036       * @param variant The representational variant requested.
037       * @return The representation. 
038       */
039      @Override
040      public Representation getRepresentation(Variant variant) {
041        if (!super.sdtManager.hasSdt(this.sdtName)) {
042          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown SDT: " + this.sdtName);
043          return null;
044        } 
045        if (variant.getMediaType().equals(MediaType.TEXT_XML)) {
046          String xmlData = super.sdtManager.getSensorDataTypeString(this.sdtName);
047          return super.getStringRepresentation(xmlData);
048        }
049        return null;
050      }
051      
052      /** 
053       * Indicate the PUT method is supported. 
054       * @return True.
055       */
056      @Override
057      public boolean allowPut() {
058          return true;
059      }
060    
061      /**
062       * Implement the PUT method that creates a new SDT. 
063       * <ul>
064       * <li> The XML must be marshallable into an SDT instance using the SDT XmlSchema definition.
065       * <li> The SDT name in the URI string must match the SDT name in the XML.
066       * <li> The authenticated user must be the admin.
067       * </ul>
068       * @param entity The XML representation of the new SDT. 
069       */
070      @Override
071      public void put(Representation entity) {
072        String entityString = null;
073        SensorDataType sdt;
074        if (!super.userManager.isAdmin(this.authUser)) {
075          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Only the admin can define SDTs");
076          return;
077        }
078    
079        // Try to make the XML payload into an SDT, return failure if this fails. 
080        try { 
081          entityString = entity.getText();
082          sdt = super.sdtManager.makeSensorDataType(entityString);
083        }
084        catch (Exception e) {
085          server.getLogger().warning("Bad Sdt Definition in PUT: " + StackTrace.toString(e));
086          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad SDT: " + entityString);
087          return;
088        }
089        // Return failure if the URI SdtName is not the same as the XML SdtName.
090        if (!(this.sdtName.equals(sdt.getName()))) {
091          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "URI/XML name mismatch");
092          return;
093        }
094        // otherwise we add it to the Manager and return success.
095        super.sdtManager.putSdt(sdt);      
096        getResponse().setStatus(Status.SUCCESS_CREATED);
097      }
098      
099      /** 
100       * Indicate the DELETE method is supported. 
101       * @return True.
102       */
103      @Override
104      public boolean allowDelete() {
105          return true;
106      }
107      
108      /**
109       * Implement the DELETE method that deletes an existing SDT given its name.
110       */
111      @Override
112      public void delete() {
113        if (!super.userManager.isAdmin(this.authUser)) {
114          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Only the admin can delete SDTs");
115          return;
116        }    
117        super.sdtManager.deleteSdt(sdtName);      
118        getResponse().setStatus(Status.SUCCESS_OK);
119      }
120    }
