001    package org.hackystat.sensorbase.resource.ping;
002    
003    import org.hackystat.sensorbase.resource.sensorbase.SensorBaseResource;
004    import org.restlet.Context;
005    import org.restlet.data.Request;
006    import org.restlet.data.Response;
007    import org.restlet.resource.Representation;
008    import org.restlet.resource.StringRepresentation;
009    import org.restlet.resource.Variant;
010    
011    /**
012     * The PingResource responds to a GET {host}/ping with the string "SensorBase".
013     * It responds to GET  {host}/ping?user={user}&password={password} with
014     * "SensorBase authenticated" if the user and password are valid, and 
015     * "SensorBase" if not valid. 
016     * @author Philip Johnson
017     */
018    public class PingResource extends SensorBaseResource {
019      /** From the URI, if authentication is desired. */
020      private String user; 
021      /** From the URI, if authentication is desired. */
022      private String password;
023      
024      /**
025       * The standard constructor.
026       * @param context The context.
027       * @param request The request object.
028       * @param response The response object.
029       */
030      public PingResource(Context context, Request request, Response response) {
031        super(context, request, response);
032        this.user = (String) request.getAttributes().get("user");
033        this.password = (String) request.getAttributes().get("password");
034      }
035      
036      /**
037       * Returns the string "DailyProjectData" or "DailyProjectData authenticated", 
038       * depending upon whether credentials are passed as form parameters and whether
039       * they are valid. 
040       * @param variant The representational variant requested.
041       * @return The representation as a string.  
042       */
043      @Override
044      public Representation getRepresentation(Variant variant) {
045        String unauthenticated = "SensorBase";
046        String authenticated = "SensorBase authenticated";
047        // Don't try to authenticate unless the user has passed both a user and password. 
048        if ((user == null) || (password == null)) {
049          return new StringRepresentation(unauthenticated);
050        }
051        boolean OK = this.userManager.isUser(user, password);
052        return new StringRepresentation((OK ? authenticated : unauthenticated));
053      }
054      
055      
056    
057    }
