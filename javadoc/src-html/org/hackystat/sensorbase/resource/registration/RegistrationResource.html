001    package org.hackystat.sensorbase.resource.registration;
002    
003    import static org.hackystat.sensorbase.server.ServerProperties.ADMIN_EMAIL_KEY;
004    import static org.hackystat.sensorbase.server.ServerProperties.HOSTNAME_KEY;
005    
006    import org.hackystat.sensorbase.mailer.Mailer;
007    import org.hackystat.sensorbase.resource.sensorbase.SensorBaseResource;
008    import org.hackystat.sensorbase.resource.users.jaxb.User;
009    import org.restlet.Context;
010    import org.restlet.data.Form;
011    import org.restlet.data.MediaType;
012    import org.restlet.data.Request;
013    import org.restlet.data.Response;
014    import org.restlet.data.Status;
015    import org.restlet.resource.Representation;
016    import org.restlet.resource.StringRepresentation;
017    import org.restlet.resource.Variant;
018    
019    /**
020     * Provides registration services for this SensorBase.
021     * Implements a simple web page for accepting a POSTed form containing an email address
022     * to register.  Sends email with the password to this user. 
023     * @author Philip Johnson
024     *
025     */
026    public class RegistrationResource extends SensorBaseResource {
027      
028      /**
029       * The standard constructor.
030       * @param context The context.
031       * @param request The request object.
032       * @param response The response object.
033       */
034      public RegistrationResource(Context context, Request request, Response response) {
035        super(context, request, response);
036      }
037      
038      /**
039       * Returns a page providing a registration form. 
040       * This requires no authorization.
041       * @param variant The representational variant requested.
042       * @return The representation. 
043       */
044      @Override
045      public Representation getRepresentation(Variant variant) {
046        String pageHtml = 
047          "<html>" +
048          "  <body>" +
049          "  Welcome to the Hackystat SensorBase." +
050          "  <p>Please enter your email address below to register." +
051          "  <p>A password to this SensorBase will be emailed to you. " +
052          "  <form action=\"register\" method=\"POST\">" +
053          "  <input name=\"email\" type=\"text\" size=\"15\"/> " +
054          "  <input  type=\"submit\" name=\"Submit\" value=\"Register\">" +
055          "  </form>";
056        Representation representation = new StringRepresentation(pageHtml);
057        representation.setMediaType(MediaType.TEXT_HTML);
058        return representation;
059      }
060      
061      /** 
062       * Indicate the POST method is supported. 
063       * @return True.
064       */
065      @Override
066      public boolean allowPost() {
067        return true;
068      }
069      
070      /**
071       * Implement the POST method that registers a new user. 
072       * @param entity The email address to be registered.
073       */
074      @Override
075      public void post(Representation entity) {
076        Form form = new Form(entity);
077        String email = form.getFirstValue("email");
078        // Return Badness if we don't have the email attribute.
079        if (email == null || "".equals(email)) {
080          server.getLogger().warning("Invalid registration request: empty email"); 
081          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Missing email parameter");
082          return;
083        }
084        User user = super.userManager.registerUser(email);
085        // Now send the email to the (non-test) user and the hackystat admin.
086        Mailer mailer = Mailer.getInstance();
087        String emailSubject = "Hackystat Version 8 Registration";
088        String emailBody = 
089          "Welcome to Hackystat Version 8. " +
090          "\nYou are registered with the server: " + server.getServerProperties().getFullHost() +
091          "\nYour user name is: " + user.getEmail() +
092          "\nYour password is: " + user.getPassword() +
093          "\n\nNote that the user name and password are both case sensitive!";
094        boolean success = mailer.send(email, emailSubject, emailBody);
095        if (success) {
096          // Don't send the administrator emails about test user registration.
097          if (!userManager.isTestUser(user)) {
098            mailer.send(server.getServerProperties().get(ADMIN_EMAIL_KEY), 
099                "Hackystat 8 Admin Registration",
100                "User " + email + " registered and received password: " + user.getPassword() + "\n" +
101                "for host: " + server.getServerProperties().get(HOSTNAME_KEY));
102          }
103    
104          String responseHtml =
105            "<html>" +
106            "  <body>" +
107            "    Thank you for registering with this SensorBase. " +
108            "    <p>" +
109            "    Your password has been sent to: " + email +
110            "  </body>" +
111            "</html>";
112          server.getLogger().info("Registered: " + email + " " + user.getPassword());
113          getResponse().setStatus(Status.SUCCESS_OK);
114          Representation representation = new StringRepresentation(responseHtml);
115          representation.setMediaType(MediaType.TEXT_HTML);
116          getResponse().setEntity(representation);
117        }
118      }
119    }
