037    /**
038     * Manages access to the User resources. 
039     * Loads default definitions if available. 
040     * 
041     * Thread Safety Note: This class must NOT invoke any methods from ProjectManager or 
042     * SensorDataManager in order to avoid potential deadlock. (ProjectManager and SensorDataManager
043     * both invoke methods from UserManager, so if UserManager were to invoke a method from
044     * either of these two classes, then we would have multiple locks not being acquired in the
045     * same order, which produces the potential for deadlock.)
046     *   
047     * @author Philip Johnson
048     */
049    public class UserManager {

078      /** 
079       * The constructor for UserManagers. 
080       * @param server The Server instance associated with this UserManager. 
081       */
082      public UserManager(Server server) {
083        this.server = server;
084        this.dbManager = (DbManager)this.server.getContext().getAttributes().get("DbManager");
085        try {
086          this.jaxbContext = 
087            JAXBContext.newInstance(
088                org.hackystat.sensorbase.resource.users.jaxb.ObjectFactory.class);
089          loadDefaultUsers(); //NOPMD it's throwing a false warning. 
090          initializeCache();  //NOPMD 
091          initializeAdminUser(); //NOPMD
092        }
093        catch (Exception e) {
094          String msg = "Exception during UserManager initialization processing";
095          server.getLogger().warning(msg + "\n" + StackTrace.toString(e));
096          throw new RuntimeException(msg, e);
097        }
098      }

100      /**
101       * Loads the default Users from the defaults file and adds them to the database. 
102       * @throws Exception If problems occur. 
103       */
104      private final void loadDefaultUsers() throws Exception {
105        // Get the default User definitions from the XML defaults file. 
106        File defaultsFile = findDefaultsFile();
107        // Add these users to the database if we've found a default file. 
108        if (defaultsFile.exists()) {
109          server.getLogger().info("Loading User defaults from " + defaultsFile.getPath()); 
110          Unmarshaller unmarshaller = jaxbContext.createUnmarshaller();
111          Users users = (Users) unmarshaller.unmarshal(defaultsFile);
112          for (User user : users.getUser()) {
113            user.setLastMod(Tstamp.makeTimestamp());
114            this.dbManager.storeUser(user, this.makeUser(user), this.makeUserRefString(user));
115          }
116        }
117      }
118      
119      /** Read in all Users from the database and initialize the in-memory cache. */
120      private final void initializeCache() {
121        try {
122          UserIndex index = makeUserIndex(this.dbManager.getUserIndex());
123          for (UserRef ref : index.getUserRef()) {
124            String email = ref.getEmail();
125            String userString = this.dbManager.getUser(email);
126            User user = makeUser(userString);
127            this.updateCache(user);
128          }
129        }
130        catch (Exception e) {
131          server.getLogger().warning("Failed to initialize users " + StackTrace.toString(e));
132        }
133      }

136      /**
137       * Ensures a User exists with the admin role given the data in the sensorbase.properties file. 
138       * The admin password will be reset to what was in the sensorbase.properties file. 
139       * Note that the "admin" role is managed non-persistently: it is read into the cache from
140       * the sensorbase.properties at startup, and any persistently stored values for it are 
141       * ignored. This, of course, will eventually cause confusion. 
142       * @throws Exception if problems creating the XML string representations of the admin user.  
143       */
144      private final void initializeAdminUser() throws Exception {
145        String adminEmail = server.getServerProperties().get(ADMIN_EMAIL_KEY);
146        String adminPassword = server.getServerProperties().get(ADMIN_PASSWORD_KEY);
147        // First, clear any existing Admin role property.
148        for (User user : this.email2user.values()) {
149          user.setRole("basic");
150        }
151        // Now define the admin user with the admin property.
152        if (this.email2user.containsKey(adminEmail)) {
153          User user = this.email2user.get(adminEmail);
154          user.setPassword(adminPassword);
155          user.setRole("admin");
156        } else {
157          User admin = new User();
158          admin.setEmail(adminEmail);
159          admin.setPassword(adminPassword);
160          admin.setRole("admin");
161          this.updateCache(admin);
162        }
163      }

165      /**
166       * Updates the in-memory cache with information about this User. 
167       * @param user The user to be added to the cache.
168       * @throws Exception If problems occur updating the cache. 
169       */
170      private final void updateCache(User user) throws Exception {
171        if (user.getLastMod() == null) {
172          user.setLastMod(Tstamp.makeTimestamp());
173        }
174        updateCache(user, this.makeUser(user), this.makeUserRefString(user));
175      }
176      
177      /**
178       * Updates the cache given all the User representations.
179       * @param user The User.
180       * @param userXml The User as an XML string. 
181       * @param userRef The User as an XML reference. 
182       */
183      private void updateCache(User user, String userXml, String userRef) {
184        this.email2user.put(user.getEmail(), user);
185        this.user2xml.put(user, userXml);
186        this.user2ref.put(user, userRef);
187      }

189      /**
190       * Checks ServerProperties for the XML_DIR property.
191       * If this property is null, returns the File for ./xml/defaults/sensordatatypes.defaults.xml.
192       * @return The File instance (which might not point to an existing file.)
193       */
194      private File findDefaultsFile() {
195        String defaultsPath = "/defaults/users.defaults.xml";
196        String xmlDir = server.getServerProperties().get(XML_DIR_KEY);
197        return (xmlDir == null) ?
198            new File (System.getProperty("user.dir") + "/xml" + defaultsPath) :
199              new File (xmlDir + defaultsPath);
200      }

202      /**
203       * Returns the XML string containing the UserIndex with all defined Users.
204       * Uses the in-memory cache of UserRef strings.  
205       * @return The XML string providing an index to all current Users.
206       */
207      public synchronized String getUserIndex() {
208        StringBuilder builder = new StringBuilder(512);
209        builder.append(userIndexOpenTag);
210        for (String ref : this.user2ref.values()) {
211          builder.append(ref);
212        }
213        builder.append(userIndexCloseTag);
214        return builder.toString();
215      }

217      /**
218       * Updates the Manager with this User. Any old definition is overwritten.
219       * @param user The User.
220       */
221      public synchronized void putUser(User user) {
222        try {
223          user.setLastMod(Tstamp.makeTimestamp());
224          String xmlUser =  this.makeUser(user);
225          String xmlRef =  this.makeUserRefString(user);
226          this.updateCache(user, xmlUser, xmlRef);
227          this.dbManager.storeUser(user, xmlUser, xmlRef);
228        }
229        catch (Exception e) {
230          server.getLogger().warning("Failed to put User" + StackTrace.toString(e));
231        }
232      }

243      /**
244       * Ensures that the passed User is no longer present in this Manager. 
245       * @param email The email address of the User to remove if currently present.
246       */
247      public synchronized void deleteUser(String email) {
248        User user = this.email2user.get(email);
249        if (user != null) {
250          this.email2user.remove(email);
251          this.user2xml.remove(user);
252          this.user2ref.remove(user);
253        }
254        this.dbManager.deleteUser(email);
255      }

342      /** 
343       * Registers a User, given their email address.
344       * If a User with the passed email address exists, then return the previously registered User.
345       * Otherwise create a new User and return it.
346       * If the email address ends with the test domain, then the password will be the email.
347       * Otherwise, a unique, randomly generated 12 character key is generated as the password. 
348       * Defines the Default Project for each new user. 
349       * @param email The email address for the user. 
350       * @return The retrieved or newly created User.
351       */
352      public synchronized User registerUser(String email) {
353        // registering happens rarely, so we'll just iterate through the userMap.
354        for (User user : this.email2user.values()) {
355          if (user.getEmail().equals(email)) {
356            return user;
357          }
358        }
359        // if we got here, we need to create a new User.
360        User user = new User();
361        user.setEmail(email);
362        user.setProperties(new Properties());
363        // Password is either their Email in the case of a test user, or the randomly generated string.
364        String password = 
365          email.endsWith(server.getServerProperties().get(TEST_DOMAIN_KEY)) ? 
366              email : PasswordGenerator.make();
367        user.setPassword(password);
368        this.putUser(user);
369        return user;
370      }

409      /**
410       * Returns the passed User instance as a String encoding of its XML representation.
411       * Final because it's called in constructor.
412       * @param user The User instance. 
413       * @return The XML String representation.
414       * @throws Exception If problems occur during translation. 
415       */
416      public final synchronized String makeUser (User user) throws Exception {
417        Marshaller marshaller = jaxbContext.createMarshaller(); 
418        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
419        dbf.setNamespaceAware(true);
420        DocumentBuilder documentBuilder = dbf.newDocumentBuilder();
421        Document doc = documentBuilder.newDocument();
422        marshaller.marshal(user, doc);
423        DOMSource domSource = new DOMSource(doc);
424        StringWriter writer = new StringWriter();
425        StreamResult result = new StreamResult(writer);
426        TransformerFactory tf = TransformerFactory.newInstance();
427        Transformer transformer = tf.newTransformer();
428        transformer.transform(domSource, result);
429        String xmlString = writer.toString();
430        // Now remove the processing instruction.  This approach seems like a total hack.
431        xmlString = xmlString.substring(xmlString.indexOf('>') + 1);
432        return xmlString;
433      }
