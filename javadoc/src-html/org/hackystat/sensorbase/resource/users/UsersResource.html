001    package org.hackystat.sensorbase.resource.users;
002    
003    import org.hackystat.sensorbase.resource.sensorbase.SensorBaseResource;
004    import org.restlet.Context;
005    import org.restlet.data.MediaType;
006    import org.restlet.data.Request;
007    import org.restlet.data.Response;
008    import org.restlet.data.Status;
009    import org.restlet.resource.Representation;
010    import org.restlet.resource.Variant;
011    
012    /**
013     * Implements a Restlet Resource representing an index of Hackystat Users. 
014     * @author Philip Johnson
015     */
016    public class UsersResource extends SensorBaseResource {
017      
018      /**
019       * Provides the following representational variants: TEXT_XML.
020       * @param context The context.
021       * @param request The request object.
022       * @param response The response object.
023       */
024      public UsersResource(Context context, Request request, Response response) {
025        super(context, request, response);
026      }
027      
028      /**
029       * Returns the representation of an index of all the users in the system. 
030       * Only the administrator can request this resource. 
031       * @param variant The representational variant requested.
032       * @return The representation. 
033       */
034      @Override
035      public Representation getRepresentation(Variant variant) {
036        if (!super.userManager.isAdmin(this.authUser)) {
037          String msg = "Only the admin can obtain the index of all users.";
038          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
039          return null;
040        }
041        if (variant.getMediaType().equals(MediaType.TEXT_XML)) {
042          String xmlData = super.userManager.getUserIndex();
043          return super.getStringRepresentation(xmlData);      
044        }
045        return null;
046      }
047    }
