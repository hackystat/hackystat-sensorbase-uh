001    package org.hackystat.sensorbase.resource.users;
002    
003    import org.hackystat.utilities.stacktrace.StackTrace;
004    import org.hackystat.sensorbase.resource.sensorbase.SensorBaseResource;
005    import org.hackystat.sensorbase.resource.users.jaxb.Properties;
006    import org.hackystat.sensorbase.resource.users.jaxb.User;
007    import org.restlet.Context;
008    import org.restlet.data.MediaType;
009    import org.restlet.data.Request;
010    import org.restlet.data.Response;
011    import org.restlet.data.Status;
012    import org.restlet.resource.Representation;
013    import org.restlet.resource.Variant;
014    
015    /**
016     * Implements a Restlet Resource for manipulating individual User resources. 
017     * @author Philip Johnson
018     */
019    public class UserResource extends SensorBaseResource {
020      
021      /**
022       * Provides the following representational variants: TEXT_XML.
023       * @param context The context.
024       * @param request The request object.
025       * @param response The response object.
026       */
027      public UserResource(Context context, Request request, Response response) {
028        super(context, request, response);
029      }
030      
031      /**
032       * Returns the representation of the User resource when requested via GET.
033       * Only the authenticated user (or the admin) can request their User resource.
034       * @param variant The representational variant requested.
035       * @return The representation. 
036       */
037      @Override
038      public Representation getRepresentation(Variant variant) {
039        if (!super.userManager.isUser(this.uriUser)) {
040          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown: " + this.uriUser);
041          return null;
042        } 
043        if (!super.userManager.isAdmin(this.authUser) && !this.uriUser.equals(this.authUser)) {
044          String msg = "User is not admin and authenticated user does not not match user in URI";
045          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
046          return null;
047        }
048        if (variant.getMediaType().equals(MediaType.TEXT_XML)) {
049          String xmlData = super.userManager.getUserString(this.uriUser);
050          return super.getStringRepresentation(xmlData);
051        }
052        return null;
053      }
054      
055      
056      /** 
057       * Indicate the DELETE method is supported. 
058       * @return True.
059       */
060      @Override
061      public boolean allowDelete() {
062        return true;
063      }
064      
065      /**
066       * Implement the DELETE method that deletes an existing User given their email.
067       * Only the authenticated user (or the admin) can delete their User resource.
068       */
069      @Override
070      public void delete() {
071        if (!super.userManager.isAdmin(this.authUser) && !this.uriUser.equals(this.authUser)) {
072          String msg = "User is not admin and authenticated user does not not match user in URI";
073          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
074          return;
075        }
076        super.userManager.deleteUser(uriUser);      
077        getResponse().setStatus(Status.SUCCESS_OK);
078      }
079      
080      /** 
081       * Indicate the POST method is supported. 
082       * @return True.
083       */
084      @Override
085      public boolean allowPost() {
086        return true;
087      }
088      
089      /**
090       * Implement the POST method that updates the properties associated with a user.
091       * <ul> 
092       * <li> The User must be currently defined in this UserManager.
093       * <li> Only the authenticated User or the Admin can update their user's properties. 
094       * <li> The payload must be an XML representation of a Properties instance.
095       * </ul>
096       * @param entity The entity to be posted.
097       */
098      @Override
099      public void post(Representation entity) {
100        // Return failure if the User doesn't exist.
101        if (!super.userManager.isUser(this.uriUser)) {
102          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown User: " + this.uriUser);
103          return;
104        }
105        if (!super.userManager.isAdmin(this.uriUser) && !this.uriUser.equals(this.authUser)) {
106          String msg = "User is not admin and authenticated user does not not match user in URI";
107          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
108          return;
109        }
110        // Attempt to construct a Properties object.
111        String entityString = null;
112        Properties newProperties;
113        // Try to make the XML payload into a Properties instance, return failure if this fails. 
114        try { 
115          entityString = entity.getText();
116          newProperties = super.userManager.makeProperties(entityString);
117        }
118        catch (Exception e) {
119          server.getLogger().warning("Bad Properties Definition: " + StackTrace.toString(e));
120          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad Properties: " + entityString);
121          return;
122        }
123        User user = super.userManager.getUser(this.uriUser);
124        super.userManager.updateProperties(user, newProperties);
125        getResponse().setStatus(Status.SUCCESS_OK);
126      }
127    }
