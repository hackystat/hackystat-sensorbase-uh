021    /**
022     * Tests the SensorBase REST API for Users and User resources.
023     * @author Philip M. Johnson
024     */
025    public class TestUsersRestApi extends SensorBaseRestApiHelper {
026    
027      /** The TestUser user email. */
028      private String testUser = "TestUser@hackystat.org";
029    
030    
031      /**
032       * Test that GET host/sensorbase/users returns an index containing TestUser and
033       * that the HREFs are OK. 
034       * @throws Exception If problems occur.
035       */
036      @Test public void getUsersIndex() throws Exception {  
037        // Create an admin client and check authentication.
038        SensorBaseClient client = new SensorBaseClient(getHostName(), adminEmail, adminPassword);
039        client.authenticate();
040        // Get the index of all users. 
041        UserIndex userIndex = client.getUserIndex();
042        // Make sure that we can iterate through the Users, find the test user, and dereference hrefs. 
043        boolean foundTestUser = false;
044        for (UserRef ref : userIndex.getUserRef()) {
045          if (testUser.equals(ref.getEmail())) {
046            foundTestUser = true;
047          }
048          // Make sure the href is OK. 
049          client.getUser(ref);
050        }
051        assertTrue("Checking that we found the TestUser", foundTestUser);
052      }
053      
054      /**
055       * Test that the SensorBaseClient.isHost() method operates correctly.
056       * @throws Exception If problems occur.
057       */
058      @Test public void isHost() throws Exception {
059        assertTrue("Checking isHost() with sensorbase", SensorBaseClient.isHost(getHostName()));
060        assertFalse("Checking isHost() with bogus URL", SensorBaseClient.isHost("http://Foo"));
061      }
062      
063      /**
064       * Test that the SensorBaseClient.isRegistered() method operates correctly.
065       * @throws Exception If problems occur.
066       */
067      @Test public void isRegistered() throws Exception {
068        assertTrue("OK register", SensorBaseClient.isRegistered(getHostName(), testUser, testUser));
069        assertFalse("Bad register", SensorBaseClient.isRegistered(getHostName(), "foo", "bar"));
070      }
071    
072      /**
073       * Test that GET host/sensorbase/users/TestUser@hackystat.org returns the TestUser test user. 
074       * @throws Exception If problems occur.
075       */
076      @Test public void getUser() throws Exception {
077        // Create the TestUser client and check authentication.
078        SensorBaseClient client = new SensorBaseClient(getHostName(), testUser, testUser);
079        client.authenticate();
080        // Retrieve the TestUser User resource and test a couple of fields.
081        User user = client.getUser(testUser);
082        assertEquals("Bad email", testUser, user.getEmail());
083        assertEquals("Bad password", testUser, user.getPassword());
084      }
085      
086      /**
087       * Tests the POST method that registers a new user. 
088       * @throws Exception If problems occur.
089       */
090      @Test public void registerUser() throws Exception {
091        // Register the TestPost@hackystat.org user.
092        String testPost = "TestPost@" + server.getServerProperties().get(TEST_DOMAIN_KEY);
093        SensorBaseClient.registerUser(getHostName(), testPost);
094        // Now that TestPost is registered, see if we can retrieve him (her?) 
095        SensorBaseClient client = new SensorBaseClient(getHostName(), testPost, testPost);
096        client.authenticate();
097        User user = client.getUser(testPost);
098        assertEquals("Bad email", testPost, user.getEmail());
099        // Clean up, get rid of this user. 
100        client.deleteUser(testPost);
101      }
102      
103      /**
104       * Tests that a user can be deleted after creation. 
105       * @throws Exception If problems occur. 
106       */
107      @Test public void deleteUser () throws Exception {
108        // Register the TestPost@hackystat.org user
109        String testPost = "TestPost@" + server.getServerProperties().get(TEST_DOMAIN_KEY);
110        SensorBaseClient.registerUser(getHostName(), testPost);
111        // Now that TestPost is registered, see if we can delete him (her?) 
112        SensorBaseClient client = new SensorBaseClient(getHostName(), testPost, testPost);
113        client.deleteUser(testPost);
114    
115        //Ensure that TestPost is no longer found as a user.
116        try {
117          client.authenticate();
118          fail("Authentication should not have succeeded.");
119        }
120        catch (SensorBaseClientException e) { //NOPMD
121          // good, we got here. 
122          // We can't use the JUnit annotation idiom because the code above us could throw
123          // the same exception type, and that would be a valid error. 
124        }
125      }
126      
127      /**
128       * Tests that a user can have their properties updated. 
129       * @throws Exception If problems occur. 
130       */
131      @Test public void postUserProperties () throws Exception {
132        // Register the TestPost@hackystat.org user.
133        String testPost = "TestPost@" + server.getServerProperties().get(TEST_DOMAIN_KEY);
134        SensorBaseClient.registerUser(getHostName(), testPost);
135        // Now that TestPost is registered, see if we can update his properties. 
136        Properties properties = new Properties();
137        Property property = new Property();
138        property.setKey("testKey");
139        property.setValue("testValue");
140        properties.getProperty().add(property);
141        
142        SensorBaseClient client = 
143          new SensorBaseClient(getHostName(), testPost, testPost);
144        client.updateUserProperties(testPost, properties);
145    
146        User user = client.getUser(testPost);
147        Property theProperty = user.getProperties().getProperty().get(0);
148        assertEquals("Got the property", "testValue", theProperty.getValue());
149        // Clean up, get rid of this user. 
150        client.deleteUser(testPost);
151      }
