207      /**
208       * Test that a batch PUT of sensor data works. 
209       * @throws Exception If problems occur.
210       */
211      @Test public void putBatchSensorData() throws Exception {
212        // First, create a sample sensor data instance.
213        XMLGregorianCalendar tstamp1 = Tstamp.makeTimestamp("2007-04-30T02:00:00.123");
214        XMLGregorianCalendar tstamp2 = Tstamp.makeTimestamp("2007-04-30T02:00:00.124");
215        SensorData data1 = makeSensorData(tstamp1, user);
216        SensorData data2 = makeSensorData(tstamp2, user);
217        
218        SensorDatas batchData = new SensorDatas();
219        batchData.getSensorData().add(data1);
220        batchData.getSensorData().add(data2);
221        
222        // Create the TestUser client and check authentication.
223        SensorBaseClient client = new SensorBaseClient(getHostName(), user, user);
224        client.authenticate();
225        // Send the sensor data.
226        client.putSensorDataBatch(batchData);
227    
228        // Now see that we can retrieve it and check a field for equality. 
229        SensorData data3 = client.getSensorData(user, tstamp1);
230        assertEquals("Checking data timestamp field", tstamp1, data3.getTimestamp());
231        
232        // Delete this sensor data. 
233        client.deleteSensorData(user, tstamp1);
234        client.deleteSensorData(user, tstamp2);
235      }

237      /**
238       * Creates a sample SensorData instance given a timestamp and a user. 
239       * @param tstamp The timestamp.
240       * @param user The user.
241       * @return The new SensorData instance. 
242       */
243      private SensorData makeSensorData(XMLGregorianCalendar tstamp, String user) {
244        String sdt = "TestSdt";
245        SensorData data = new SensorData();
246        String tool = "Subversion";
247        data.setTool(tool);
248        data.setOwner(user);
249        data.setSensorDataType(sdt);
250        data.setTimestamp(tstamp);
251        data.setResource("file://foo/bar/baz.txt");
252        data.setRuntime(tstamp);
253        Property property = new Property();
254        property.setKey("SampleField");
255        property.setValue("The test value for Sample Field");
256        Properties properties = new Properties();
257        properties.getProperty().add(property);
258        data.setProperties(properties);
259        return data;
260      }
