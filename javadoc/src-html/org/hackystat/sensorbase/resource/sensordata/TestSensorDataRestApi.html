054      /**
055       * Test that DELETE host/sensorbase/sensordata/user deletes all data. 
056       * @throws Exception If problems occur.
057       */
058      @Test public void deleteAllData() throws Exception {
059        String testDeleteUser = "TestDeleteUser@hackystat.org";
060        SensorBaseClient.registerUser(getHostName(), testDeleteUser);
061        SensorBaseClient client = new SensorBaseClient(getHostName(), testDeleteUser, testDeleteUser);
062        client.authenticate();
063        // put some sensor data.
064        // Send a couple of sensor data instances.
065        for (int i = 0; i < 10; i++) {
066          client.putSensorData(makeSensorData(Tstamp.makeTimestamp(), testDeleteUser));
067        }
068        // Make sure we have the 10 instances. 
069        SensorDataIndex index = client.getSensorDataIndex(testDeleteUser);
070        assertTrue("Checking delete data size", index.getSensorDataRef().size() >= 10);
071        // Now delete it.
072        client.deleteSensorData(testDeleteUser);
073        // Now check that it's been deleted.
074        index = client.getSensorDataIndex(testDeleteUser);
075        assertEquals("Checking deleted data gone", 0, index.getSensorDataRef().size());
076      }

231      /**
232       * Test that a batch PUT of sensor data works. 
233       * @throws Exception If problems occur.
234       */
235      @Test public void putBatchSensorData() throws Exception {
236        // First, create a sample sensor data instance.
237        XMLGregorianCalendar tstamp1 = Tstamp.makeTimestamp("2007-04-30T02:00:00.123");
238        XMLGregorianCalendar tstamp2 = Tstamp.makeTimestamp("2007-04-30T02:00:00.124");
239        SensorData data1 = makeSensorData(tstamp1, user);
240        SensorData data2 = makeSensorData(tstamp2, user);
241        
242        SensorDatas batchData = new SensorDatas();
243        batchData.getSensorData().add(data1);
244        batchData.getSensorData().add(data2);
245        
246        // Create the TestUser client and check authentication.
247        SensorBaseClient client = new SensorBaseClient(getHostName(), user, user);
248        client.authenticate();
249        // Send the sensor data.
250        client.putSensorDataBatch(batchData);
251    
252        // Now see that we can retrieve it and check a field for equality. 
253        SensorData data3 = client.getSensorData(user, tstamp1);
254        assertEquals("Checking data timestamp field", tstamp1, data3.getTimestamp());
255        
256        // Delete this sensor data. 
257        client.deleteSensorData(user, tstamp1);
258        client.deleteSensorData(user, tstamp2);
259      }

261      /**
262       * Creates a sample SensorData instance given a timestamp and a user. 
263       * @param tstamp The timestamp.
264       * @param user The user.
265       * @return The new SensorData instance. 
266       */
267      private SensorData makeSensorData(XMLGregorianCalendar tstamp, String user) {
268        String sdt = "TestSdt";
269        SensorData data = new SensorData();
270        String tool = "Subversion";
271        data.setTool(tool);
272        data.setOwner(user);
273        data.setSensorDataType(sdt);
274        data.setTimestamp(tstamp);
275        data.setResource("file://foo/bar/baz.txt");
276        data.setRuntime(tstamp);
277        Property property = new Property();
278        property.setKey("SampleField");
279        property.setValue("The test value for Sample Field");
280        Properties properties = new Properties();
281        properties.getProperty().add(property);
282        data.setProperties(properties);
283        return data;
284      }
