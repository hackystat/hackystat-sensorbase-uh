124      /**
125       * Test that PUT and DELETE of 
126       * host/sensorbase/sensordata/TestUser@hackystat.org/2007-04-30T02:00:00.000 works.
127       * @throws Exception If problems occur.
128       */
129      @Test public void putSensorData() throws Exception {
130        // First, create a sample sensor data instance.
131        XMLGregorianCalendar tstamp = Tstamp.makeTimestamp("2007-04-30T02:00:00.000");
132        SensorData data = makeSensorData(tstamp, user);
133        
134        // Create the TestUser client and check authentication.
135        SensorBaseClient client = new SensorBaseClient(getHostName(), user, user);
136        client.authenticate();
137        // Send the sensor data.
138        client.putSensorData(data);
139    
140        // Now see that we can retrieve it and check a field for equality. 
141        SensorData data2 = client.getSensorData(user, tstamp);
142        assertEquals("Checking data timestamp field", tstamp, data2.getTimestamp());
143        
144        // Test that DELETE gets rid of this sensor data.
145        client.deleteSensorData(user, tstamp);
146        
147        // Test that a second DELETE succeeds, even though da buggah is no longer in there.
148        client.deleteSensorData(user, tstamp);
149      }

151      /**
152       * Test that a batch PUT of sensor data works. 
153       * @throws Exception If problems occur.
154       */
155      @Test public void putBatchSensorData() throws Exception {
156        // First, create a sample sensor data instance.
157        XMLGregorianCalendar tstamp1 = Tstamp.makeTimestamp("2007-04-30T02:00:00.123");
158        XMLGregorianCalendar tstamp2 = Tstamp.makeTimestamp("2007-04-30T02:00:00.124");
159        SensorData data1 = makeSensorData(tstamp1, user);
160        SensorData data2 = makeSensorData(tstamp2, user);
161        
162        SensorDatas batchData = new SensorDatas();
163        batchData.getSensorData().add(data1);
164        batchData.getSensorData().add(data2);
165        
166        // Create the TestUser client and check authentication.
167        SensorBaseClient client = new SensorBaseClient(getHostName(), user, user);
168        client.authenticate();
169        // Send the sensor data.
170        client.putSensorDataBatch(batchData);
171    
172        // Now see that we can retrieve it and check a field for equality. 
173        SensorData data3 = client.getSensorData(user, tstamp1);
174        assertEquals("Checking data timestamp field", tstamp1, data3.getTimestamp());
175        
176        // Delete this sensor data. 
177        client.deleteSensorData(user, tstamp1);
178        client.deleteSensorData(user, tstamp2);
179      }

181      /**
182       * Creates a sample SensorData instance given a timestamp and a user. 
183       * @param tstamp The timestamp.
184       * @param user The user.
185       * @return The new SensorData instance. 
186       */
187      private SensorData makeSensorData(XMLGregorianCalendar tstamp, String user) {
188        String sdt = "TestSdt";
189        SensorData data = new SensorData();
190        String tool = "Subversion";
191        data.setTool(tool);
192        data.setOwner(user);
193        data.setSensorDataType(sdt);
194        data.setTimestamp(tstamp);
195        data.setResource("file://foo/bar/baz.txt");
196        data.setRuntime(tstamp);
197        Property property = new Property();
198        property.setKey("SampleField");
199        property.setValue("The test value for Sample Field");
200        Properties properties = new Properties();
201        properties.getProperty().add(property);
202        data.setProperties(properties);
203        return data;
204      }

207       * Tests the makeSensorData method.
208       * @throws Exception If problems occur.
209       */
210      @Test
211      public void testMakeSensorData() throws Exception {
212        Map<String, String> keyValMap = new HashMap<String, String>();
213        // Create the TestUser client.
214        SensorBaseClient client = new SensorBaseClient(getHostName(), user, user);
215        // See that we can create a SensorData instance with all defaults.
216        client.makeSensorData(keyValMap);
217        // Add a couple of fields and make a new one.
218        String tool = "Eclipse";
219        keyValMap.put("Tool", tool);
220        keyValMap.put("MyProperty", "foobar");
221        SensorData data = client.makeSensorData(keyValMap);
222        assertEquals("Checking sensor data val", tool, data.getTool());
223      }
