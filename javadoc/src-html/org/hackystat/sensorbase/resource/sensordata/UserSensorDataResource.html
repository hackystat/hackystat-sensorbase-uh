021    /**
022     * The resource for all URIs that extend sensordata, including;
023     * <ul>
024     * <li> host/sensordata/{user}
025     * <li> host/sensordata/{user}?sdt={sensordatatype}
026     * <li> host/sensordata/{user}/{timestamp}
027     * </ul>
028     * 
029     * @author Philip Johnson
030     */
031    public class UserSensorDataResource extends SensorBaseResource {
032    
033      /** The user, or null if not found. */
034      private User user; 
035      /** To be retrieved from the URL, or else null if not found. */
036      private String sdtName;
037      /** To be retrieved from the URL, or else null if not found. */
038      private String timestamp;
039      
040      /**
041       * Provides the following representational variants: TEXT_XML.
042       * @param context The context.
043       * @param request The request object.
044       * @param response The response object.
045       */
046      public UserSensorDataResource(Context context, Request request, Response response) {
047        super(context, request, response);
048        this.user = super.userManager.getUser(uriUser);
049        this.sdtName = (String) request.getAttributes().get("sensordatatype");
050        this.timestamp = (String) request.getAttributes().get("timestamp");
051      }
052      
053      /**
054       * Returns a SensorDataIndex when a GET is called with:
055       * <ul>
056       * <li> sensordata/{email}
057       * <li> sensordata/{email}?sdt={sensordatatype}
058       * </ul>
059       * Returns a SensorData when a GET is called with:
060       * <ul>
061       * <li> sensordata/{email}/{timestamp}
062       * </ul>
063       * <p>
064       * The user must be defined, and the authenticated user must be the uriUser or the Admin.
065       * @param variant The representational variant requested.
066       * @return The representation. 
067       */
068      @Override
069      public Representation getRepresentation(Variant variant) {
070        if (this.user == null) {
071          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown user: " + this.uriUser);
072          return null;
073        } 
074        if (!super.userManager.isAdmin(this.authUser) && !this.uriUser.equals(this.authUser)) {
075          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, super.badAuth);
076          return null;
077        }
078        if (variant.getMediaType().equals(MediaType.TEXT_XML)) {
079          // sensordata/{email}
080          if ((this.sdtName == null) && (this.timestamp == null)) {
081            String xmlData = super.sensorDataManager.getSensorDataIndex(this.user);
082            return super.getStringRepresentation(xmlData);
083          }
084          // sensordata/{email}?sdt={sensordatatype}
085          else if (this.timestamp == null) {
086            String xmlData = super.sensorDataManager.getSensorDataIndex(this.user);
087            return super.getStringRepresentation(xmlData);
088          }
089          // sensordata/{email}/{timestamp}
090          else {
091            // First, try to parse the timestamp string, and return error if it doesn't parse.
092            XMLGregorianCalendar tstamp;
093            try {
094              tstamp = Tstamp.makeTimestamp(this.timestamp);
095            }
096            catch (Exception e) {
097              getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad Timestamp " + timestamp);
098              return null;
099            }
100            // Now, see if we actually have the SensorData.
101            if (!super.sensorDataManager.hasSensorData(user, tstamp)) {
102              getResponse().setStatus(Status.CLIENT_ERROR_NOT_FOUND, "Unknown Sensor Data");
103              return null;
104            }
105            // We have the SensorData, so retrieve its xml string representation and return it.
106            try {
107              String xmlData = super.sensorDataManager.getSensorData(this.user, tstamp);
108              return getStringRepresentation(xmlData);
109            }
110            catch (Exception e) {
111              // The marshallSensorData threw an exception, which is unrecoverable.
112              return null;
113            }
114          }
115        }
116        return null;
117      }
118      
119      /** 
120       * Indicate the PUT method is supported. 
121       * @return True.
122       */
123      @Override
124      public boolean allowPut() {
125          return true;
126      }
127    
128      /**
129       * Implement the PUT method that creates a new sensor data instance.
130       * <ul>
131       * <li> The XML must be marshallable into a sensor data instance.
132       * <li> The timestamp in the URL must match the timestamp in the XML.
133       * <li> The User and SDT must exist.
134       * </ul>
135       * Note that we are not validating that this sensor data instance contains all of the 
136       * Required Properties specified by the SDT.  This should be done later, on demand, as part of
137       * analyses. 
138       * <p>
139       * We are also not at this point checking to see whether the User and SDT exist. 
140       * @param entity The XML representation of the new sensor data instance.. 
141       */
142      @Override
143      public void put(Representation entity) {
144        // If this User does not exist, return an error.
145        if (this.user == null) {
146          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown user: " + this.uriUser);
147          return;
148        } 
149        if (!super.userManager.isAdmin(this.uriUser) && !this.uriUser.equals(this.authUser)) {
150          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, super.badAuth);
151          return;
152        }
153        
154        // Get the payload.
155        String entityString = null;
156        try {
157          entityString = entity.getText();
158        }
159        catch (IOException e) {
160          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad payload.");
161          return;
162        }
163        
164        //if the "timestamp" is "batch", then our payload is <SensorDatas>, otherwise <SensorData>
165        if (this.timestamp.equals("batch")) {
166          putSensorDatas(entityString);
167        }
168        else {
169          putSensorData(entityString);
170        }
171      }
172      
173      /**
174       * Put a SensorData XML payload.
175       * @param entityString An entity string that should represent a SensorData instance.
176       */
177      private void putSensorData(String entityString) {
178        SensorData data;
179        // Try to make the XML payload into sensor data, return failure if this fails. 
180        try { 
181          data = this.sensorDataManager.makeSensorData(entityString);
182        }
183        catch (Exception e) {
184          server.getLogger().warning("Bad Sensor Data in PUT: " + StackTrace.toString(e));
185          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad Sensor Data: " + entityString);
186          return;
187        }
188        // Return failure if the payload XML timestamp doesn't match the URI timestamp.
189        if ((this.timestamp == null) || (!this.timestamp.equals(data.getTimestamp().toString()))) {
190          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Inconsistent timestamps");
191          return;
192        }
193        // Return failure if the SensorData Owner doesn't match the UriUser
194        if (!this.uriUser.equals(super.sensorDataManager.convertOwnerToEmail(data.getOwner()))) {
195          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, 
196              "SensorData payload owner field does not match user field in URI");
197          return;
198        }
199        // otherwise we add it to the Manager and return success.
200        super.sensorDataManager.putSensorData(data);      
201        getResponse().setStatus(Status.SUCCESS_CREATED);
202      }
203      
204      /**
205       * Put a SensorDatas payload.
206       * @param entityString An entity string that should represent a SensorDatas instance. 
207       */
208      private void putSensorDatas(String entityString) {
209        SensorDatas datas;
210        // Try to make the XML payload into a collection of SensorData, return failure if this fails. 
211        try { 
212          datas = this.sensorDataManager.makeSensorDatas(entityString);
213        }
214        catch (Exception e) {
215          server.getLogger().warning("Bad Sensor Data in PUT: " + StackTrace.toString(e));
216          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, 
217              "Bad Batch SensorData: " + entityString);
218          return;
219        }
220        // Makes sure all SensorData matches the UriUser, otherwise fail and don't add anything.
221        for (SensorData data : datas.getSensorData()) {
222          if (!this.uriUser.equals(super.sensorDataManager.convertOwnerToEmail(data.getOwner()))) {
223            getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, 
224                "At least one SensorData owner field does not match user field in URI");
225            return;
226          }
227        }
228        // Otherwise we should be OK.  Add all of them and return success.
229        for (SensorData data : datas.getSensorData()) {
230          super.sensorDataManager.putSensorData(data);      
231        }
232        getResponse().setStatus(Status.SUCCESS_CREATED);
233      }

242          return true;
243      }

245      /**
246       * Implement the DELETE method that ensures the specified sensor data instance no longer exists.
247       * <ul>
248       * <li> The user must exist.
249       * <li> The timestamp must be well-formed.
250       * <li> The authenticated user must be the uriUser or the admin. 
251       * </ul>
252       */
253      @Override
254      public void delete() {
255        // If this User does not exist, return an error.
256        if (this.user == null) {
257          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown user: " + this.uriUser);
258          return;
259        } 
260        if (!super.userManager.isAdmin(this.uriUser) && !this.uriUser.equals(this.authUser)) {
261          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, super.badAuth);
262          return;
263        }
264        XMLGregorianCalendar tstamp;
265        try {
266          tstamp = Tstamp.makeTimestamp(this.timestamp);
267        }
268        catch (Exception e) { 
269          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad timestamp " + this.timestamp);
270          return;
271        }
272        super.sensorDataManager.deleteData(this.user, tstamp);      
273        getResponse().setStatus(Status.SUCCESS_OK);
274      }
