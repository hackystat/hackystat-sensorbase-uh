021    /**
022     * The resource for all URIs that extend sensordata, including;
023     * <ul>
024     * <li> host/sensordata/{user}
025     * <li> host/sensordata/{user}?sdt={sensordatatype}
026     * <li> host/sensordata/{user}?lastModStartTime={lastModTimestamp}&lastModEndTime={lastModTimestamp}
027     * <li> host/sensordata/{user}/{timestamp}
028     * </ul>
029     * 
030     * @author Philip Johnson
031     */
032    public class UserSensorDataResource extends SensorBaseResource {
033    
034      /** The user, or null if not found. */
035      private User user; 
036      /** To be retrieved from the URL, or else null if not found. */
037      private String sdtName;
038      /** To be retrieved from the URL, or else null if not found. */
039      private String timestamp;
040      /** To be retrieved from the URL, or else null if not found. */
041      private String lastModStartTime;
042      /** To be retrieved from the URL, or else null if not found. */
043      private String lastModEndTime;
044      
045      /**
046       * Provides the following representational variants: TEXT_XML.
047       * @param context The context.
048       * @param request The request object.
049       * @param response The response object.
050       */
051      public UserSensorDataResource(Context context, Request request, Response response) {
052        super(context, request, response);
053        this.user = super.userManager.getUser(uriUser);
054        this.sdtName = (String) request.getAttributes().get("sensordatatype");
055        this.timestamp = (String) request.getAttributes().get("timestamp");
056        this.lastModStartTime = (String) request.getAttributes().get("lastModStartTime");
057        this.lastModEndTime = (String) request.getAttributes().get("lastModEndTime");
058      }
059      
060      /**
061       * Returns a SensorDataIndex when a GET is called with:
062       * <ul>
063       * <li> sensordata/{email}
064       * <li> sensordata/{email}?sdt={sensordatatype}
065       * <li> sensordata/{user}?lastModStartTime={timestamp}&lastModEndTime={timestamp}
066       * </ul>
067       * Returns a SensorData when a GET is called with:
068       * <ul>
069       * <li> sensordata/{email}/{timestamp}
070       * </ul>
071       * <p>
072       * The user must be defined, and the authenticated user must be the uriUser or the Admin.
073       * @param variant The representational variant requested.
074       * @return The representation. 
075       */
076      @Override
077      public Representation getRepresentation(Variant variant) {
078        // Return error if unknown user.
079        if (this.user == null) {
080          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown user: " + this.uriUser);
081          return null;
082        } 
083        // Return error if authUser is not the UriUser but it's not the admin.
084        if (!super.userManager.isAdmin(this.authUser) && !this.uriUser.equals(this.authUser)) {
085          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, super.badAuth);
086          return null;
087        }
088        // Now check to make sure they want XML.
089        if (variant.getMediaType().equals(MediaType.TEXT_XML)) {
090          // Return index of all data for URI: sensordata/{email} 
091          if ((this.sdtName == null) && (this.timestamp == null) && (this.lastModStartTime == null)) {
092            String xmlData = super.sensorDataManager.getSensorDataIndex(this.user);
093            return super.getStringRepresentation(xmlData);
094          }
095          // Return index of data for a given SDT for URI: sensordata/{email}?sdt={sensordatatype}
096          if ((this.sdtName != null) && (this.timestamp == null) && (this.lastModStartTime == null)) {
097              String xmlData = super.sensorDataManager.getSensorDataIndex(this.user, this.sdtName);
098              return super.getStringRepresentation(xmlData);
099          } 
100          // Return index of data since the tstamp for URI: 
101          // sensordata/{user}?lastModStartTime={timestamp}&lastModEndTime={timestamp}
102          if ((this.sdtName == null) && (this.timestamp == null) && (this.lastModStartTime != null)) {
103            // First, check to see that we can convert the lastModStartTime and EndTime into a tstamp.
104            XMLGregorianCalendar lastModStart;
105            XMLGregorianCalendar lastModEnd;
106            try {
107              lastModStart = Tstamp.makeTimestamp(this.lastModStartTime);
108            }
109            catch (Exception e) {
110              getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad StartTime " 
111                  + lastModStartTime);
112              return null;
113            }
114            try {
115              lastModEnd = Tstamp.makeTimestamp(this.lastModEndTime);
116            }
117            catch (Exception e) {
118              getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad EndTime " 
119                  + lastModEndTime);
120              return null;
121            }
122            // Now, get the data and return.
123            String xmlData = 
124              super.sensorDataManager.getSensorDataIndexLastMod(this.user, lastModStart, lastModEnd);
125            return super.getStringRepresentation(xmlData);
126          } 
127          // Return sensordata representation for URI: sensordata/{email}/{timestamp}
128          if ((this.sdtName == null) && (this.timestamp != null) && (this.lastModStartTime == null)) {
129            // First, try to parse the timestamp string, and return error if it doesn't parse.
130            XMLGregorianCalendar tstamp;
131            try {
132              tstamp = Tstamp.makeTimestamp(this.timestamp);
133            }
134            catch (Exception e) {
135              getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad Timestamp " + timestamp);
136              return null;
137            }
138            // Now, see if we actually have the SensorData.
139            if (!super.sensorDataManager.hasSensorData(user, tstamp)) { //NOPMD
140              getResponse().setStatus(Status.CLIENT_ERROR_NOT_FOUND, "Unknown Sensor Data");
141              return null;
142            }
143            // We have the SensorData, so retrieve its xml string representation and return it.
144            try {
145              String xmlData = super.sensorDataManager.getSensorData(this.user, tstamp);
146              return getStringRepresentation(xmlData);
147            }
148            catch (Exception e) {
149              // The marshallSensorData threw an exception, which is unrecoverable.
150              getResponse().setStatus(Status.SERVER_ERROR_INTERNAL, "Problem unmarshalling data.");
151              return null;
152            }
153          }
154        }
155        // Otherwise we don't understand the URI that they invoked us with.
156        getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Illegal URI");
157        return null;
158      }

169      /**
170       * Implement the PUT method that creates a new sensor data instance.
171       * <ul>
172       * <li> The XML must be marshallable into a sensor data instance.
173       * <li> The timestamp in the URL must match the timestamp in the XML.
174       * <li> The User and SDT must exist.
175       * </ul>
176       * Note that we are not validating that this sensor data instance contains all of the 
177       * Required Properties specified by the SDT.  This should be done later, on demand, as part of
178       * analyses. 
179       * <p>
180       * We are also not at this point checking to see whether the User and SDT exist. 
181       * @param entity The XML representation of the new sensor data instance.. 
182       */
183      @Override
184      public void put(Representation entity) {
185        // If this User does not exist, return an error.
186        if (this.user == null) {
187          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown user: " + this.uriUser);
188          return;
189        } 
190        if (!super.userManager.isAdmin(this.uriUser) && !this.uriUser.equals(this.authUser)) {
191          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, super.badAuth);
192          return;
193        }
194        
195        // Get the payload.
196        String entityString = null;
197        try {
198          entityString = entity.getText();
199        }
200        catch (IOException e) {
201          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad payload.");
202          return;
203        }
204        
205        //if the "timestamp" is "batch", then our payload is <SensorDatas>, otherwise <SensorData>
206        if (this.timestamp.equals("batch")) {
207          putSensorDatas(entityString);
208        }
209        else {
210          putSensorData(entityString);
211        }
212      }
213      
214      /**
215       * Put a SensorData XML payload.
216       * @param entityString An entity string that should represent a SensorData instance.
217       */
218      private void putSensorData(String entityString) {
219        SensorData data;
220        // Try to make the XML payload into sensor data, return failure if this fails. 
221        try { 
222          data = this.sensorDataManager.makeSensorData(entityString);
223        }
224        catch (Exception e) {
225          server.getLogger().warning("Bad Sensor Data in PUT: " + StackTrace.toString(e));
226          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad Sensor Data: " + entityString);
227          return;
228        }
229        // Return failure if the payload XML timestamp doesn't match the URI timestamp.
230        if ((this.timestamp == null) || (!this.timestamp.equals(data.getTimestamp().toString()))) {
231          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Inconsistent timestamps");
232          return;
233        }
234        // Return failure if the SensorData Owner doesn't match the UriUser
235        if (!this.uriUser.equals(super.sensorDataManager.convertOwnerToEmail(data.getOwner()))) {
236          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, 
237              "SensorData payload owner field does not match user field in URI");
238          return;
239        }
240        // otherwise we add it to the Manager and return success.
241        super.sensorDataManager.putSensorData(data);      
242        getResponse().setStatus(Status.SUCCESS_CREATED);
243      }
244      
245      /**
246       * Put a SensorDatas payload.
247       * @param entityString An entity string that should represent a SensorDatas instance. 
248       */
249      private void putSensorDatas(String entityString) {
250        SensorDatas datas;
251        // Try to make the XML payload into a collection of SensorData, return failure if this fails. 
252        try { 
253          datas = this.sensorDataManager.makeSensorDatas(entityString);
254        }
255        catch (Exception e) {
256          server.getLogger().warning("Bad Sensor Data in PUT: " + StackTrace.toString(e));
257          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, 
258              "Bad Batch SensorData: " + entityString);
259          return;
260        }
261        // Makes sure all SensorData matches the UriUser, otherwise fail and don't add anything.
262        for (SensorData data : datas.getSensorData()) {
263          if (!this.uriUser.equals(super.sensorDataManager.convertOwnerToEmail(data.getOwner()))) {
264            getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, 
265                "At least one SensorData owner field does not match user field in URI");
266            return;
267          }
268        }
269        // Otherwise we should be OK.  Add all of them and return success.
270        for (SensorData data : datas.getSensorData()) {
271          super.sensorDataManager.putSensorData(data);      
272        }
273        getResponse().setStatus(Status.SUCCESS_CREATED);
274      }

286      /**
287       * Implement the DELETE method that ensures the specified sensor data instance no longer exists.
288       * <ul>
289       * <li> The user must exist.
290       * <li> The timestamp must be well-formed.
291       * <li> The authenticated user must be the uriUser or the admin. 
292       * </ul>
293       */
294      @Override
295      public void delete() {
296        // If this User does not exist, return an error.
297        if (this.user == null) {
298          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown user: " + this.uriUser);
299          return;
300        } 
301        if (!super.userManager.isAdmin(this.uriUser) && !this.uriUser.equals(this.authUser)) {
302          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, super.badAuth);
303          return;
304        }
305        XMLGregorianCalendar tstamp;
306        try {
307          tstamp = Tstamp.makeTimestamp(this.timestamp);
308        }
309        catch (Exception e) { 
310          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad timestamp " + this.timestamp);
311          return;
312        }
313        super.sensorDataManager.deleteData(this.user, tstamp);      
314        getResponse().setStatus(Status.SUCCESS_OK);
315      }
