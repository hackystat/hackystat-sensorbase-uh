018    /**
019     * The resource for processing GET/PUT/DELETE host/projects/{email}/{projectname}.
020     * Returns a representation of the Project resource associated with this user. 
021     * 
022     * @author Philip Johnson
023     */
024    public class UserProjectResource extends SensorBaseResource {
025      
026      /** The user, or null if the uriUser does not name a defined User. */
027      private User user; 
028      /** To be retrieved from the URL. */
029      private String projectName;
030      
031      /**
032       * Provides the following representational variants: TEXT_XML.
033       * @param context The context.
034       * @param request The request object.
035       * @param response The response object.
036       */
037      public UserProjectResource(Context context, Request request, Response response) {
038        super(context, request, response);
039        this.projectName = (String) request.getAttributes().get("projectname");
040        this.user = super.userManager.getUser(uriUser);
041      }
042      
043      /**
044       * Returns an XML representation of the Project associated with this User.
045       * <ul>
046       * <li> The uriUser must be defined as a User.
047       * <li> The Project must be defined for this User.
048       * <li> The authenticated user must be the admin, or uriUser, or a member of the project, or 
049       * invited to be in the Project.
050       * </ul>
051       * 
052       * @param variant The representational variant requested, or null if conditions are violated.
053       * @return The representation. 
054       */
055      @Override
056      public Representation getRepresentation(Variant variant) {
057        // The uriUser must be a defined User.
058        if (this.user == null) {
059          String msg = "No user corresponding to: " + this.uriUser;
060          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
061          return null;
062        }
063        // The named project must be defined.
064        if (!super.projectManager.hasProject(this.user, this.projectName)) {
065          String msg = "No Project named " + this.projectName + " for user " + this.uriUser;
066          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
067          return null;
068        }    
069        // The authorized user must be an admin, or the project owner, or a member, or invitee.
070        if (!super.userManager.isAdmin(this.authUser) && !this.uriUser.equals(this.authUser) &&
071            !super.projectManager.isMember(this.user, this.projectName, this.authUser) &&
072            !super.projectManager.isInvited(this.user, this.projectName, this.authUser)) {
073          String msg = "User " + this.authUser + "is not authorized to view this Project.";
074          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
075          return null;
076        }
077        // It's all good, so return the Project representation.
078        if (variant.getMediaType().equals(MediaType.TEXT_XML)) {
079          try {
080            String xmlData = super.projectManager.getProjectString(this.user, this.projectName);
081            return super.getStringRepresentation(xmlData);
082          }
083          catch (Exception e) {
084            String msg = "Couldn't marshall project " + this.projectName + " into XML.";
085            getResponse().setStatus(Status.SERVER_ERROR_INTERNAL, msg);
086            return null;
087          }
088        }
089        return null;
090      }
091      
092      /** 
093       * Indicate the PUT method is supported. 
094       * @return True.
095       */
096      @Override
097      public boolean allowPut() {
098          return true;
099      }
100    
101      /**
102       * Implement the PUT method that creates a new Project or updates an existing Project.
103       * <ul>
104       * <li> The XML must be marshallable into a Project instance using the Project XmlSchema.
105       * <li> The User must exist.
106       * <li> The Project name in the URI string must match the Project name in the XML.
107       * <li> The authenticated user must be the uriUser or the admin.
108       * <li> The project cannot be the Default project.   
109       * <li> All members in the new project representation must have been members previously.
110       * <li> All Members and Invitees must be defined Users.
111       * <li> No Invitee can be a Member.
112       * <li> The project owner cannot be a Member or Invitee.
113       * </ul>
114       * @param entity The XML representation of the new Project.
115       */
116      @Override
117      public void put(Representation entity) {
118        // Error if uriUser is not defined.
119        if (this.user == null) {
120          String msg = "Bad PUT: No user corresponding to: " + this.uriUser;
121          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
122          return;
123        }  
124        // Error if authorized User is not the admin or the uriUser.
125        if (!super.userManager.isAdmin(this.authUser) && !this.uriUser.equals(this.authUser)) {
126          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, super.badAuth);
127          return;
128        }
129        String entityString = null;
130        Project newProject;
131        // Try to make the XML payload into a Project, return failure if this fails. 
132        try { 
133          entityString = entity.getText();
134          newProject = super.projectManager.makeProject(entityString);
135        }
136        catch (Exception e) {
137          server.getLogger().warning("Bad Project in PUT: " + StackTrace.toString(e));
138          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad Project: " + entityString);
139          return;
140        }
141        // Error if the URI ProjectName is not the same as the XML Project name.
142        if (!(this.projectName.equals(newProject.getName()))) {
143          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Different URI/XML project names");
144          return;
145        }
146        // Error if the project is the Default project.
147        if (this.projectName.equals(ProjectManager.DEFAULT_PROJECT_NAME)) {
148          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Cannot PUT the Default project");
149          return;
150        }
151        // Error if the uriUser is not the Admin or the Project Owner
152        if (!super.userManager.isAdmin(this.uriUser) && !this.uriUser.equals(newProject.getOwner())) {
153          String msg = "User " + user.getEmail() + " is not the admin or Project Owner.";
154          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
155          return;
156        }
157        
158        // Get or create a list of new members (possibly empty).
159        List<String> newMembers = ((newProject.getMembers() == null) ? new ArrayList<String>()
160            : newProject.getMembers().getMember());
161    
162        // Error if oldProject exists and new one contains a member not in old one.
163        for (String newMember : newMembers) {
164          if (!super.projectManager.isMember(this.user, this.projectName, newMember)) {
165            String msg = "New project contains non-member: " + newMember;
166            getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
167            return;
168          }
169        }
170        
171        // Get a (possibly empty) list of the invitees.
172        List<String> newInvitees = ((newProject.getInvitations() == null) ? 
173            new ArrayList<String>() : newProject.getInvitations().getInvitation());
174    
175        // Make sure all newMembers and newInvitees are defined Users.
176        for (String member : newMembers) {
177          if (!super.userManager.isUser(member)) {
178            String msg = "Member is not a registered user: " + member;
179            getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
180            return;
181          }
182        }
183        for (String invitee : newInvitees) {
184          if (!super.userManager.isUser(invitee)) {
185            String msg = "Invited member is not a registered user: " + invitee;
186            getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
187            return;
188          }
189        }
190        // No invitee can be a (new) member.
191        for (String invitee : newInvitees) {
192          if (newMembers.contains(invitee)) {
193            String msg = "Invited member cannot already be a member: " + invitee;
194            getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
195            return;
196          }
197        }
198        // Project owner cannot be a member or invitee.
199        if (newMembers.contains(this.uriUser)) {
200          String msg = "Project owner cannot also be a member: " + this.uriUser;
201          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
202          return;
203        }
204        if (newInvitees.contains(this.uriUser)) {
205          String msg = "Project owner cannot also be an invited member: " + this.uriUser;
206          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
207          return;
208        }
209    
210        // otherwise we add it to the Manager and return success.
211        super.projectManager.putProject(newProject);      
212        getResponse().setStatus(Status.SUCCESS_CREATED);
213      }

215      /** 
216       * Indicate the DELETE method is supported. 
217       * @return True.
218       */
219      @Override
220      public boolean allowDelete() {
221          return true;
222      }
223      
224      /**
225       * Implement the DELETE method that deletes an existing Project for a given User.
226       * <ul> 
227       * <li> The User must be currently defined.
228       * <li> The authenticated user must be the uriUser or the Admin. 
229       * <li> The User must be the admin or the Owner.
230       * </ul>
231       * If the Project doesn't exist, that's fine, it's still "deleted".
232       */
233      @Override
234      public void delete() {
235        //  If this User does not exist, return an error.
236        if (this.user == null) {
237          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown user: " + this.uriUser);
238          return;
239        } 
240        if (!super.userManager.isAdmin(this.authUser) && !this.uriUser.equals(this.authUser)) {
241          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, super.badAuth);
242          return;
243        }    
244        // Otherwise, delete it and return successs.
245        super.projectManager.deleteProject(this.user, this.projectName);      
246        getResponse().setStatus(Status.SUCCESS_OK);
247      }
