001    package org.hackystat.sensorbase.resource.projects;
002    
003    import org.hackystat.sensorbase.resource.sensorbase.SensorBaseResource;
004    import org.restlet.Context;
005    import org.restlet.data.MediaType;
006    import org.restlet.data.Request;
007    import org.restlet.data.Response;
008    import org.restlet.data.Status;
009    import org.restlet.resource.Representation;
010    import org.restlet.resource.Variant;
011    
012    /**
013     * Implements the Resource for processing GET {host}/projects requests to obtain an index for
014     * all Projects defined in this SensorBase for all users.  
015     * Requires the admin user. 
016     * @author Philip Johnson
017     */
018    public class ProjectsResource extends SensorBaseResource {
019    
020      /**
021       * The standard constructor.
022       * @param context The context.
023       * @param request The request object.
024       * @param response The response object.
025       */
026      public ProjectsResource(Context context, Request request, Response response) {
027        super(context, request, response);
028      }
029      
030      /**
031       * Returns an index of all Projects for all Users, or null if the request is not authorized. 
032       * This requires admin authorization. 
033       * @param variant The representational variant requested.
034       * @return The representation. 
035       */
036      @Override
037      public Representation getRepresentation(Variant variant) {
038        if (!super.userManager.isAdmin(this.authUser)) {
039          String msg = "Only the admin can obtain the index of all users.";
040          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
041          return null;
042        }   
043        if (variant.getMediaType().equals(MediaType.TEXT_XML)) {
044          String xmlData = super.projectManager.getProjectIndex();
045          return super.getStringRepresentation(xmlData);      
046        }
047        return null;
048      }
049    }
