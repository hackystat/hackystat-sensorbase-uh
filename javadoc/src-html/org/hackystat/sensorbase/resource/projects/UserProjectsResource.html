001    package org.hackystat.sensorbase.resource.projects;
002    
003    import org.hackystat.sensorbase.resource.sensorbase.SensorBaseResource;
004    import org.hackystat.sensorbase.resource.users.jaxb.User;
005    import org.restlet.Context;
006    import org.restlet.data.MediaType;
007    import org.restlet.data.Request;
008    import org.restlet.data.Response;
009    import org.restlet.data.Status;
010    import org.restlet.resource.Representation;
011    import org.restlet.resource.Variant;
012    
013    /**
014     * The resource for processing GET host/projects/{user} requests.
015     * Returns an index of all Project resource names associated with this user. 
016     * 
017     * @author Philip Johnson
018     */
019    public class UserProjectsResource extends SensorBaseResource {
020      
021      /** The user, or null if not found. */
022      private User user;
023      
024      /**
025       * Provides the following representational variants: TEXT_XML.
026       * @param context The context.
027       * @param request The request object.
028       * @param response The response object.
029       */
030      public UserProjectsResource(Context context, Request request, Response response) {
031        super(context, request, response);
032        this.user = super.userManager.getUser(this.uriUser);
033      }
034      
035      /**
036       * Returns a ProjectIndex of all projects associated with this User.
037       * <ul>
038       * <li> The user must be defined.
039       * <li> The authenticated user must be the uriUser or the Admin. 
040       * </ul>
041       * 
042       * @param variant The representational variant requested.
043       * @return The representation. 
044       */
045      @Override
046      public Representation getRepresentation(Variant variant) {
047        // If this User does not exist, return an error.
048        if (this.user == null) {
049          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown user: " + super.uriUser);
050          return null;
051        }  
052        if (!super.userManager.isAdmin(this.authUser) && !this.uriUser.equals(this.authUser)) {
053          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, super.badAuth);
054          return null;
055        }
056        if (variant.getMediaType().equals(MediaType.TEXT_XML)) {
057          String xmlData = super.projectManager.getProjectIndex(this.user);
058          return super.getStringRepresentation(xmlData);      
059          }
060        return null;
061      }
062    }
