046        assertTrue("Checking for project data", index.getProjectRef().size() > 1);

123       * Test that the GET of a project with the SDT parameter works correctly.
124       * Assumes that the default XML data files are loaded, which result in two sensor data entries
125       * for TestUser@hackystat.org on 2007-04-30, one with SDT=TestSdt and one with SDT=SampleSdt.
126       * 
127       * @throws Exception If problems occur.
128       */
129      @Test
130      public void getTestUserProjectSdtParam() throws Exception {
131        // Create the TestUser client and check authentication.
132        SensorBaseClient client = new SensorBaseClient(getHostName(), testUser, testUser);
133        client.authenticate();
134        // Retrieve the SensorData for the TestProject project within the time
135        // interval, and make sure we have the two entries that we expect. 
136        XMLGregorianCalendar startTime = Tstamp.makeTimestamp("2007-04-30T09:00:00.000");
137        XMLGregorianCalendar endTime = Tstamp.makeTimestamp("2007-04-30T10:00:00.000");
138        SensorDataIndex index = client.getProjectSensorData(testUser, testProject, startTime, endTime);
139        assertEquals("Checking our test data has 2 entries", 2, index.getSensorDataRef().size());
140        
141        // Now test the SDT param call to make sure we get only the SDT of interest.
142        index = client.getProjectSensorData(testUser, testProject, startTime, endTime, "TestSdt");
143        assertEquals("Checking our test data has 1 entries", 1, index.getSensorDataRef().size());
144      }

146      /**
147       * Test that PUT and DELETE of host/projects/{user}/{project} works.
148       * 
149       * @throws Exception If problems occur.
150       */
151      @Test
152      public void putProject() throws Exception {
153        // First, create a sample Project.
154        String owner = testUser;
155        String projectName = "TestProject1";
156        Project project = new Project();
157        project.setOwner(owner);
158        project.setName(projectName);
159        project.setDescription("Test Project1");
160        XMLGregorianCalendar tstamp = Tstamp.makeTimestamp();
161        project.setStartTime(tstamp);
162        project.setEndTime(tstamp);
163        UriPatterns uris = new UriPatterns();
164        uris.getUriPattern().add("**/test/**");
165        project.setUriPatterns(uris);
166    
167        // Create the TestUser client, check authentication, and post the Project to
168        // the server.
169        SensorBaseClient client = new SensorBaseClient(getHostName(), testUser, testUser);
170        client.authenticate();
171        client.putProject(project);
172    
173        // Check that we can now retrieve it.
174        Project project2 = client.getProject(owner, projectName);
175        assertEquals("Testing for GET TestProject1", projectName, project2.getName());
176    
177        // Test that DELETE gets rid of this Project.
178        client.deleteProject(owner, projectName);
179      }

182       * Test that PUT multiple times does not cause a problem for getProjectIndex.
183       * 
184       * @throws Exception If problems occur.
185       */
186      @Ignore
187      @Test
188      public void putMultipleProject() throws Exception {
189        // First, create a sample Project.
190        String owner = testUser;
191        String projectName = "TestProject1";
192        Project project = new Project();
193        project.setOwner(owner);
194        project.setName(projectName);
195        project.setDescription("Test Project1");
196        XMLGregorianCalendar tstamp = Tstamp.makeTimestamp();
197        project.setStartTime(tstamp);
198        project.setEndTime(tstamp);
199        UriPatterns uris = new UriPatterns();
200        uris.getUriPattern().add("**/test/**");
201        project.setUriPatterns(uris);
202    
203        // Create the TestUser client, check authentication, and post the Project to
204        // the server.
205        // Create an admin client and check authentication.
206        SensorBaseClient client = new SensorBaseClient(getHostName(), adminEmail, adminPassword);
207        client.authenticate();
208        // Get the index of all Projects.
209        ProjectIndex pIndex1 = client.getProjectIndex();
210        int numProjectsBefore = pIndex1.getProjectRef().size();
211        client.putProject(project);
212        ProjectIndex pIndex2 = client.getProjectIndex();
213        int numProjectsAfter = pIndex2.getProjectRef().size();
214        assertEquals("Checking ProjectIndex after adding duplicate Project", numProjectsBefore,
215            numProjectsAfter);
216        pIndex2 = client.getUserProjectIndex(testUser);
217        numProjectsAfter = pIndex2.getProjectRef().size();
218        assertEquals("2nd check ProjectIndex after adding duplicate Project", numProjectsBefore,
219            numProjectsAfter);
220        int forLoopCount = 0;
221        for (ProjectRef ref : pIndex2.getProjectRef()) {
222          ref.getHref(); // eliminate warning associated with unused 'ref' variable.
223          forLoopCount++;
224        }
225        assertEquals("For loop count check", forLoopCount, numProjectsBefore);
226    
227      }
