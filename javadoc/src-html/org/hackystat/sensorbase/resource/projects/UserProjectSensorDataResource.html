001    package org.hackystat.sensorbase.resource.projects;
002    
003    import org.hackystat.sensorbase.resource.sensorbase.SensorBaseResource;
004    import org.hackystat.sensorbase.resource.users.jaxb.User;
005    import org.hackystat.utilities.tstamp.Tstamp;
006    import org.restlet.Context;
007    import org.restlet.data.MediaType;
008    import org.restlet.data.Request;
009    import org.restlet.data.Response;
010    import org.restlet.data.Status;
011    import org.restlet.resource.Representation;
012    import org.restlet.resource.Variant;
013    
014    /**
015     * The resource for processing GET host/projects/{user}/{projectname}/sensordata.
016     * Returns an index to the SensorData resources associated with this  Project.
017     * This includes all of the SensorData from all members that matches the UriPattern.
018     * Note that this could be quite large if you do not also specify a start and end time.
019     * 
020     * @author Philip Johnson
021     */
022    public class UserProjectSensorDataResource extends SensorBaseResource {
023      
024      /** The user corresponding to email, or null if not found. */
025      private User user;
026      /** To be retrieved from the URL. */
027      private String projectName;
028      /** An optional query parameter */
029      private String startTime;
030      /** An optional query string parameter. */
031      private String endTime;
032      /** An optional query parameter. */
033      private String sdt;
034    
035      
036      /**
037       * Provides the following representational variants: TEXT_XML.
038       * @param context The context.
039       * @param request The request object.
040       * @param response The response object.
041       */
042      public UserProjectSensorDataResource(Context context, Request request, Response response) {
043        super(context, request, response);
044        this.projectName = (String) request.getAttributes().get("projectname"); 
045        this.startTime = (String) request.getAttributes().get("startTime");
046        this.endTime = (String) request.getAttributes().get("endTime");
047        this.sdt = (String) request.getAttributes().get("sdt");
048        this.user = super.userManager.getUser(super.uriUser);
049      }
050      
051      /**
052       * Returns a SensorDataIndex of all SensorData associated with this Project.  This
053       * includes all SensorData from all Members in this project over the 
054       * (optional) specified time period  for the (optional) SDT that match at least
055       * one of the UriPatterns in the project definition.
056       * 
057       * Returns an error condition if:
058       * <ul>
059       * <li> The user does not exist.
060       * <li> The authenticated user is not the uriUser or the Admin or a member of the project.
061       * <li> The Project Resource named by the User and Project does not exist.
062       * <li> startTime or endTime is not an XMLGregorianCalendar string.
063       * <li> One or the other but not both of startTime and endTime is provided.
064       * <li> endTime is earlier than startTime.
065       * </ul>
066       * 
067       * @param variant The representational variant requested.
068       * @return The representation. 
069       */
070      @Override
071      public Representation getRepresentation(Variant variant) {
072        // The user must be defined.
073        if (this.user == null) {
074          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown user");
075          return null;
076        } 
077        // The project must be defined.
078        if (!super.projectManager.hasProject(this.user, this.projectName)) {
079          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown project");
080          return null;
081        }
082        // The authorized user must be an admin, or the project owner, or a member, or invitee.
083        if (!super.userManager.isAdmin(this.authUser) && !this.uriUser.equals(this.authUser) &&
084            !super.projectManager.isMember(this.user, this.projectName, this.authUser) &&
085            !super.projectManager.isInvited(this.user, this.projectName, this.authUser)) {
086          String msg = "User " + this.authUser + "is not authorized to obtain data from this Project.";
087          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, msg);
088          return null;
089        }
090        // If startTime or endTime is provided, but is not an XMLGregorianCalendar, then return error.
091        if ((this.startTime != null) && (!Tstamp.isTimestamp(this.startTime))) {
092          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad startTime");
093          return null;
094        }
095        if ((this.endTime != null) && (!Tstamp.isTimestamp(this.endTime))) {
096          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad endTime");
097          return null;
098        }
099        // If end time is greater than start time, return an error.
100        if ((this.endTime != null) && (Tstamp.isTimestamp(this.endTime)) &&
101            (this.startTime != null) && (Tstamp.isTimestamp(this.startTime)) &&
102            (Tstamp.greaterThan(this.startTime, this.endTime))) {
103            getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Start time after end time.");
104            return null;
105        }
106        // If startType is provided but not endTime, or vice versa, return an error. 
107        if ((this.endTime == null) && (this.startTime != null) ||
108            (this.endTime != null) && (this.startTime == null)) {
109          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "startTime & endTime required.");
110          return null;
111        }
112    
113        if (variant.getMediaType().equals(MediaType.TEXT_XML)) {
114          try {
115            if (startTime == null) {
116              // Return all sensor data for this project if no query parameters.
117              String data = super.projectManager.getProjectSensorDataIndex(this.user, this.projectName);
118              return SensorBaseResource.getStringRepresentation(data);
119            }
120            else {
121              // Return the sensor data starting at startTime and ending with endTime. 
122              String data = super.projectManager.getProjectSensorDataIndex(this.user, this.projectName,
123                  this.startTime, this.endTime, this.sdt);
124              return SensorBaseResource.getStringRepresentation(data);
125            }
126          }
127          catch (Exception e) {
128            getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Problems marshalling");
129            return null;
130          }
131        }
132        return null;
133      }
134    }
