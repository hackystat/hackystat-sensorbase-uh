001    package org.hackystat.sensorbase.resource.projects;
002    
003    import org.hackystat.sensorbase.resource.sensorbase.SensorBaseResource;
004    import org.hackystat.sensorbase.resource.users.jaxb.User;
005    import org.hackystat.utilities.tstamp.Tstamp;
006    import org.restlet.Context;
007    import org.restlet.data.MediaType;
008    import org.restlet.data.Request;
009    import org.restlet.data.Response;
010    import org.restlet.data.Status;
011    import org.restlet.resource.Representation;
012    import org.restlet.resource.Variant;
013    
014    /**
015     * The resource for processing GET host/projects/{user}/{projectname}/sensordata.
016     * Returns an index to the SensorData resources associated with this User and Project.
017     * 
018     * @author Philip Johnson
019     */
020    public class UserProjectSensorDataResource extends SensorBaseResource {
021      
022      /** The user corresponding to email, or null if not found. */
023      private User user;
024      /** To be retrieved from the URL. */
025      private String projectName;
026      /** An optional query parameter */
027      private String startTime;
028      /** An optional query string parameter. */
029      private String endTime;
030      /** An optional query parameter. */
031      private String sdt;
032    
033      
034      /**
035       * Provides the following representational variants: TEXT_XML.
036       * @param context The context.
037       * @param request The request object.
038       * @param response The response object.
039       */
040      public UserProjectSensorDataResource(Context context, Request request, Response response) {
041        super(context, request, response);
042        this.projectName = (String) request.getAttributes().get("projectname"); 
043        this.startTime = (String) request.getAttributes().get("startTime");
044        this.endTime = (String) request.getAttributes().get("endTime");
045        this.sdt = (String) request.getAttributes().get("sdt");
046        this.user = super.userManager.getUser(super.uriUser);
047      }
048      
049      /**
050       * Returns a SensorDataIndex of all SensorData associated with this Project and User.
051       * Returns an error condition if:
052       * <ul>
053       * <li> The user does not exist.
054       * <li> The authenticated user is not the uriUser or the Admin. 
055       * <li> The Project Resource named by the User and Project does not exist.
056       * <li> startTime or endTime is not an XMLGregorianCalendar string.
057       * <li> One or the other but not both of startTime and endTime is provided.
058       * <li> endTime is greater than startTime.
059       * </ul>
060       * 
061       * @param variant The representational variant requested.
062       * @return The representation. 
063       */
064      @Override
065      public Representation getRepresentation(Variant variant) {
066        if (this.user == null) {
067          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown user");
068          return null;
069        }  
070        if (!super.userManager.isAdmin(this.authUser) && !this.uriUser.equals(this.authUser)) {
071          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, super.badAuth);
072          return null;
073        }
074        // If this User/Project pair does not exist, return an error.
075        if (!super.projectManager.hasProject(this.user, this.projectName)) {
076          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Unknown project");
077          return null;
078        }
079        // If startTime or endTime is provided, but is not an XMLGregorianCalendar, then return error.
080        if ((this.startTime != null) && (!Tstamp.isTimestamp(this.startTime))) {
081          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad startTime");
082          return null;
083        }
084        if ((this.endTime != null) && (!Tstamp.isTimestamp(this.endTime))) {
085          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad endTime");
086          return null;
087        }
088        // If end time is greater than start time, return an error.
089        if ((this.endTime != null) && (Tstamp.isTimestamp(this.endTime)) &&
090            (this.startTime != null) && (Tstamp.isTimestamp(this.startTime)) &&
091            (Tstamp.greaterThan(this.startTime, this.endTime))) {
092            getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Start time after end time.");
093            return null;
094        }
095        // If startType is provided but not endTime, or vice versa, return an error. 
096        if ((this.endTime == null) && (this.startTime != null) ||
097            (this.endTime != null) && (this.startTime == null)) {
098          getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "startTime & endTime required.");
099          return null;
100        }
101    
102        if (variant.getMediaType().equals(MediaType.TEXT_XML)) {
103          try {
104            if (startTime == null) {
105              // Return all sensor data for this user and project if no query parameters.
106              String data = super.projectManager.getProjectSensorDataIndex(this.user, this.projectName);
107              return SensorBaseResource.getStringRepresentation(data);
108            }
109            else {
110              // Return the sensor data starting at startTime and ending with endTime. 
111              String data = super.projectManager.getProjectSensorDataIndex(this.user, this.projectName,
112                  this.startTime, this.endTime, this.sdt);
113              return SensorBaseResource.getStringRepresentation(data);
114            }
115          }
116          catch (Exception e) {
117            getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Problems marshalling");
118            return null;
119          }
120        }
121        return null;
122      }
123    }
