082      /** 
083       * The constructor for ProjectManagers. 
084       * There is one ProjectManager per Server. 
085       * @param server The Server instance associated with this ProjectManager. 
086       */
087      public ProjectManager(Server server) {
088        this.server = server;
089        this.userManager = 
090          (UserManager)this.server.getContext().getAttributes().get("UserManager");    
091        this.dbManager = (DbManager)this.server.getContext().getAttributes().get("DbManager");
092        try {
093          this.jaxbContext = 
094            JAXBContext.newInstance("org.hackystat.sensorbase.resource.projects.jaxb");
095          loadDefaultProjects(); //NOPMD it's throwing a false warning. 
096          initializeCache();  //NOPMD 
097          initializeDefaultProjects(); //NOPMD
098        }
099        catch (Exception e) {
100          String msg = "Exception during ProjectManager initialization processing";
101          server.getLogger().warning(msg + "\n" + StackTrace.toString(e));
102          throw new RuntimeException(msg, e);
103        }
104      }

106      /**
107       * Loads the default Projects from the defaults file and adds them to the database. 
108       * @throws Exception If problems occur. 
109       */
110      private final void loadDefaultProjects() throws Exception {
111        // Get the default User definitions from the XML defaults file. 
112        File defaultsFile = findDefaultsFile();
113        // Add these users to the database if we've found a default file. 
114        if (defaultsFile.exists()) {
115          server.getLogger().info("Loading Project defaults from " + defaultsFile.getPath()); 
116          Unmarshaller unmarshaller = jaxbContext.createUnmarshaller();
117          Projects projects = (Projects) unmarshaller.unmarshal(defaultsFile);
118          for (Project project : projects.getProject()) {
119            if (project.getLastMod() == null) {
120              project.setLastMod(Tstamp.makeTimestamp());
121            }
122            this.dbManager.storeProject(project, this.makeProject(project), 
123                this.makeProjectRefString(project));
124          }
125        }
126      } 
127      
128      /** Read in all Projects from the database and initialize the in-memory cache. */
129      private final void initializeCache() {
130        try {
131          ProjectIndex index = makeProjectIndex(this.dbManager.getProjectIndex());
132    
133          for (ProjectRef ref : index.getProjectRef()) {
134            String owner = ref.getOwner();
135            User user = this.userManager.getUser(owner);
136            // Check to make sure user exists.  DB is not normalized! 
137            if (user == null) {
138              String msg = "Project with undefined user '" + owner + "' found while initializing " 
139              + " project cache from database. Project will be ignored.";
140              server.getLogger().warning(msg);   
141            }
142            else {
143              String projectName = ref.getName();
144              String projectString = this.dbManager.getProject(user, projectName);
145              Project project = makeProject(projectString);
146              this.updateCache(project);
147            }
148          }
149        }
150        catch (Exception e) {
151          server.getLogger().warning("Failed to initialize users " + StackTrace.toString(e));
152        }
153      }
154      
155      /**
156       * Updates the in-memory cache with information about this Project. 
157       * @param project The project to be added to the cache.
158       * @throws Exception If problems occur updating the cache. 
159       */
160      private final void updateCache(Project project) throws Exception {
161        updateCache(project, this.makeProject(project), this.makeProjectRefString(project));
162      }
163      
164      /**
165       * Updates the cache given all the Project representations.
166       * Throws unchecked exceptions if the Owner is not defined as a User.
167       * @param project The Project.
168       * @param projectXml The Project as an XML string. 
169       * @param projectRef The Project as an XML reference. 
170       */
171      private final void updateCache(Project project, String projectXml, String projectRef) {
172        // First, put the [owner, project] mapping
173        String email = project.getOwner();
174        User user = userManager.getUser(email);
175        if (user == null) {
176          throw new IllegalArgumentException("Project with undefined User " + email + " " + project);
177        }
178        if (!owner2name2project.containsKey(user)) {
179          owner2name2project.put(user, new HashMap<String, Project>());
180        }
181        owner2name2project.get(user).put(project.getName(), project);
182        this.project2xml.put(project, projectXml);
183        this.project2ref.put(project, projectRef);
184      }

187      /** Make sure that all Users have a "Default" project defined for them. */ 
188      private void initializeDefaultProjects() {
189        for (User user : userManager.getUsers()) {
190          if (!hasProject(user, "Default")) {
191            addDefaultProject(user);
192          }
193        }
194      }

196      /**
197       * Checks the ServerProperties for the XML_DIR property.
198       * If this property is null, returns the File for ./xml/defaults/sensordatatypes.defaults.xml.
199       * @return The File instance (which might not point to an existing file.)
200       */
201      private File findDefaultsFile() {
202        String defaultsPath = "/defaults/projects.defaults.xml";
203        String xmlDir = this.server.getServerProperties().get(XML_DIR_KEY);
204        return (xmlDir == null) ?
205            new File (System.getProperty("user.dir") + "/xml" + defaultsPath) :
206              new File (xmlDir + defaultsPath);
207      }

209      /**
210       * Converts an "Owner" string to an email address.
211       * The owner string might be a URI (starting with http) or an email address. 
212       * @param owner The owner string. 
213       * @return The email address corresponding to the owner string. 
214       */
215      public synchronized String convertOwnerToEmail(String owner) {
216        if (owner.startsWith(http)) {
217          int lastSlash = owner.lastIndexOf('/');
218          if (lastSlash < 0) {
219            throw new IllegalArgumentException("Could not convert owner to URI");
220          }
221          return owner.substring(lastSlash + 1); 
222        }
223        // Otherwise owner is already the email. 
224        return owner;
225      }

227      /**
228       * Returns the owner string as a URI.
229       * The owner string could either be an email address or the URI. 
230       * @param owner The owner string. 
231       * @return The URI corresponding to the owner string. 
232       */
233      public synchronized String convertOwnerToUri(String owner) {
234        return (owner.startsWith(http)) ? owner :
235          this.server.getServerProperties().getFullHost() + "users/" + owner;
236      }

238      /**
239       * Returns the XML string containing the ProjectIndex with all defined Projects.
240       * Uses the in-memory cache of ProjectRef strings.  
241       * @return The XML string providing an index to all current Projects.
242       */
243      public synchronized String getProjectIndex() {
244        StringBuilder builder = new StringBuilder(512);
245        builder.append(projectIndexOpenTag);
246        for (String ref : this.project2ref.values()) {
247          builder.append(ref);
248        }
249        builder.append(projectIndexCloseTag);
250        return builder.toString();
251      }
252      
253      /**
254       * Returns the XML string containing the ProjectIndex with all defined Projects for the Owner.
255       * Uses the in-memory cache of ProjectRef strings.  
256       * @param owner The owner whose Projects are to be retrieved.
257       * @return The XML string providing an index to all Projects for this owner.
258       */
259      public synchronized String getProjectIndex(User owner) {
260        StringBuilder builder = new StringBuilder(512);
261        builder.append(projectIndexOpenTag);
262        if (owner != null) {
263          Map<String, Project> name2project = this.owner2name2project.get(owner);
264          if (name2project != null) {
265            for (Project project : name2project.values()) {
266              builder.append(this.project2ref.get(project));    
267            }
268          }
269        }
270        builder.append(projectIndexCloseTag);
271        return builder.toString();
272      }

274      /**
275       * Updates the Manager with this Project. Any old definition is overwritten.
276       * @param project The Project.
277       */
278      public synchronized void putProject(Project project) {
279        try {
280          project.setLastMod(Tstamp.makeTimestamp());
281          String xmlProject =  this.makeProject(project);
282          String xmlRef =  this.makeProjectRefString(project);
283          this.updateCache(project, xmlProject, xmlRef);
284          this.dbManager.storeProject(project, xmlProject, xmlRef);
285        }
286        catch (Exception e) {
287          server.getLogger().warning("Failed to put Project" + StackTrace.toString(e));
288        }
289      }

291      /**
292       * Returns true if the passed Project name is defined for this User (who must be the owner).
293       * @param  owner The project owner (can be null). 
294       * @param  projectName A project name (can be null).
295       * @return True if a Project with that name is owned by that User.  False if the User or
296       * Project is not defined. 
297       */
298      public synchronized boolean hasProject(User owner, String projectName) {
299        return 
300        (owner != null) &&
301        (projectName != null) &&
302        this.owner2name2project.containsKey(owner) &&
303        this.owner2name2project.get(owner).containsKey(projectName);
304      }
305      
306      /**
307       * Ensures that the passed Project is no longer present in this Manager. 
308       * @param owner The user who owns this Project.
309       * @param projectName The name of the project.
310       */
311      public synchronized void deleteProject(User owner, String projectName) {
312        if (this.owner2name2project.containsKey(owner)) {
313          Project project = this.owner2name2project.get(owner).get(projectName);
314          this.project2ref.remove(project);
315          this.project2xml.remove(project);
316          this.owner2name2project.get(owner).remove(projectName);
317        }
318        this.dbManager.deleteProject(owner, projectName);
319      }

322       * Returns the Project Xml String associated with this User and project name.
323       * @param owner The user that owns this project.
324       * @param projectName The name of the project.
325       * @return The Project XML string, or null if not found.
326       */
327      public synchronized String getProjectString(User owner, String projectName) {
328        if (hasProject(owner, projectName)) {
329          Project project = this.owner2name2project.get(owner).get(projectName);
330          return this.project2xml.get(project);
331        }
332        return null;
333      }

335      /**
336       * Returns a set containing the current Project instances. 
337       * For thread safety, a fresh Set of Projects is built each time this is called. 
338       * @return A Set containing the current Projects. 
339       */
340      public synchronized Set<Project> getProjects() {
341        Set<Project> projectSet = new HashSet<Project>(projectSetSize);
342        for (User user : this.owner2name2project.keySet()) {
343          for (String projectName : this.owner2name2project.get(user).keySet()) {
344            projectSet.add(this.owner2name2project.get(user).get(projectName));
345          }
346        }
347        return projectSet;
348      }

351      /**
352       * Returns an XML SensorDataIndex String for all data associated with this User and Project.
353       * Assumes that the owner and projectName define an existing Project.
354       * @param owner The User that owns this Project.
355       * @param projectName the Project name.
356       * @return The XML SensorDataIndex string providing an index to all data for this user/project.
357       * @throws Exception If things go wrong. 
358       */
359      public synchronized String getProjectSensorDataIndex(User owner, String projectName) 
360      throws Exception {
361        SensorDataManager sensorDataManager = 
362          (SensorDataManager)this.server.getContext().getAttributes().get("SensorDataManager");
363        Project project = this.getProject(owner, projectName);
364        XMLGregorianCalendar startTime = project.getStartTime();
365        XMLGregorianCalendar endTime = project.getEndTime();
366        List<UriPattern> patterns = UriPattern.getPatterns(project);
367        return sensorDataManager.getSensorDataIndex(owner, startTime, endTime, patterns, null);
368      }

371      /**
372       * Returns the XML SensorDataIndex string for the data associated with this Project within the 
373       * specified start and end times.
374       * Note that the Project start and end times may further constrain the returned set of data. 
375       * This method chooses the greater of startString and the Project startTime, and the lesser of
376       * endString and the Project endTime. 
377       * Assumes that User and Project are valid.
378       * @param owner The User who owns this Project. 
379       * @param projectName the Project name.
380       * @param startString The startTime as a string. 
381       * @param endString The endTime as a string.
382       * @param sdt The SensorDataType of interest, or null if all sensordatatypes are to be retrieved.
383       * @return The XML String providing a SensorDataIndex to the sensor data in this project
384       * starting at startTime and ending at endTime. 
385       * @throws Exception if startString or endString are not XMLGregorianCalendars.
386       */  
387      public synchronized String getProjectSensorDataIndex(User owner, 
388          String projectName, String startString, String endString, String sdt) throws Exception {
389        SensorDataManager sensorDataManager = 
390          (SensorDataManager)this.server.getContext().getAttributes().get("SensorDataManager");
391        Project project = this.getProject(owner, projectName);
392        XMLGregorianCalendar startTime = Tstamp.makeTimestamp(startString);
393        XMLGregorianCalendar endTime = Tstamp.makeTimestamp(endString);
394        // make startTime the greater of startTime and the Project startTime. 
395        startTime = (Tstamp.greaterThan(startTime, project.getStartTime())) ?
396            startTime : project.getStartTime();
397        // make endTime the lesser of endTime and the Project endTime.
398        endTime = (Tstamp.lessThan(endTime, project.getEndTime())) ? 
399            endTime : project.getEndTime();
400        List<UriPattern> patterns = UriPattern.getPatterns(project);
401        return sensorDataManager.getSensorDataIndex(owner, startTime, endTime, patterns, sdt);
402        
403      }

405      /**
406       * Creates and stores the "Default" project for the specified user. 
407       * @param owner The user who will own this Project.
408       */
409      public final synchronized void addDefaultProject(User owner) {
410        Project project = new Project();
411        project.setDescription("The default Project");
412        project.setStartTime(Tstamp.getDefaultProjectStartTime());
413        project.setEndTime(Tstamp.getDefaultProjectEndTime());
414        project.setMembers(new Members());
415        project.setName("Default");
416        project.setOwner(owner.getEmail());
417        project.setProperties(new Properties());
418        project.setLastMod(Tstamp.makeTimestamp());
419        UriPatterns uriPatterns = new UriPatterns();
420        uriPatterns.getUriPattern().add("**");
421        project.setUriPatterns(uriPatterns);
422        putProject(project);
423      }

470      /**
471       * Returns the passed Project instance as a String encoding of its XML representation.
472       * Final because it's called in constructor.
473       * @param project The Project instance. 
474       * @return The XML String representation.
475       * @throws Exception If problems occur during translation. 
476       */
477      public final synchronized String makeProject (Project project) throws Exception {
478        Marshaller marshaller = jaxbContext.createMarshaller(); 
479        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
480        dbf.setNamespaceAware(true);
481        DocumentBuilder documentBuilder = dbf.newDocumentBuilder();
482        Document doc = documentBuilder.newDocument();
483        marshaller.marshal(project, doc);
484        DOMSource domSource = new DOMSource(doc);
485        StringWriter writer = new StringWriter();
486        StreamResult result = new StreamResult(writer);
487        TransformerFactory tf = TransformerFactory.newInstance();
488        Transformer transformer = tf.newTransformer();
489        transformer.transform(domSource, result);
490        String xmlString = writer.toString();
491        // Now remove the processing instruction.  This approach seems like a total hack.
492        xmlString = xmlString.substring(xmlString.indexOf('>') + 1);
493        return xmlString;
494      }

515        StreamResult result = new StreamResult(writer);
516        TransformerFactory tf = TransformerFactory.newInstance();
517        Transformer transformer = tf.newTransformer();
518        transformer.transform(domSource, result);
519        String xmlString = writer.toString();
520        // Now remove the processing instruction.  This approach seems like a total hack.

523      }

525      /**
526       * Returns a ProjectRef instance constructed from a Project instance.
527       * @param project The Project instance. 
528       * @return A ProjectRef instance. 
529       */
530      public synchronized ProjectRef makeProjectRef(Project project) {
531        ProjectRef ref = new ProjectRef();
532        String ownerEmail = convertOwnerToEmail(project.getOwner());
533        ref.setName(project.getName());
534        ref.setOwner(ownerEmail);
535        ref.setLastMod(project.getLastMod());
536        ref.setHref(this.server.getHostName() + "projects/" + ownerEmail + "/" + project.getName()); 
537        return ref;
538      }
