088       * The constructor for ProjectManagers. 
089       * There is one ProjectManager per Server. 
090       * @param server The Server instance associated with this ProjectManager.

092      public ProjectManager(Server server) {
093        this.server = server;
094        this.userManager = 
095          (UserManager)this.server.getContext().getAttributes().get("UserManager");    
096        this.dbManager = (DbManager)this.server.getContext().getAttributes().get("DbManager");
097        try {
098          this.jaxbContext = 
099            JAXBContext.newInstance("org.hackystat.sensorbase.resource.projects.jaxb");
100          loadDefaultProjects(); //NOPMD it's throwing a false warning. 
101          initializeCache();  //NOPMD 
102          initializeDefaultProjects(); //NOPMD
103        }
104        catch (Exception e) {
105          String msg = "Exception during ProjectManager initialization processing";
106          server.getLogger().warning(msg + "\n" + StackTrace.toString(e));
107          throw new RuntimeException(msg, e);
108        }
109      }

111      /**
112       * Loads the default Projects from the defaults file and adds them to the database. 
113       * @throws Exception If problems occur. 
114       */
115      private final void loadDefaultProjects() throws Exception {
116        // Get the default User definitions from the XML defaults file. 
117        File defaultsFile = findDefaultsFile();
118        // Add these users to the database if we've found a default file. 
119        if (defaultsFile.exists()) {
120          server.getLogger().info("Loading Project defaults from " + defaultsFile.getPath()); 
121          Unmarshaller unmarshaller = jaxbContext.createUnmarshaller();
122          Projects projects = (Projects) unmarshaller.unmarshal(defaultsFile);
123          for (Project project : projects.getProject()) {
124            if (project.getLastMod() == null) {
125              project.setLastMod(Tstamp.makeTimestamp());
126            }
127            this.dbManager.storeProject(project, this.makeProject(project), 
128                this.makeProjectRefString(project));
129          }
130        }
131      }

133      /** Read in all Projects from the database and initialize the in-memory cache. */
134      private final void initializeCache() {
135        try {
136          ProjectIndex index = makeProjectIndex(this.dbManager.getProjectIndex());
137    
138          for (ProjectRef ref : index.getProjectRef()) {
139            String owner = ref.getOwner();
140            User user = this.userManager.getUser(owner);
141            // Check to make sure user exists.  DB is not normalized! 
142            if (user == null) {
143              String msg = "Project with undefined user '" + owner + "' found while initializing " 
144              + " project cache from database. Project will be ignored.";
145              server.getLogger().fine(msg);   
146            }
147            else {
148              String projectName = ref.getName();
149              if (this.hasProject(user, projectName)) {
150                String msg = "Duplicate project for " + user + " with name " + projectName + 
151                " found in database. Ignoring.";
152                server.getLogger().warning(msg);
153              }
154              else {
155                String projectString = this.dbManager.getProject(user, projectName);
156                Project project = makeProject(projectString);
157                this.updateCache(project);
158              }
159            }
160          }
161        }
162        catch (Exception e) {
163          server.getLogger().warning("Failed to initialize users " + StackTrace.toString(e));
164        }
165      }

167      /**
168       * Updates the in-memory cache with information about this Project. 
169       * @param project The project to be added to the cache.
170       * @throws Exception If problems occur updating the cache. 
171       */
172      private final void updateCache(Project project) throws Exception {
173        // Make sure project has an Invitations element.
174        if (project.getInvitations() == null) {
175          project.setInvitations(new Invitations());
176        }
177        // Fix bogus default project start/end dates. 
178        if (project.getName().equals("Default") && Tstamp.isBogusStartTime(project.getStartTime())) {
179            project.setStartTime(Tstamp.getDefaultProjectStartTime());
180            project.setEndTime(Tstamp.getDefaultProjectEndTime());
181        }
182        updateCache(project, this.makeProject(project), this.makeProjectRefString(project));
183      }

185      /**
186       * Updates the cache given all the Project representations.
187       * Throws unchecked exceptions if the Owner is not defined as a User.
188       * @param project The Project.
189       * @param projectXml The Project as an XML string. 
190       * @param projectRef The Project as an XML reference. 
191       */
192      private final void updateCache(Project project, String projectXml, String projectRef) {
193        // First, put the [owner, project] mapping
194        String email = project.getOwner();
195        User user = userManager.getUser(email);
196        if (user == null) {
197          throw new IllegalArgumentException("Project with undefined User " + email + " " + project);
198        }
199        if (!owner2name2project.containsKey(user)) {
200          owner2name2project.put(user, new HashMap<String, Project>());
201        }
202        owner2name2project.get(user).put(project.getName(), project);
203        this.project2xml.put(project, projectXml);
204        this.project2ref.put(project, projectRef);
205      }

209      private void initializeDefaultProjects() {
210        for (User user : userManager.getUsers()) {
211          if (!hasProject(user, "Default")) {
212            addDefaultProject(user);
213          }
214        }

230      /**
231       * Converts an "Owner" string to an email address.
232       * The owner string might be a URI (starting with http) or an email address. 
233       * @param owner The owner string. 
234       * @return The email address corresponding to the owner string. 
235       */
236      public synchronized String convertOwnerToEmail(String owner) {
237        if (owner.startsWith(http)) {
238          int lastSlash = owner.lastIndexOf('/');
239          if (lastSlash < 0) {
240            throw new IllegalArgumentException("Could not convert owner to URI");
241          }
242          return owner.substring(lastSlash + 1); 
243        }
244        // Otherwise owner is already the email. 
245        return owner;
246      }

260       * Returns the XML string containing the ProjectIndex with all defined Projects.
261       * Uses the in-memory cache of ProjectRef strings.  
262       * @return The XML string providing an index to all current Projects.
263       */
264      public synchronized String getProjectIndex() {
265        StringBuilder builder = new StringBuilder(512);
266        builder.append(projectIndexOpenTag);
267        for (String ref : this.project2ref.values()) {
268          builder.append(ref);
269        }
270        builder.append(projectIndexCloseTag);
271        return builder.toString();
272      }

274      /**
275       * Returns the XML string containing the ProjectIndex with all Projects associated with 
276       * this user.
277       * Uses the in-memory cache of ProjectRef strings.  
278       * @param user The user whose associated Projects are to be retrieved. All projects for
279       * which this user is an owner, member, or invitee are returned.
280       * @return The XML string providing an index to all Projects associated with this user.
281       */
282      public synchronized String getProjectIndex(User user) {
283        String email = user.getEmail();
284        StringBuilder builder = new StringBuilder(512);
285        builder.append(projectIndexOpenTag);
286        for (Map<String, Project> name2project : this.owner2name2project.values()) {
287          for (Project project : name2project.values()) {
288            Members members = project.getMembers();
289            Invitations invitations = project.getInvitations();
290            if (project.getOwner().equals(email) ||
291                (members != null) && (members.getMember().contains(email)) ||
292                (invitations != null) && (invitations.getInvitation().contains(email))) {
293              builder.append(this.project2ref.get(project));   
294            }
295          }
296        }
297        builder.append(projectIndexCloseTag);
298        return builder.toString();
299      }

319       * Returns true if the passed Project name is defined for this User (who must be the owner).
320       * @param  owner The project owner (can be null). 
321       * @param  projectName A project name (can be null).
322       * @return True if a Project with that name is owned by that User.  False if the User or
323       * Project is not defined. 
324       */
325      public synchronized boolean hasProject(User owner, String projectName) {
326        return 
327        (owner != null) &&
328        (projectName != null) &&
329        this.owner2name2project.containsKey(owner) &&
330        this.owner2name2project.get(owner).containsKey(projectName);
331      }

333      /**
334       * Returns true if member is a member of the project owned by owner. 
335       * @param owner The owner of projectName.
336       * @param projectName The name of the project owned by owner.
337       * @param member The user whose membership is being checked.
338       * @return True if member is a member of project, false otherwise. 
339       */
340      public synchronized boolean isMember(User owner, String projectName, String member) {
341        // Return false if owner, project, member are invalid.
342        if ((owner == null) || (member == null) || (projectName == null) ||
343            !this.owner2name2project.containsKey(owner) ||
344            !this.owner2name2project.get(owner).containsKey(projectName)) {
345          return false;
346        }
347        // Now we can get the project.
348        Project project = this.owner2name2project.get(owner).get(projectName);
349        // Return false if the <Members> field is null.
350        if (!project.isSetMembers()) {
351          return false;
352        }
353        // Look for the member in the list.
354        List<String> members = project.getMembers().getMember();
355        for (String currMember : members) {
356          if (currMember.equals(member)) {
357            return true;
358          }
359        }
360        // Got here, which means that we never found the member.
361        return false;
362      }
363      
364      /**
365       * Returns true if member is invited to be a member of the project owned by owner. 
366       * @param owner The owner of projectName.
367       * @param projectName The name of the project owned by owner.
368       * @param invitee The user whose invitation status is being checked.
369       * @return True if member is invited to be a member of project, false otherwise. 
370       */
371      public synchronized boolean isInvited(User owner, String projectName, String invitee) {
372        // Return false if owner, project, member are invalid.
373        if ((owner == null) || (invitee == null) || (projectName == null) ||
374            !this.owner2name2project.containsKey(owner) ||
375            !this.owner2name2project.get(owner).containsKey(projectName)) {
376          return false;
377        }
378        // Now we can get the project.
379        Project project = this.owner2name2project.get(owner).get(projectName);
380        // Return false if the <Members> field is null.
381        if (!project.isSetInvitations()) {
382          return false;
383        }
384        // Look for the member in the list.
385        List<String> invitees = project.getInvitations().getInvitation();
386        for (String currInvitee : invitees) {
387          if (currInvitee.equals(invitee)) {
388            return true;
389          }
390        }
391        // Got here, which means that we never found the invitee.
392        return false;
393      }

397       * Ensures that the passed Project is no longer present in this Manager. 
398       * @param owner The user who owns this Project.
399       * @param projectName The name of the project.
400       */
401      public synchronized void deleteProject(User owner, String projectName) {
402        if (this.owner2name2project.containsKey(owner)) {
403          Project project = this.owner2name2project.get(owner).get(projectName);
404          if (project != null) {
405            this.project2ref.remove(project);
406            this.project2xml.remove(project);
407            this.owner2name2project.get(owner).remove(projectName);
408          }
409        }
410        this.dbManager.deleteProject(owner, projectName);
411      }

429       * For thread safety, a fresh Set of Projects is built each time this is called.

432      public synchronized Set<Project> getProjects() {
433        Set<Project> projectSet = new HashSet<Project>(projectSetSize);

435          for (String projectName : this.owner2name2project.get(user).keySet()) {

438        }

443      /**
444       * Returns an XML SensorDataIndex String for all data associated with the Project
445       * owned by this user.
446       * Assumes that the owner and projectName define an existing Project.
447       * @param owner The User that owns this Project.
448       * @param projectName the Project name.
449       * @return The XML SensorDataIndex string providing an index to all data for this user/project.
450       * @throws Exception If things go wrong. 
451       */
452      public synchronized String getProjectSensorDataIndex(User owner, String projectName) 
453      throws Exception {
454        SensorDataManager sensorDataManager = 
455          (SensorDataManager)this.server.getContext().getAttributes().get("SensorDataManager");
456        Project project = this.getProject(owner, projectName);
457        XMLGregorianCalendar startTime = project.getStartTime();
458        XMLGregorianCalendar endTime = project.getEndTime();
459        List<UriPattern> patterns = UriPattern.getPatterns(project);
460        List<User> users = new ArrayList<User>();
461        users.add(owner);
462        for (String member : project.getMembers().getMember()) {
463          User user = userManager.getUser(member);
464          if (user != null) {
465            users.add(user);
466          }
467        }
468        return sensorDataManager.getSensorDataIndex(users, startTime, endTime, patterns, null);
469      }

472      /**
473       * Returns the XML SensorDataIndex string for the data associated with this Project within the 
474       * specified start and end times.
475       * Note that the Project start and end times may further constrain the returned set of data. 
476       * This method chooses the greater of startString and the Project startTime, and the lesser of
477       * endString and the Project endTime. 
478       * Assumes that User and Project are valid.
479       * @param owner The User who owns this Project. 
480       * @param projectName the Project name.
481       * @param startString The startTime as a string. 
482       * @param endString The endTime as a string.
483       * @param sdt The SensorDataType of interest, or null if all sensordatatypes are to be retrieved.
484       * @return The XML String providing a SensorDataIndex to the sensor data in this project
485       * starting at startTime and ending at endTime. 
486       * @throws Exception if startString or endString are not XMLGregorianCalendars.
487       */  
488      public synchronized String getProjectSensorDataIndex(User owner, 
489          String projectName, String startString, String endString, String sdt) throws Exception {
490        SensorDataManager sensorDataManager = 
491          (SensorDataManager)this.server.getContext().getAttributes().get("SensorDataManager");
492        Project project = this.getProject(owner, projectName);
493        XMLGregorianCalendar startTime = Tstamp.makeTimestamp(startString);
494        XMLGregorianCalendar endTime = Tstamp.makeTimestamp(endString);
495        // make startTime the greater of startTime and the Project startTime. 
496        startTime = (Tstamp.greaterThan(startTime, project.getStartTime())) ?
497            startTime : project.getStartTime();
498        // make endTime the lesser of endTime and the Project endTime.
499        endTime = (Tstamp.lessThan(endTime, project.getEndTime())) ? 
500            endTime : project.getEndTime();
501        List<UriPattern> patterns = UriPattern.getPatterns(project);
502        List<User> users = new ArrayList<User>();
503        users.add(owner);
504        for (String member : project.getMembers().getMember()) {
505          User user = userManager.getUser(member);
506          if (user != null) {
507            users.add(user);
508          }
509        }
510        return sensorDataManager.getSensorDataIndex(users, startTime, endTime, patterns, sdt);
511        
512      }

528        UriPatterns uriPatterns = new UriPatterns();
529        uriPatterns.getUriPattern().add("**");
530        project.setUriPatterns(uriPatterns);
531        putProject(project);
532      }

579      /**
580       * Returns the passed Project instance as a String encoding of its XML representation.
581       * Final because it's called in constructor.
582       * @param project The Project instance. 
583       * @return The XML String representation.
584       * @throws Exception If problems occur during translation. 
585       */
586      public final synchronized String makeProject (Project project) throws Exception {
587        Marshaller marshaller = jaxbContext.createMarshaller(); 
588        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
589        dbf.setNamespaceAware(true);
590        DocumentBuilder documentBuilder = dbf.newDocumentBuilder();
591        Document doc = documentBuilder.newDocument();
592        marshaller.marshal(project, doc);
593        DOMSource domSource = new DOMSource(doc);
594        StringWriter writer = new StringWriter();
595        StreamResult result = new StreamResult(writer);
596        TransformerFactory tf = TransformerFactory.newInstance();
597        Transformer transformer = tf.newTransformer();
598        transformer.transform(domSource, result);
599        String xmlString = writer.toString();
600        // Now remove the processing instruction.  This approach seems like a total hack.
601        xmlString = xmlString.substring(xmlString.indexOf('>') + 1);
602        return xmlString;
603      }

605      /**
606       * Returns the passed Project instance as a String encoding of its XML representation 
607       * as a ProjectRef object.
608       * Final because it's called in constructor.
609       * @param project The Project instance. 
610       * @return The XML String representation of it as a ProjectRef
611       * @throws Exception If problems occur during translation. 
612       */
613      public final synchronized String makeProjectRefString (Project project) 
614      throws Exception {
615        ProjectRef ref = makeProjectRef(project);
616        Marshaller marshaller = jaxbContext.createMarshaller(); 
617        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
618        dbf.setNamespaceAware(true);
619        DocumentBuilder documentBuilder = dbf.newDocumentBuilder();
620        Document doc = documentBuilder.newDocument();
621        marshaller.marshal(ref, doc);
622        DOMSource domSource = new DOMSource(doc);
623        StringWriter writer = new StringWriter();
624        StreamResult result = new StreamResult(writer);
625        TransformerFactory tf = TransformerFactory.newInstance();
626        Transformer transformer = tf.newTransformer();
627        transformer.transform(domSource, result);
628        String xmlString = writer.toString();
629        // Now remove the processing instruction.  This approach seems like a total hack.
630        xmlString = xmlString.substring(xmlString.indexOf('>') + 1);
631        return xmlString;
632      }
