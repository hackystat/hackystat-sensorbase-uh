001    package org.hackystat.sensorbase.logger;
002    
003    import java.io.PrintWriter;
004    import java.io.StringWriter;
005    
006    /**
007     * Provides a simple solution to the common problem of obtaining a String containing the stack
008     * trace produced by an exception.
009     * <p>
010     * Call StackTrace.toString(e) to get the string corresponding to the Exception e.
011     * 
012     * @author Philip Johnson, Takuya Yamashita
013     */
014    public class StackTrace {
015      /**
016       * Converts the Throwable.getStackTrace to a String representation for logging.
017       * @param throwable The Throwable exception.
018       * @return A String containing the StackTrace. 
019       */
020      public static String toString(Throwable throwable) {
021        StringWriter stringWriter = new StringWriter();
022        throwable.printStackTrace(new PrintWriter(stringWriter));
023        return stringWriter.toString();    
024      }
025    }
