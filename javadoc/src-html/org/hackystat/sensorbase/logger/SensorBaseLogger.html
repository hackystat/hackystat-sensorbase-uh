001    package org.hackystat.sensorbase.logger;
002    
003    import java.io.File;
004    import java.util.logging.ConsoleHandler;
005    import java.util.logging.FileHandler;
006    import java.util.logging.Level;
007    import java.util.logging.LogManager;
008    import java.util.logging.Logger;
009    
010    
011    /**
012     * Supports logging of informational and error messages by this SensorBase.
013     * @author Philip Johnson
014     *
015     */
016    public class SensorBaseLogger {
017      /** The string used to identify the logger and log files associated with this logger. */
018      private static String loggerName = "sensorbase-uh";
019    
020      /**
021       * Returns a logger for this service, creating it if it couldn't be found.
022       *
023       * @return   The logger.
024       */
025      public static Logger getLogger() {
026        Logger logger = LogManager.getLogManager().getLogger(loggerName);
027        if (logger == null) {
028          try {
029            logger = Logger.getLogger(loggerName);
030            logger.setUseParentHandlers(false);
031    
032            // Define a file handler that writes to the ~/.hackystat/logs directory, creating it if nec.
033            File logDir = new File(System.getProperty("user.home") + "/.hackystat/logs/");
034            logDir.mkdirs();
035            String fileName = logDir + "/" + loggerName + ".%u.log";
036            FileHandler fileHandler = new FileHandler(fileName, 500000, 1, true);
037            fileHandler.setFormatter(new OneLineFormatter());
038            logger.addHandler(fileHandler);
039    
040            // Define a console handler to also write the message to the console if in webserver.
041            ConsoleHandler consoleHandler = new ConsoleHandler();
042            consoleHandler.setFormatter(new OneLineFormatter());
043            logger.addHandler(consoleHandler);
044            setLoggingLevel(Level.INFO);
045          }
046          catch (Exception e) {
047            System.out.println("Error instantiating logger.");
048          }
049        }
050        return logger;
051      }
052    
053      /**
054       * Sets the logging level to be used for the Hackystat logger. 
055       * @param level The new Level.
056       */
057      public static void setLoggingLevel(Level level) {
058        Logger logger = LogManager.getLogManager().getLogger(loggerName);
059        logger.setLevel(level);
060        logger.getHandlers()[0].setLevel(level);
061        logger.getHandlers()[1].setLevel(level);
062      }
063    }
