001    package org.hackystat.sensorbase.logger;
002    
003    import java.text.SimpleDateFormat;
004    import java.util.Date;
005    import java.util.logging.Formatter;
006    import java.util.logging.LogRecord;
007    
008      /**
009       * Provides a one line formatter for use with Hackystat logging. Supports optional date stamp
010       * prefix. If the date stamp prefix is enabled, then a cr is also added.
011       *
012       * @author Philip Johnson
013       */
014    class OneLineFormatter extends Formatter {
015    
016        /**
017         * Formats the passed log string as a single line. Prefixes the log string with a date stamp.
018         *
019         * @param record  A log record.
020         * @return The message string.
021         */
022        public String format(LogRecord record) {
023          SimpleDateFormat dateFormat = new SimpleDateFormat("MM/dd HH:mm:ss");
024          StringBuffer buff = new StringBuffer();
025          buff.append(dateFormat.format(new Date()));
026          buff.append(" ");
027          buff.append(record.getMessage());
028          buff.append(System.getProperty("line.separator"));
029          return buff.toString();
030        }
031      }
