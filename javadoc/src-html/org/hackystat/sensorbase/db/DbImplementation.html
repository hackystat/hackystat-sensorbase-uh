001    package org.hackystat.sensorbase.db;
002    
003    import java.util.List;
004    import java.util.logging.Logger;
005    
006    import javax.xml.datatype.XMLGregorianCalendar;
007    
008    import org.hackystat.sensorbase.resource.projects.ProjectManager;
009    import org.hackystat.sensorbase.resource.projects.jaxb.Project;
010    import org.hackystat.sensorbase.resource.sensordata.SensorDataManager;
011    import org.hackystat.sensorbase.resource.sensordata.jaxb.SensorData;
012    import org.hackystat.sensorbase.resource.sensordatatypes.SdtManager;
013    import org.hackystat.sensorbase.resource.sensordatatypes.jaxb.SensorDataType;
014    import org.hackystat.sensorbase.resource.users.UserManager;
015    import org.hackystat.sensorbase.resource.users.jaxb.User;
016    import org.hackystat.sensorbase.server.Server;
017    import org.hackystat.sensorbase.uripattern.UriPattern;
018    
019    /**
020     * Provides a specification of the operations that must be implemented by every
021     * SensorBase storage system.  Also provides some 'helper' methods, which return
022     * the Managers for the various resources. 
023     * @author Philip Johnson
024     */
025    public abstract class DbImplementation {
026      
027      /**
028       * To be called as part of the startup process for a storage system. This method should:
029       * <ul>
030       * <li> Check to see if this storage system has already been created during a previous session.
031       * <li> If no storage system exists, it should create one and initialize it appropriately.
032       * </ul>
033       */
034      public abstract void initialize();
035      
036      /**
037       * Returns true if the initialize() method did indeed create a fresh storage system.
038       * This is used by the ResourceManagers to determine if they should read in default data or not.
039       * @return True if the storage system is freshly created.
040       */
041      public abstract boolean isFreshlyCreated();
042      
043      /**
044       * Returns the XML SensorDataIndex for all sensor data in this server.
045       * @return The XML String containing an index to all Sensor Data.
046       */
047      public abstract String getSensorDataIndex();
048      
049      /**
050       * Returns the XML SensorDataIndex for all sensor data for this user. 
051       * @param user The User whose sensor data is to be returned. 
052       * @return The XML String providing an index of all relevent sensor data resources.
053       */
054      public abstract String getSensorDataIndex(User user);
055      
056      /**
057       * Returns the XML SensorDataIndex for all sensor data for this user and sensor data type.
058       * @param user The User whose sensor data is to be returned. 
059       * @param sdtName The sensor data type name.
060       * @return The XML String providing an index of all relevent sensor data resources.
061       */
062      public abstract String getSensorDataIndex(User user, String sdtName);
063      
064      /**
065       * Returns an XML SensorDataIndex representing the SensorData for the given user between
066       * start and end time whose resource string matches at least one of the UriPatterns.
067       * @param user The user whose SensorData will be returned.
068       * @param startTime The earliest Sensor Data to be returned.
069       * @param endTime The latest SensorData to be returned.
070       * @param uriPatterns At least one UriPattern must match the SensorData resource field.
071       * @return An XML String containing a SensorDataIndex of matching SensorData. 
072       */
073      public abstract String getSensorDataIndex(User user, XMLGregorianCalendar startTime, 
074          XMLGregorianCalendar endTime, List<UriPattern> uriPatterns);
075      
076      /**
077       * Returns the SensorData instance as XML string, or null if not found.
078       * @param user The user.
079       * @param timestamp The timestamp associated with this sensor data.
080       * @return The SensorData XML string, or null.
081       */
082      public abstract String getSensorData(User user, XMLGregorianCalendar timestamp);
083      
084      /**
085       * Returns true if the passed [key, timestamp] has sensor data defined for it.
086       * @param user The user.
087       * @param timestamp The timestamp
088       * @return True if there is any sensor data for this [key, sdtName, timestamp].
089       */
090      public abstract boolean hasSensorData(User user, XMLGregorianCalendar timestamp);
091      
092      /**
093       * Persists a SensorData instance.  If SensorData with this [email, timestamp] 
094       * already exists in the storage system, it should be overwritten.
095       * @param data The sensor data. 
096       * @param xmlSensorData The SensorData marshalled into an XML String.  
097       * @param xmlSensorDataRef The corresponding SensorDataRef marshalled into an XML String
098       * @return True if the sensor data was successfully inserted.
099       */
100      public abstract boolean storeSensorData(SensorData data, String xmlSensorData, 
101          String xmlSensorDataRef);
102      
103    
104      /**
105       * Ensures that sensor data with the given user and timestamp is no longer
106       * present in this manager.
107       * @param user The user.
108       * @param timestamp The timestamp associated with this sensor data.
109       */
110      public abstract void deleteSensorData(User user, XMLGregorianCalendar timestamp);
111      
112      /**
113       * Returns the XML SensorDataTypeIndex for all SDTs in this server.
114       * @return The XML String containing an index to all SensorDataTypes.
115       */
116      public abstract String getSensorDataTypeIndex();
117      
118      /**
119       * Returns the SensorDataType instance as XML string, or null if not found.
120       * @param sdtName The SDT name.
121       * @return The SensorDataType XML string, or null.
122       */
123      public abstract String getSensorDataType(String sdtName);
124      
125      
126      /**
127       * Persists a SensorDataType instance.  If a SensorDataType with this name
128       * already exists in the storage system, it will be overwritten.
129       * @param sdt The sensor data type 
130       * @param xmlSensorDataType The SensorDataType marshalled into an XML String.  
131       * @param xmlSensorDataTypeRef The corresponding SensorDataTypeRef marshalled into an XML String
132       * @return True if the SDT was successfully inserted.
133       */
134      public abstract boolean storeSensorDataType(SensorDataType sdt, String xmlSensorDataType, 
135          String xmlSensorDataTypeRef);
136      
137      /**
138       * Ensures that the SensorDataType with the given name is no longer present in this manager.
139       * @param sdtName The SDT name.
140       */
141      public abstract void deleteSensorDataType(String sdtName);
142      
143      /**
144       * Returns the XML UserIndex for all Users in this server.
145       * @return The XML String containing an index to all Users.
146       */
147      public abstract String getUserIndex();
148      
149      /**
150       * Returns the User instance as XML string, or null if not found.
151       * @param email The user's email.
152       * @return The User XML string, or null.
153       */
154      public abstract String getUser(String email);
155      
156      /**
157       * Persists a User instance.  If a User with this name
158       * already exists in the storage system, it will be overwritten.
159       * @param user The user
160       * @param xmlUser The User marshalled into an XML String.  
161       * @param xmlUserRef The corresponding UserRef marshalled into an XML String
162       * @return True if the user was successfully inserted.
163       */
164      public abstract boolean storeUser(User user, String xmlUser, String xmlUserRef);
165      
166      /**
167       * Ensures that the User with the given email is no longer present in this manager.
168       * @param email The user's email address.
169       */
170      public abstract void deleteUser(String email);
171      
172      /**
173       * Returns the XML ProjectIndex for all Projects in this server.
174       * @return The XML String containing an index to all Projects.
175       */
176      public abstract String getProjectIndex();
177      
178      /**
179       * Returns the Project instance as XML string, or null if not found.
180       * @param owner The user who owns the project.
181       * @param projectName The name of the Project.
182       * @return The Project XML string, or null.
183       */
184      public abstract String getProject(User owner, String projectName);
185      
186      /**
187       * Persists a Project instance.  If a Project with this owner and name
188       * already exists in the storage system, it will be overwritten.
189       * @param project The Project.
190       * @param xmlProject The Project marshalled into an XML String.  
191       * @param xmlProjectRef The corresponding ProjectRef marshalled into an XML String
192       * @return True if the user was successfully inserted.
193       */
194      public abstract boolean storeProject(Project project, String xmlProject, String xmlProjectRef);
195      
196      /**
197       * Ensures that the Project with the given owner and projectName is no longer present in the db.
198       * @param owner The User who owns this project.
199       * @param projectName The name of the Project.
200       */
201      public abstract void deleteProject(User owner, String projectName);
202      
203      
204      /** Keeps a pointer to this Server for use in accessing the managers. */
205      protected Server server;
206      
207      /** Keep a pointer to the Logger. */
208      protected Logger logger;
209      
210      /**
211       * Constructs a new DbImplementation.
212       * @param server The server. 
213       */
214      public DbImplementation(Server server) {
215        this.server = server;
216        this.logger = server.getLogger();
217      }
218      
219      /**
220       * Returns the UserManager associated with this server.
221       * Since the DbManager is initialized before all other managers, we will simply 
222       * get these other Managers on demand and not cache them. 
223       * @return The User Manager. 
224       */
225      protected UserManager getUserManager() {
226        return (UserManager)server.getContext().getAttributes().get("UserManager");
227      }
228      
229      /**
230       * Returns the SensorDataManager associated with this server. 
231       * Since the DbManager is initialized before all other managers, we will simply 
232       * get these other Managers on demand and not cache them. 
233       * @return The Sensor Data Manager. 
234       */
235      protected SensorDataManager getSensorDataManager() {
236        return (SensorDataManager)server.getContext().getAttributes().get("SensorDataManager");
237      }
238      
239      /**
240       * Returns the SdtManager associated with this server. 
241       * Since the DbManager is initialized before all other managers, we will simply 
242       * get these other Managers on demand and not cache them. 
243       * @return The SdtManager. 
244       */
245      protected SdtManager getSdtManager() {
246        return (SdtManager)server.getContext().getAttributes().get("SdtManager");
247      }
248      
249      /**
250       * Returns the ProjectManager associated with this server. 
251       * Since the DbManager is initialized before all other managers, we will simply 
252       * get these other Managers on demand and not cache them. 
253       * @return The ProjectManager. 
254       */
255      protected ProjectManager getProjectManager() {
256        return (ProjectManager)server.getContext().getAttributes().get("ProjectManager");
257      }
258      
259    }
