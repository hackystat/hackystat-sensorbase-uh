032      /**
033       * Creates a new DbManager which manages access to the underlying persistency layer(s).
034       * Instantiates the underlying storage system to use. 
035       * @param server The Restlet server instance. 
036       */
037      public DbManager(Server server) {
038        //Defaults to: "org.hackystat.sensorbase.db.derby.DerbyImplementation"
039        String dbClassName = server.getServerProperties().get(DB_IMPL_KEY); 
040        Class<?> dbClass = null;
041        //First, try to find the class specified in the sensorbase.properties file (or the default) 
042        try {
043          dbClass = Class.forName(dbClassName);
044        }
045        catch (ClassNotFoundException e) {
046          String msg = "DB error instantiating " + dbClassName + ". Could not find this class.";
047          server.getLogger().warning(msg + "\n" + StackTrace.toString(e));
048          throw new IllegalArgumentException(e);
049        }
050        // Next, try to find a constructor that accepts a Server as its parameter. 
051        Class<?>[] constructorParam = {org.hackystat.sensorbase.server.Server.class};
052        Constructor<?> dbConstructor = null;
053        try {
054          dbConstructor = dbClass.getConstructor(constructorParam);
055        }
056        catch (Exception e) {
057          String msg = "DB error instantiating " + dbClassName + ". Could not find Constructor(server)";
058          server.getLogger().warning(msg + "\n" + StackTrace.toString(e));
059          throw new IllegalArgumentException(e);
060        }
061        // Next, try to create an instance of DbImplementation from the Constructor.
062        Object[] serverArg = {server};
063        try {
064          this.dbImpl = (DbImplementation) dbConstructor.newInstance(serverArg);
065        }
066        catch (Exception e) {
067          String msg = "DB error instantiating " + dbClassName + ". Could not create instance.";
068          server.getLogger().warning(msg + "\n" + StackTrace.toString(e));
069          throw new IllegalArgumentException(e);
070        }
071        //this.dbImpl = new DerbyImplementation(this.server);
072        this.dbImpl.initialize();
073      }
