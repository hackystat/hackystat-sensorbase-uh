065      /**
066       * Instantiates the Derby implementation.  Throws a Runtime exception if the Derby
067       * jar file cannot be found on the classpath.
068       * @param server The SensorBase server instance. 
069       */
070      public DerbyImplementation(Server server) {
071        super(server);
072        // Set the directory where the DB will be created and/or accessed.
073        // This must happen before loading the driver. 
074        String dbDir = server.getServerProperties().get(DB_DIR_KEY);
075        System.getProperties().put(derbySystemKey, dbDir);
076        // Try to load the derby driver. 
077        try {
078          Class.forName(driver); 
079        } 
080        catch (java.lang.ClassNotFoundException e) {
081          String msg = "Derby: Exception during DbManager initialization: Derby not on CLASSPATH.";
082          this.logger.warning(msg + "\n" + StackTrace.toString(e));
083          throw new RuntimeException(msg, e);
084        }
085      }

088      /** {@inheritDoc} */
089      @Override
090      public void initialize() {
091        try {
092          // Initialize the database table structure if necessary.
093          this.isFreshlyCreated = !isPreExisting();
094          String dbStatusMsg = (this.isFreshlyCreated) ? 
095              "Derby: uninitialized." : "Derby: previously initialized.";
096          this.logger.info(dbStatusMsg);
097          if (this.isFreshlyCreated) {
098            this.logger.info("Derby: creating DB in: " + System.getProperty(derbySystemKey));
099            createTables();
100          }
101        }
102        catch (Exception e) {
103          String msg = "Derby: Exception during DerbyImplementation initialization:";
104          this.logger.warning(msg + "\n" + StackTrace.toString(e));
105          throw new RuntimeException(msg, e);
106        }

110      /**
111       * Determine if the database has already been initialized with correct table definitions. 
112       * Table schemas are checked by seeing if a dummy insert on the table will work OK.
113       * @return True if the database exists and tables are set up correctly.
114       * @throws SQLException If problems occur accessing the database or the tables are set right. 
115       */
116      private boolean isPreExisting() throws SQLException {
117        Connection conn = null;
118        Statement s = null;
119        try {
120          conn = DriverManager.getConnection(connectionURL);
121          s = conn.createStatement();
122          s.execute(testSensorDataTableStatement);
123          s.execute(testSensorDataTypeTableStatement);
124          s.execute(testUserTableStatement);
125          s.execute(testProjectTableStatement);
126        }  
127        catch (SQLException e) {
128          String theError = (e).getSQLState();
129          if ("42X05".equals(theError)) {
130            // Database doesn't exist.
131            return false;
132          }  
133          else if ("42X14".equals(theError) || "42821".equals(theError))  {
134            // Incorrect table definition. 
135            throw e;   
136          } 
137          else { 
138            // Unknown SQLException
139            throw e; 
140          }
141        }
142        finally {
143          if (s != null) {
144            s.close();
145          }
146          if (conn != null) {
147            conn.close();
148          }
149        }
150        // If table exists will get -  WARNING 02000: No row was found 
151        return true;
152      }

163          s = conn.createStatement();

166          s.execute(createSensorDataTypeTableStatement);
167          s.execute(indexSensorDataTypeTableStatement);
168          s.execute(createUserTableStatement);

171          s.execute(indexProjectTableStatement);
172          s.close();
173        }
174        finally {
175          s.close();
176          conn.close();
177        }
178      }

217      /** {@inheritDoc} */
218     @Override
219      public boolean storeSensorData(SensorData data, String xmlSensorData, String xmlSensorDataRef) {
220        Connection conn = null;
221        PreparedStatement s = null;
222        try {
223          conn = DriverManager.getConnection(connectionURL);
224          s = conn.prepareStatement("INSERT INTO SensorData VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)");
225          // Order: Owner Tstamp Sdt Runtime Tool Resource XmlSensorData XmlSensorDataRef LastMod
226          s.setString(1, data.getOwner());
227          s.setTimestamp(2, Tstamp.makeTimestamp(data.getTimestamp()));
228          s.setString(3, data.getSensorDataType());
229          s.setTimestamp(4, Tstamp.makeTimestamp(data.getRuntime()));
230          s.setString(5, data.getTool());
231          s.setString(6, data.getResource());
232          s.setString(7, xmlSensorData);
233          s.setString(8, xmlSensorDataRef);
234          s.setTimestamp(9, new Timestamp(new Date().getTime()));
235          s.executeUpdate();
236          this.logger.fine("Derby: Inserted " + data.getOwner() + " " + data.getTimestamp());
237        }
238        catch (SQLException e) {
239          if (DUPLICATE_KEY.equals(e.getSQLState())) {
240            try {
241              // Do an update, not an insert.
242              s = conn.prepareStatement(
243                  "UPDATE SensorData SET "
244                  + " Sdt=?, Runtime=?, Tool=?, Resource=?, XmlSensorData=?, " 
245                  + " XmlSensorDataRef=?, LastMod=?"
246                  + " WHERE Owner=? AND Tstamp=?");
247              s.setString(1, data.getSensorDataType());
248              s.setTimestamp(2, Tstamp.makeTimestamp(data.getRuntime()));
249              s.setString(3, data.getTool());
250              s.setString(4, data.getResource());
251              s.setString(5, xmlSensorData);
252              s.setString(6, xmlSensorDataRef);
253              s.setTimestamp(7, new Timestamp(new Date().getTime()));
254              s.setString(8, data.getOwner());
255              s.setTimestamp(9, Tstamp.makeTimestamp(data.getTimestamp()));
256              s.executeUpdate();
257              this.logger.fine("Derby: Updated " + data.getOwner() + " " + data.getTimestamp());
258            }
259            catch (SQLException f) {
260              this.logger.info(derbyError + StackTrace.toString(f));
261            }
262          }
263        }
264        finally {
265          try {
266            s.close();
267            conn.close();
268          }
269          catch (SQLException e) {
270            this.logger.warning(errorClosingMsg + StackTrace.toString(e));
271          }
272        }
273        return true;
274      }

332      /** {@inheritDoc} */
333      @Override
334      public boolean hasSensorData(User user, XMLGregorianCalendar timestamp) {
335        Connection conn = null;
336        PreparedStatement s = null;
337        ResultSet rs = null;
338        boolean isFound = false;
339        try {
340          conn = DriverManager.getConnection(connectionURL);
341          // 
342          String statement = 
343            "SELECT XmlSensorDataRef FROM SensorData WHERE "
344            + ownerEquals + user.getEmail() + andClause 
345            + " Tstamp='" + Tstamp.makeTimestamp(timestamp) + "'";
346          server.getLogger().fine(executeQueryMsg + statement);
347          s = conn.prepareStatement(statement);
348          rs = s.executeQuery();
349          // If a record was retrieved, we'll enter the loop, otherwise we won't. 
350          while (rs.next()) {
351            isFound = true;
352          }
353        }
354        catch (SQLException e) {
355          this.logger.info("Derby: Error in hasSensorData()" + StackTrace.toString(e));
356        }
357        finally {
358          try {
359            rs.close();
360            s.close();
361            conn.close();
362          }
363          catch (SQLException e) {
364            this.logger.warning("Derby: Error closing the connection" + StackTrace.toString(e));
365          }
366        }
367        return isFound;
368      }

657      public boolean storeProject(Project project, String xmlProject, String xmlProjectRef) {
658        Connection conn = null;
659        PreparedStatement s = null;
660        try {
661          conn = DriverManager.getConnection(connectionURL);
662          s = conn.prepareStatement("INSERT INTO Project VALUES (?, ?, ?, ?, ?, ?, ?)");
663          // Order: Owner ProjectName StartTime EndTime XmlProject XmlProjectRef LastMod
664          s.setString(1, project.getOwner());
665          s.setString(2, project.getName());
666          s.setTimestamp(3, Tstamp.makeTimestamp(project.getStartTime()));
667          s.setTimestamp(4, Tstamp.makeTimestamp(project.getEndTime()));
668          s.setString(5, xmlProject);
669          s.setString(6, xmlProjectRef);
670          s.setTimestamp(7, Tstamp.makeTimestamp(project.getLastMod()));
671          s.executeUpdate();
672          this.logger.fine("Derby: Inserted " + project.getOwner() + " " + project.getName());
673        }
674        catch (SQLException e) {
675          if (DUPLICATE_KEY.equals(e.getSQLState())) {
676            try {
677              // Do an update, not an insert.
678              s = conn.prepareStatement(
679                  "UPDATE Project SET "
680                  + " StartTime=?, EndTime=?, XmlProject=?, " 
681                  + " XmlProjectRef=?, LastMod=?"
682                  + " WHERE Owner=? AND ProjectName=?");
683              s.setTimestamp(1, Tstamp.makeTimestamp(project.getStartTime()));
684              s.setTimestamp(2, Tstamp.makeTimestamp(project.getEndTime()));
685              s.setString(3, xmlProject);
686              s.setString(4, xmlProjectRef);
687              s.setTimestamp(5, Tstamp.makeTimestamp(project.getEndTime()));
688              s.setString(6, project.getOwner());
689              s.setString(7, project.getName());
690              s.executeUpdate();
691              this.logger.fine("Derby: Updated " + project.getOwner() + " " + project.getName());
692            }
693            catch (SQLException f) {
694              this.logger.info(derbyError + StackTrace.toString(f));
695            }
696          }
697        }
698        finally {
699          try {
700            s.close();
701            conn.close();
702          }
703          catch (SQLException e) {
704            this.logger.warning(errorClosingMsg + StackTrace.toString(e));
705          }
706        }
707        return true;
708      }

714       * @param resourceName The resource name, such as "Project". 
715       * @param statement The SQL Statement to be used to retrieve the resource references.

719        StringBuilder builder = new StringBuilder(512);
720        builder.append("<").append(resourceName).append("Index>");

722        Connection conn = null;
