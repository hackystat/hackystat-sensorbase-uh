065      /**
066       * Instantiates the Derby implementation.  Throws a Runtime exception if the Derby
067       * jar file cannot be found on the classpath.
068       * @param server The SensorBase server instance. 
069       */
070      public DerbyImplementation(Server server) {
071        super(server);
072        // Set the directory where the DB will be created and/or accessed.
073        // This must happen before loading the driver. 
074        String dbDir = server.getServerProperties().get(DB_DIR_KEY);
075        System.getProperties().put(derbySystemKey, dbDir);
076        // Try to load the derby driver. 
077        try {
078          Class.forName(driver); 
079        } 
080        catch (java.lang.ClassNotFoundException e) {
081          String msg = "Derby: Exception during DbManager initialization: Derby not on CLASSPATH.";
082          this.logger.warning(msg + "\n" + StackTrace.toString(e));
083          throw new RuntimeException(msg, e);
084        }
085      }

088      /** {@inheritDoc} */
089      @Override
090      public void initialize() {
091        try {
092          // Create a shutdown hook that shuts down derby.
093          Runtime.getRuntime().addShutdownHook(new Thread() {
094            /** Run the shutdown hook for shutting down Derby. */
095            @Override 
096            public void run() {
097              Connection conn = null;
098              try {
099                conn = DriverManager.getConnection("jdbc:derby:;shutdown=true");
100              }
101              catch (Exception e) {
102                System.out.println("Derby shutdown hook results: " + e.getMessage());
103              }
104              finally {
105                try {
106                  conn.close();
107                }
108                catch (Exception e) { //NOPMD
109                  // we tried.
110                }
111              }
112            }
113          });
114          // Initialize the database table structure if necessary.
115          this.isFreshlyCreated = !isPreExisting();
116          String dbStatusMsg = (this.isFreshlyCreated) ? 
117              "Derby: uninitialized." : "Derby: previously initialized.";
118          this.logger.info(dbStatusMsg);
119          if (this.isFreshlyCreated) {
120            this.logger.info("Derby: creating DB in: " + System.getProperty(derbySystemKey));
121            createTables();
122          }
123        }
124        catch (Exception e) {
125          String msg = "Derby: Exception during DerbyImplementation initialization:";
126          this.logger.warning(msg + "\n" + StackTrace.toString(e));
127          throw new RuntimeException(msg, e);
128        }
129    
130      }

132      /**
133       * Determine if the database has already been initialized with correct table definitions. 
134       * Table schemas are checked by seeing if a dummy insert on the table will work OK.
135       * @return True if the database exists and tables are set up correctly.
136       * @throws SQLException If problems occur accessing the database or the tables are set right. 
137       */
138      private boolean isPreExisting() throws SQLException {
139        Connection conn = null;
140        Statement s = null;
141        try {
142          conn = DriverManager.getConnection(connectionURL);
143          s = conn.createStatement();
144          s.execute(testSensorDataTableStatement);
145          s.execute(testSensorDataTypeTableStatement);
146          s.execute(testUserTableStatement);
147          s.execute(testProjectTableStatement);
148        }  
149        catch (SQLException e) {
150          String theError = (e).getSQLState();
151          if ("42X05".equals(theError)) {
152            // Database doesn't exist.
153            return false;
154          }  
155          else if ("42X14".equals(theError) || "42821".equals(theError))  {
156            // Incorrect table definition. 
157            throw e;   
158          } 
159          else { 
160            // Unknown SQLException
161            throw e; 
162          }
163        }
164        finally {
165          if (s != null) {
166            s.close();
167          }
168          if (conn != null) {
169            conn.close();
170          }
171        }
172        // If table exists will get -  WARNING 02000: No row was found 
173        return true;
174      }

194          s.close();
195        }
196        finally {
197          s.close();
198          conn.close();
199        }

239      /** {@inheritDoc} */
240     @Override
241      public boolean storeSensorData(SensorData data, String xmlSensorData, String xmlSensorDataRef) {
242        Connection conn = null;
243        PreparedStatement s = null;
244        try {
245          conn = DriverManager.getConnection(connectionURL);
246          s = conn.prepareStatement("INSERT INTO SensorData VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)");
247          // Order: Owner Tstamp Sdt Runtime Tool Resource XmlSensorData XmlSensorDataRef LastMod
248          s.setString(1, data.getOwner());
249          s.setTimestamp(2, Tstamp.makeTimestamp(data.getTimestamp()));
250          s.setString(3, data.getSensorDataType());
251          s.setTimestamp(4, Tstamp.makeTimestamp(data.getRuntime()));
252          s.setString(5, data.getTool());
253          s.setString(6, data.getResource());
254          s.setString(7, xmlSensorData);
255          s.setString(8, xmlSensorDataRef);
256          s.setTimestamp(9, new Timestamp(new Date().getTime()));
257          s.executeUpdate();
258          this.logger.fine("Derby: Inserted " + data.getOwner() + " " + data.getTimestamp());
259        }
260        catch (SQLException e) {
261          if (DUPLICATE_KEY.equals(e.getSQLState())) {
262            try {
263              // Do an update, not an insert.
264              s = conn.prepareStatement(
265                  "UPDATE SensorData SET "
266                  + " Sdt=?, Runtime=?, Tool=?, Resource=?, XmlSensorData=?, " 
267                  + " XmlSensorDataRef=?, LastMod=?"
268                  + " WHERE Owner=? AND Tstamp=?");
269              s.setString(1, data.getSensorDataType());
270              s.setTimestamp(2, Tstamp.makeTimestamp(data.getRuntime()));
271              s.setString(3, data.getTool());
272              s.setString(4, data.getResource());
273              s.setString(5, xmlSensorData);
274              s.setString(6, xmlSensorDataRef);
275              s.setTimestamp(7, new Timestamp(new Date().getTime()));
276              s.setString(8, data.getOwner());
277              s.setTimestamp(9, Tstamp.makeTimestamp(data.getTimestamp()));
278              s.executeUpdate();
279              this.logger.fine("Derby: Updated " + data.getOwner() + " " + data.getTimestamp());
280            }
281            catch (SQLException f) {
282              this.logger.info(derbyError + StackTrace.toString(f));
283            }
284          }
285        }
286        finally {
287          try {
288            s.close();
289            conn.close();
290          }
291          catch (SQLException e) {
292            this.logger.warning(errorClosingMsg + StackTrace.toString(e));
293          }
294        }
295        return true;
296      }

367      /** {@inheritDoc} */
368      @Override
369      public boolean hasSensorData(User user, XMLGregorianCalendar timestamp) {
370        Connection conn = null;
371        PreparedStatement s = null;
372        ResultSet rs = null;
373        boolean isFound = false;
374        try {
375          conn = DriverManager.getConnection(connectionURL);
376          // 
377          String statement = 
378            "SELECT XmlSensorDataRef FROM SensorData WHERE "
379            + ownerEquals + user.getEmail() + andClause 
380            + " Tstamp='" + Tstamp.makeTimestamp(timestamp) + "'";
381          server.getLogger().fine(executeQueryMsg + statement);
382          s = conn.prepareStatement(statement);
383          rs = s.executeQuery();
384          // If a record was retrieved, we'll enter the loop, otherwise we won't. 
385          while (rs.next()) {
386            isFound = true;
387          }
388        }
389        catch (SQLException e) {
390          this.logger.info("Derby: Error in hasSensorData()" + StackTrace.toString(e));
391        }
392        finally {
393          try {
394            rs.close();
395            s.close();
396            conn.close();
397          }
398          catch (SQLException e) {
399            this.logger.warning("Derby: Error closing the connection" + StackTrace.toString(e));
400          }
401        }
402        return isFound;
403      }

695        try {
696          conn = DriverManager.getConnection(connectionURL);
697          s = conn.prepareStatement("INSERT INTO Project VALUES (?, ?, ?, ?, ?, ?, ?)");
698          // Order: Owner ProjectName StartTime EndTime XmlProject XmlProjectRef LastMod
699          s.setString(1, project.getOwner());
700          s.setString(2, project.getName());
701          s.setTimestamp(3, Tstamp.makeTimestamp(project.getStartTime()));
702          s.setTimestamp(4, Tstamp.makeTimestamp(project.getEndTime()));
703          s.setString(5, xmlProject);
704          s.setString(6, xmlProjectRef);
705          s.setTimestamp(7, Tstamp.makeTimestamp(project.getLastMod()));
706          s.executeUpdate();
707          this.logger.fine("Derby: Inserted " + project.getOwner() + " " + project.getName());
708        }
709        catch (SQLException e) {
710          if (DUPLICATE_KEY.equals(e.getSQLState())) {
711            try {
712              // Do an update, not an insert.
713              s = conn.prepareStatement(
714                  "UPDATE Project SET "
715                  + " StartTime=?, EndTime=?, XmlProject=?, " 
716                  + " XmlProjectRef=?, LastMod=?"
717                  + " WHERE Owner=? AND ProjectName=?");
718              s.setTimestamp(1, Tstamp.makeTimestamp(project.getStartTime()));
719              s.setTimestamp(2, Tstamp.makeTimestamp(project.getEndTime()));
720              s.setString(3, xmlProject);
721              s.setString(4, xmlProjectRef);
722              s.setTimestamp(5, Tstamp.makeTimestamp(project.getEndTime()));
723              s.setString(6, project.getOwner());
724              s.setString(7, project.getName());
725              s.executeUpdate();
726              this.logger.fine("Derby: Updated " + project.getOwner() + " " + project.getName());
727            }
728            catch (SQLException f) {
729              this.logger.info(derbyError + StackTrace.toString(f));
730            }
731          }
732        }
733        finally {
734          try {
735            s.close();
736            conn.close();
737          }
738          catch (SQLException e) {
739            this.logger.warning(errorClosingMsg + StackTrace.toString(e));
740          }
741        }
742        return true;
743      }
